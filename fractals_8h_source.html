45 bool CreateFractal_NormalScale(Grid& grid, HangingNodeRefiner_Grid& href,
46  number scaleFac, size_t numIterations,
47  TAPosition& aPosVRT)
48 {
49  if(!grid.has_vertex_attachment(aPosVRT)){
50  UG_LOG("WARNING in CreateFractal_NormalScale: given position attachment is ");
51  UG_LOG("not attached to the grids vertices. Aborting...\n");
52  return false;
53  }
54 
55 // store the old refinement callback of href
56  IRefinementCallback* oldCallback = href.get_refinement_callback();
57 
58 // create the new one.
59  FractalProjector refCallback(grid, scaleFac);
60  href.set_refinement_callback(&refCallback);
61 
62 // iterate for the specified number of times
63  for(size_t i = 0; i < numIterations; ++i){
64 
65  if(grid.num_volumes() > 0){
66  // iterate over all faces and mark them for refinement, if they are boundary faces.
67  for(FaceIterator iter = grid.faces_begin();
68  iter != grid.faces_end(); ++iter)
69  {
70  if(IsVolumeBoundaryFace(grid, *iter))
71  href.mark(*iter);
72  }
73  }
74  else if(grid.num_faces() > 0){
75  // markall faces
76  href.mark(grid.faces_begin(), grid.faces_end());
77  }
78  else{
79  // mark all edges
80  href.mark(grid.edges_begin(), grid.edges_end());
81  }
82 
83  // refine them
84  href.refine();
85 
86  // change the scalefac
87  refCallback.set_scale_fac(-0.5 * refCallback.get_scale_fac());
88  //refCallback.set_scale_fac(-0.5 * refCallback.get_scale_fac());
89  //refCallback.set_scale_fac(refCallback.get_scale_fac() * refCallback.get_scale_fac());
90 
91  }
92 
93 // done. restore href
94  href.set_refinement_callback(oldCallback);
95  return true;
96 }
