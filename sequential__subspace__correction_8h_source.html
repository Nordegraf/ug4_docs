100 template <typename TDomain, typename TAlgebra, typename TGroupObj>
101 class LocalIndexSubspace : public ILocalSubspace<TDomain, TAlgebra, TGroupObj>
102 {
103 public:
105  typedef TAlgebra algebra_type;
106 
108  typedef typename TAlgebra::vector_type vector_type;
109 
111  typedef typename TAlgebra::matrix_type matrix_type;
112 
114  typedef std::vector<size_t> index_vector;
115  typedef DenseVector< VariableArray1<number> > vector_type_local;
116  typedef DenseMatrix< VariableArray2<number> > matrix_type_local;
117 
119  virtual ~LocalIndexSubspace<TDomain, TAlgebra, TGroupObj>(){}
120 
122  virtual bool preprocess(const vector_type &c) {return true;}
123 
125  virtual void init(TGroupObj*, const vector_type &) = 0;
126 
128  virtual void extract_matrix(const matrix_type &A)
129  {
130  typedef typename TAlgebra::matrix_type::const_row_iterator const_row_iterator;
131  const static int blockSize = TAlgebra::blockSize;
132  const size_t numIndex = this->size();
133 
134  // fill local block matrix
135  bool bFound;
136  m_Aloc.resize(numIndex, numIndex);
137  m_Aloc = 0.0;
138  for (size_t j = 0; j<numIndex; j++)
139  {
140  for (size_t k=0;k<numIndex;k++)
141  {
142  const_row_iterator it = A.get_connection(m_vInd[j],m_vInd[k], bFound);
143  if(bFound){
144  m_Aloc.subassign(j*blockSize,k*blockSize,it.value());
145  }
146  }
147  }
148  }
149 
150 
152  virtual void extract_rhs(const vector_type &d)
153  {
154  const size_t numIndex = this->size();
155  const static int blockSize = TAlgebra::blockSize;
156 
157  // compute s[j] := d[j] - sum_k A(j,k)*c[k]
158  // note: the loop over k is the whole matrix row (not only selected indices)
159  m_dloc.resize(numIndex);
160  for (size_t j = 0; j<numIndex; j++)
161  {
162  typename TAlgebra::vector_type::value_type sumj = d[m_vInd[j]];
163  m_dloc.subassign(j*blockSize,sumj);
164  }
165  }
166 
168  virtual void extract_rhs(const vector_type &d, const matrix_type &A, const vector_type &c)
169  {
170  typedef typename TAlgebra::matrix_type::const_row_iterator const_row_iterator;
171  const static int blockSize = TAlgebra::blockSize;
172  const size_t numIndex = this->size();
173 
174  // compute s[j] := d[j] - sum_k A(j,k)*c[k]
175  // note: the loop over k is the whole matrix row (not only selected indices)
176  m_dloc.resize(numIndex);
177  for (size_t j = 0; j<numIndex; j++)
178  {
179  typename TAlgebra::vector_type::value_type sumj = d[m_vInd[j]];
180  for(const_row_iterator it = A.begin_row(m_vInd[j]); it != A.end_row(m_vInd[j]) ; ++it)
181  {
182  MatMultAdd(sumj, 1.0, sumj, -1.0, it.value(), c[it.index()]);
183  };
184 
185  m_dloc.subassign(j*blockSize,sumj);
186 
187  };
188  };
189 
191  virtual void update_solution(vector_type &u, double omega=1.0)
192  {
193  const size_t numIndex = this->size();
194 
195  // solve block
196  m_uloc.resize(numIndex);
197  InverseMatMult(m_uloc, 1.0, m_Aloc, m_dloc);
198  for (size_t j=0;j<numIndex;j++)
199  {
200  u[m_vInd[j]] += omega*m_uloc[j];
201  }
202 
203  }
204 
205 
206  virtual size_t size() { return m_vInd.size(); }
207 
208 protected:
209 
211  index_vector m_vInd;
212 
214  vector_type_local m_uloc;
215  vector_type_local m_dloc;
216  matrix_type_local m_Aloc;
217 
218 };

549 template<typename TGroupObj, typename TDomain, typename TAlgebra>
550 void ParallelSubspaceCorrectionLoop(const typename TAlgebra::matrix_type& A,
551  GridFunction<TDomain, TAlgebra>& c,
552  const typename TAlgebra::vector_type& d,
553  number omega_relax,
554  ILocalSubspace<TDomain, TAlgebra, TGroupObj> &subspace,
555  typename GridFunction<TDomain, TAlgebra>::template traits<TGroupObj>::const_iterator objIterBegin,
556  typename GridFunction<TDomain, TAlgebra>::template traits<TGroupObj>::const_iterator objIterEnd
557 )
558 {
559  // Loop over all grouping objects.
560  typedef typename GridFunction<TDomain, TAlgebra>::template traits<TGroupObj>::const_iterator GroupObjIter;
561  for(GroupObjIter iter = objIterBegin; iter != objIterEnd; ++iter)
562  {
563  // Apply subspace correction (w.r.t. obj.)
564  TGroupObj* groupObj = *iter;
565  subspace.init(groupObj, c);
566  subspace.extract_matrix(A);
567  subspace.extract_rhs(d); // w/o updates in c => parallel
568  subspace.update_solution(c, omega_relax);
569  }
570 }
