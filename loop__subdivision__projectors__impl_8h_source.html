78 
79  if(is_crease_vertex(parent)){

98  vector3 w = subdiv.ref_even_crease_weights();
99 
100  VecScaleAdd(m_aaTargetPos[vrt], w.x(), aaPos[parent],
101  w.y(), p0, w.z(), p1);
102  }
103  else{
104  m_aaTargetPos[vrt] = aaPos[parent];

215 void SubdivisionLoopProjector<TAPosition>::
216 new_vertex(Vertex* vrt, Vertex* parent)
217 {
218  SubdivRules_PLoop& subdiv = SubdivRules_PLoop::inst();
219  Grid::VertexAttachmentAccessor<TAPosition>& aaPos = BaseClass::m_aaPos;
220 
221  assert(aaPos.valid() && "make sure to initialise the refiner-callback correctly.");
222  assert(m_aaTargetPos.valid() && "make sure to initialise the refiner-callback correctly.");
223 
224 // check whether the vertex lies inside a volume geometry. If it does,
225 // perform linear refinement.
226  Grid& g = *BaseClass::m_pGrid;
227  bool volumesExist = g.num<Volume>() > 0;
228 
229  if(volumesExist){m_aaTargetPos[vrt] = CalculateCenter(parent, BaseClass::m_aaPos);
230  if(!IsBoundaryVertex3D(g, parent)){
231  aaPos[vrt] = aaPos[parent];
232  return;
233  }
234  }
235 
236  if(is_crease_vertex(parent)){
237  // get the neighboured crease edges
238  Edge* nbrs[2];
239  size_t numNbrs = 0;
240  for(Grid::AssociatedEdgeIterator iter =
241  g.associated_edges_begin(parent);
242  iter != g.associated_edges_end(parent); ++iter)
243  {
244  if(is_crease_edge(*iter)){
245  nbrs[numNbrs] = *iter;
246  ++numNbrs;
247  if(numNbrs == 2)
248  break;
249  }
250  }
251 
252  if(numNbrs == 2){
253  pos_type& p0 = aaPos[GetConnectedVertex(nbrs[0], parent)];
254  pos_type& p1 = aaPos[GetConnectedVertex(nbrs[1], parent)];
255  vector3 w = subdiv.ref_even_crease_weights();
256 
257  VecScaleAdd(m_aaTargetPos[vrt], w.x(), aaPos[parent],
258  w.y(), p0, w.z(), p1);
259  }
260  else{
261  m_aaTargetPos[vrt] = aaPos[vrt];
262  }
263  }
264  else{
265  // perform loop subdivision on even vertices
266  // first get neighboured vertices
267  //todo: replace this by a method
268  size_t valence = 0;
269  pos_type p;
270  VecSet(p, 0);
271 
272  for(Grid::AssociatedEdgeIterator iter =
273  g.associated_edges_begin(parent);
274  iter != g.associated_edges_end(parent); ++iter)
275  {
276  if((!volumesExist) || IsBoundaryEdge3D(g, *iter)){
277  VecAdd(p, p, aaPos[GetConnectedVertex(*iter, parent)]);
278  ++valence;
279  }
280  }
281 
282  number centerWgt = subdiv.ref_even_inner_center_weight(valence);
283  number nbrWgt = subdiv.ref_even_inner_nbr_weight(valence);
284 
285  VecScaleAdd(m_aaTargetPos[vrt],
286  centerWgt, aaPos[parent],
287  nbrWgt, p);
288 /*
289  number beta = get_beta(valence);
290 
291  VecScaleAdd(m_aaPos[vrt], beta, p,
292  1.0 - (number)valence * beta, m_aaPos[parent]);
293 */
294 
295  }

298 template <class TAPosition>
299 void SubdivisionLoopProjector<TAPosition>::
300 new_vertex(Vertex* vrt, Edge* parent)
301 {
302  using namespace std;
303  using std::swap;
304 
305  SubdivRules_PLoop& subdiv = SubdivRules_PLoop::inst();
306  Grid::VertexAttachmentAccessor<TAPosition>& aaPos = BaseClass::m_aaPos;
307 
308  assert(aaPos.valid() && "make sure to initialise the refiner-callback correctly.");
309  assert(m_aaTargetPos.valid() && "make sure to initialise the refiner-callback correctly.");
310 
311 // check whether the parent edge lies inside a volume geometry. If it does,
312 // perform linear refinement.
313  Grid& g = *BaseClass::m_pGrid;
314  if(g.num<Volume>() > 0){
315  if(!IsBoundaryEdge3D(g, parent)){
316  m_aaTargetPos[vrt] = CalculateCenter(parent, aaPos);
317  return;
318  }
319  }
320 
321  if(is_crease_edge(parent)){
322  vector2 wghts = subdiv.ref_odd_crease_weights();
323  VecScaleAdd(m_aaTargetPos[vrt], wghts.x(), aaPos[parent->vertex(0)],
324  wghts.y(), aaPos[parent->vertex(1)]);
325  }
326  else{
327  // apply loop-subdivision on inner elements
328  // get the neighboured triangles
329  Face* f[2];
330  int numAssociatedBndFaces = 0;
331  if(g.num<Volume>() > 0){
332  vector<Face*> faces;
333  CollectAssociated(faces, g, parent);
334  for(size_t i = 0; i < faces.size(); ++i){
335  if(IsBoundaryFace3D(g, faces[i])){
336  if(numAssociatedBndFaces < 2){
337  f[numAssociatedBndFaces] = faces[i];
338  }
339  ++numAssociatedBndFaces;
340  }
341  }
342  }
343  else{
344  numAssociatedBndFaces = GetAssociatedFaces(f, g, parent, 2);
345  }
346  if(numAssociatedBndFaces == 2){
347  if(f[0]->num_vertices() == 3 && f[1]->num_vertices() == 3){
348  // the 4 vertices that are important for the calculation
349  Vertex* v[4];
350  v[0] = parent->vertex(0); v[1] = parent->vertex(1);
351  v[2] = GetConnectedVertex(parent, f[0]);
352  v[3] = GetConnectedVertex(parent, f[1]);
353 
354  vector4 wghts;
355 
356  bool isCrease0 = is_crease_vertex(v[0]);
357  bool isCrease1 = is_crease_vertex(v[1]);
358  // if exactly one of the two is a crease vertex, special
359  // weighting has to be performed
360  //todo: this does not yet work for inner creases.
361  if((isCrease0 && !isCrease1) || (!isCrease0 && isCrease1))
362  {
363  // the crease vertex has to be in v[0]
364  if(isCrease1)
365  swap(v[0], v[1]);
366 
367  // todo: replace this with a method call
368  // get the number of edges that are connected to v[0]
369  // todo: only check edges that are on the correct side of the crease.
370  size_t valence = 0;
371  for(Grid::AssociatedEdgeIterator iter =
372  g.associated_edges_begin(v[0]);
373  iter != g.associated_edges_end(v[0]); ++iter)
374  {
375  ++valence;
376  }
377 
378  wghts = subdiv.ref_odd_inner_weights(valence);
379  }
380  else{
381  wghts = subdiv.ref_odd_inner_weights();
382  }
383 
384  VecScaleAdd(m_aaTargetPos[vrt],
385  wghts.x(), aaPos[v[0]], wghts.y(), aaPos[v[1]],
386  wghts.z(), aaPos[v[2]], wghts.w(), aaPos[v[3]]);
387 
388  }
389  else
390  m_aaTargetPos[vrt] = CalculateCenter(parent, aaPos);
391  }
392  else
393  m_aaTargetPos[vrt] = CalculateCenter(parent, aaPos);
394  }
395 }
