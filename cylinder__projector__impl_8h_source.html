113 template <class TAPosition>
114 template <class TElem>
115 void CylinderProjector<TAPosition>::
116 perform_projection(Vertex* vrt, TElem* parent)
117 {
118  assert(m_aaPos.valid() && "make sure to initialise the refiner-callback correctly.");
119 
120 // calculate the new position by linear interpolation and project that point
121 // onto the cylinder.
122  typename TElem::ConstVertexArray vrts = parent->vertices();
123  size_t numVrts = parent->num_vertices();
124  number avDist = 0;
125  pos_type parentCenter;
126  VecSet(parentCenter, 0);
127 
128  for(size_t i = 0; i < numVrts; ++i){
129  const pos_type& p = m_aaPos[vrts[i]];
130  avDist += DistancePointToRay(p, m_center, m_axis);
131  VecAdd(parentCenter, parentCenter, p);
132  }
133 
134  avDist /= (number)numVrts;
135  VecScale(parentCenter, parentCenter, 1. / (number)numVrts);
136 
137  pos_type proj, v;
138  ProjectPointToRay(proj, parentCenter, m_center, m_axis);
139  VecSubtract(v, parentCenter, proj);
140  number len = VecLength(v);
141  if(len > SMALL * avDist){ // if avDist is very small, len may be small, too
142  VecScale(v, v, avDist/len);
143  VecAdd(m_aaPos[vrt], proj, v);
144  }
145  else
146  m_aaPos[vrt] = parentCenter;
147 }
