109 template <class TAPosition>
110 template <class TElem>
111 void SphericalFalloffProjector<TAPosition>::
112 perform_projection(Vertex* vrt, TElem* parent)
113 {
114  assert(m_aaPos.valid() && "make sure to initialise the refiner-callback correctly.");
115 // first calculate the average distance of corners of parent
116  typename TElem::ConstVertexArray vrts = parent->vertices();
117  size_t numVrts = parent->num_vertices();
118  number avDist = 0;
119  pos_type parentCenter;
120  VecSet(parentCenter, 0);
121 
122  for(size_t i = 0; i < numVrts; ++i){
123  const pos_type& p = m_aaPos[vrts[i]];
124  avDist += VecDistance(p, m_center);
125  VecAdd(parentCenter, parentCenter, p);
126  }
127 
128  avDist /= (number)numVrts;
129  VecScale(parentCenter, parentCenter, 1. / (number)numVrts);
130 
131  number ia = (avDist - m_innerRadius) / (m_outerRadius - m_innerRadius);
132 
133  if(ia > 1)
134  m_aaPos[vrt] = parentCenter;
135  else{
136  // calculate cylindrical projection
137  pos_type sphereProj;
138  VecSubtract(sphereProj, parentCenter, m_center);
139  number len = VecLength(sphereProj);
140  if(len > SMALL * avDist){ // if avDist is very small, len may be small, too
141  VecScale(sphereProj, sphereProj, avDist / len);
142  VecAdd(sphereProj, sphereProj, m_center);
143 
144  if(ia <= 0)
145  m_aaPos[vrt] = sphereProj;
146  else
147  VecScaleAdd(m_aaPos[vrt], 1.-ia, sphereProj, ia, parentCenter);
148  }
149  else
150  m_aaPos[vrt] = parentCenter;
151  }
152 }
