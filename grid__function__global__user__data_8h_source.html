148  inline bool evaluate(number& value, const MathVector<dim>& x) const
149  {
150 
151  element_t* elem = NULL;
152  try{
153 
154  if(!FindContainingElement(elem, m_tree, x)){
155  return false;
156  }
157 
158  // get corners of element
159  std::vector<MathVector<dim> > vCornerCoords;
160  CollectCornerCoordinates(vCornerCoords, *elem, *m_spGridFct->domain());
161 
162  // reference object id
163  const ReferenceObjectID roid = elem->reference_object_id();
164 
165  // get local position of DoF
166  DimReferenceMapping<dim, dim>& map
167  = ReferenceMappingProvider::get<dim, dim>(roid, vCornerCoords);
168  MathVector<dim> locPos;
169  VecSet(locPos, 0.5);
170  map.global_to_local(locPos, x);
171 
172  // evaluate at shapes at ip
173  const LocalShapeFunctionSet<dim>& rTrialSpace =
174  LocalFiniteElementProvider::get<dim>(roid, m_lfeID);
175  std::vector<number> vShape;
176  rTrialSpace.shapes(vShape, locPos);
177 
178  // get multiindices of element
179  std::vector<DoFIndex> ind;
180  m_spGridFct->dof_indices(elem, m_fct, ind);
181 
182  // compute solution at integration point
183  value = 0.0;
184  for(size_t sh = 0; sh < vShape.size(); ++sh)
185  {
186  const number valSH = DoFRef(*m_spGridFct, ind[sh]);
187  value += valSH * vShape[sh];
188  }
189 
190  // point is found
191  return true;
192  }
193  UG_CATCH_THROW("GlobalGridFunctionNumberData: Evaluation failed."
194  << "Point: " << x << ", Element: "
195  << ElementDebugInfo(*m_spGridFct->domain()->grid(), elem));
196  }

199  inline void evaluate_global(number& value, const MathVector<dim>& x) const
200  {
201  // evaluate at this proc
202  bool bFound = this->evaluate(value, x);
203 
204 #ifdef UG_PARALLEL
205  // share value between all procs
206  pcl::ProcessCommunicator com;
207  int numFound = (bFound ? 1 : 0);
208  numFound = com.allreduce(numFound, PCL_RO_SUM);
209 
210  // get overall value
211  if(!bFound) value = 0.0;
212  number globValue = com.allreduce(value, PCL_RO_SUM) / numFound;
213 
214  if(numFound == 0)
215  UG_THROW("Point "<<x<<" not found on all "<<pcl::NumProcs()<<" procs.");
216 
217  // check correctness for continuous spaces
218  // note: if the point is found more than one it is located on the
219  // boundary of some element. thus, if the space is continuous, those
220  // values should match on all procs.
221  if(bFound)
222  if(LocalFiniteElementProvider::continuous(m_lfeID)){
223  if( fabs(value) > 1e-10 && fabs((globValue - value) / value) > 1e-8)
224  UG_THROW("Global mean "<<globValue<<" != local value "<<value);
225  }
226 
227  // set as global value
228  value = globValue;
229  bFound = true;
230 #endif
231 
232  if(!bFound)
233  UG_THROW("Couldn't find an element containing the specified point: " << x);
234  }
235 
236  // evaluates at given position
237  number evaluate(std::vector<number> vPos)
238  {
239  if((int)vPos.size() != dim)
240  UG_THROW("Expected "<<dim<<" components, but given "<<vPos.size());
241 
242  MathVector<dim> x;
243  for(int i = 0; i < dim; i++) x[i] = vPos[i];
244 
245  number value;
246  evaluate_global(value, x);
247 
248  return value;
249  }
