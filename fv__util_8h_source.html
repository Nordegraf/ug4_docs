60  vOut = vCornerCoords[0];
61  for(size_t j = 1; j < num; ++j)

251 struct fv1_traits_ReferenceFace3d : public fv1_traits_ReferenceFace
252 {
253  template <typename TElem>
254  static void NormalOnSCVF_Face(MathVector<3>& outNormal,
255  const MathVector<3>* vSCVFCorner,
256  const MathVector<3>* vElemCorner)
257  {
258  // compute Normal to Face (right-handed)
259  MathVector<3> ElemNormal;
260  ElementNormal<TElem,3>(ElemNormal, vElemCorner);
261 
262  // compute a Point such that the triangle given by vSCVFCorner and p
263  // has the following property:
264  // a) The new Triangle is normal to the ElementFace
265  // b) The new Triangle contains the scvf
266  MathVector<3> p;
267  VecAdd(p, vSCVFCorner[0], ElemNormal);
268 
269  MathVector<3> vNewTriangleCorner[3];
270  vNewTriangleCorner[0] = vSCVFCorner[0];
271  vNewTriangleCorner[1] = vSCVFCorner[1];
272  vNewTriangleCorner[2] = p;
273 
274  // now compute Normal to new triangle. This Normal will be normal to
275  // the scvf and within the original element
276  ElementNormal<ReferenceTriangle, 3>(outNormal, vNewTriangleCorner);
277 
278  // scale to normal to the size of the scvf
279  const number size = VecDistance(vSCVFCorner[0], vSCVFCorner[1]);
280  VecNormalize(outNormal, outNormal);
281  VecScale(outNormal, outNormal, size);
282  }
283  static void NormalOnSCVF(MathVector<3>& outNormal,
284  const MathVector<3>* vSCVFCorner,
285  const MathVector<3>* vElemCorner)
286  {
287  UG_THROW("not implemented.")
288  }
289  static void NormalOnBF(MathVector<3>& outNormal,
290  const MathVector<3>* vSCVFCorner,

549 {
551  static const int dim = TDim;
552 
554  typedef DimReferenceElement<dim> ref_elem_type;
555 
557  static void dim_get_num_SCV_and_SCVF
558  (
559  const ref_elem_type& refElem,
560  ReferenceObjectID roid,
561  size_t& numSCV,
562  size_t& numSCVF
563  )
564  {
565  if(roid != ROID_PYRAMID && roid != ROID_OCTAHEDRON)
566  {
567  numSCV = refElem.num(0);
568  numSCVF = refElem.num(1);
569  }
570  else if(dim == 3 && roid == ROID_PYRAMID)
571  {
572  UG_WARNING("Pyramid Finite Volume Geometry for 1st order currently "
573  "implemented in DimFV1Geom EXPERIMENTATLLY. Please contact "
574  "Martin Stepniewski or Andreas Vogel if you see this message.")
575 
576  numSCV = 4*refElem.num(1);
577  numSCVF = 2*refElem.num(1);
578  }
579  else if(dim == 3 && roid == ROID_OCTAHEDRON)
580  {
581  // Case octahedron
582  numSCV = 16;
583  numSCVF = 24;
584  }
585  else
586  UG_THROW ("fv1_dim_traits_base: Unsupported combination of dimension and reference element.");
587  }
588 
590  static void get_dim_scvf_from_to
591  (
592  const ref_elem_type& refElem,
593  ReferenceObjectID roid,
594  size_t i,
595  size_t& From,
596  size_t& To
597  )
598  {
599  if (roid != ROID_PYRAMID && roid != ROID_OCTAHEDRON)
600  {
601  From = refElem.id(1, i, 0, 0);
602  To = refElem.id(1, i, 0, 1);
603  }
604  // special case pyramid (scvf not mappable by edges)
605  else if (dim == 3 && roid == ROID_PYRAMID)
606  {
607  // map according to order defined in ComputeSCVFMidID
608  From = fv1_traits_ReferencePyramid::scvf_from_to ((const ReferencePyramid&) refElem, i, 0);
609  To = fv1_traits_ReferencePyramid::scvf_from_to ((const ReferencePyramid&) refElem, i, 1);
610  }
611  // special case octahedron (scvf not mappable by edges)
612  else if(dim == 3 && roid == ROID_OCTAHEDRON)
613  {
614  // map according to order defined in ComputeSCVFMidID
615  From = fv1_traits_ReferenceOctahedron::scvf_from_to ((const ReferenceOctahedron&) refElem, i, 0);
616  To = fv1_traits_ReferenceOctahedron::scvf_from_to ((const ReferenceOctahedron&) refElem, i, 1);
617  }
618  else
619  UG_THROW ("fv1_dim_traits_base: Unsupported combination of dimension and reference element.");
620  }
621 
623  static size_t dim_scv_node_id
624  (
625  const ref_elem_type& refElem,
626  ReferenceObjectID roid,
627  size_t i
628  )
629  {
630  // "classical" elements
631  if (roid != ROID_PYRAMID && roid != ROID_OCTAHEDRON)
632  return i;
633

640  return fv1_traits_ReferenceOctahedron::scv_node_id ((const ReferenceOctahedron&) refElem, i);

643  }
