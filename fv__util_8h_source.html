57 template <typename TPosition>
58 void AveragePositions(TPosition& vOut, const TPosition* vCornerCoords, size_t num)
59 {
60  vOut = vCornerCoords[0];
61  for(size_t j = 1; j < num; ++j)
62  {
63  vOut += vCornerCoords[j];
64  }
65  vOut *= 1./(number)num;
66 }

210 struct fv1_traits_ReferenceFace3d : public fv1_traits_ReferenceFace
211 {
212  template <typename TElem>
213  static void NormalOnSCVF_Face(MathVector<3>& outNormal,
214  const MathVector<3>* vSCVFCorner,
215  const MathVector<3>* vElemCorner)
216  {
217  // compute Normal to Face (right-handed)
218  MathVector<3> ElemNormal;
219  ElementNormal<TElem,3>(ElemNormal, vElemCorner);
220 
221  // compute a Point such that the triangle given by vSCVFCorner and p
222  // has the following property:
223  // a) The new Triangle is normal to the ElementFace
224  // b) The new Triangle contains the scvf
225  MathVector<3> p;
226  VecAdd(p, vSCVFCorner[0], ElemNormal);
227 
228  MathVector<3> vNewTriangleCorner[3];
229  vNewTriangleCorner[0] = vSCVFCorner[0];
230  vNewTriangleCorner[1] = vSCVFCorner[1];
231  vNewTriangleCorner[2] = p;
232 
233  // now compute Normal to new triangle. This Normal will be normal to
234  // the scvf and within the original element
235  ElementNormal<ReferenceTriangle, 3>(outNormal, vNewTriangleCorner);
236 
237  // scale to normal to the size of the scvf
238  const number size = VecDistance(vSCVFCorner[0], vSCVFCorner[1]);
239  VecNormalize(outNormal, outNormal);
240  VecScale(outNormal, outNormal, size);
241  }
242  static void NormalOnSCVF(MathVector<3>& outNormal,
243  const MathVector<3>* vSCVFCorner,
244  const MathVector<3>* vElemCorner)
245  {
246  UG_THROW("not implemented.")
247  }
248  static void NormalOnBF(MathVector<3>& outNormal,
249  const MathVector<3>* vSCVFCorner,
250  const MathVector<3>* vElemCorner)
251  {
252  UG_THROW("not implemented.")
253  }
254 };
