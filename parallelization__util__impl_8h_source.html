42 template <class TIndVec>
43 void GenerateGlobalConsecutiveIndices(TIndVec& indsOut, size_t numLocalInds,
44  const AlgebraLayouts& layouts)
45 {
46  typedef typename TIndVec::value_type index_t;
47 
48  PU_PROFILE_FUNC();
49  indsOut.clear();
50  indsOut.resize(numLocalInds, 0);
51 
52 // count local unique indices. Ignore vmasters and hslaves.
53 // we're using uint64 here to stay compatible with mpi
54  uint64 numLocalUniqueInds = 0;
55  const index_t invalidIndex = -1;
56  SetLayoutValues(&indsOut, layouts.slave(), invalidIndex);
57  SetLayoutValues(&indsOut, layouts.vertical_master(), invalidIndex);
58 
59  for(size_t i = 0; i < numLocalInds; ++i){
60  if(indsOut[i] != invalidIndex){
61  ++numLocalUniqueInds;
62  }
63  }
64 
65 // we'll now gather the numbers of local unique indices on proces 0 and create
66 // an offset map for each process. This offset map will then be scattered to
67 // the separate processes again.
68  const pcl::ProcessCommunicator& procCom = layouts.proc_comm();
69  std::vector<uint64> p0_uniqueIndNumbers(procCom.size());
70 
71  procCom.gather(&numLocalUniqueInds, 1, PCL_DT_UNSIGNED_LONG_LONG,
72  &p0_uniqueIndNumbers.front(), 1, PCL_DT_UNSIGNED_LONG_LONG, 0);
73 
74  if(procCom.get_local_proc_id() == 0){
75  // convert the number of unique indices of each process to offsets
76  uint64 offset = 0;
77  for(size_t i = 0; i < p0_uniqueIndNumbers.size(); ++i){
78  size_t newOffset = offset + p0_uniqueIndNumbers[i];
79  p0_uniqueIndNumbers[i] = offset;
80  offset = newOffset;
81  }
82  }
83 
84  uint64 indOffset = 0;
85  procCom.scatter(&p0_uniqueIndNumbers.front(), 1, PCL_DT_UNSIGNED_LONG_LONG,
86  &indOffset, 1, PCL_DT_UNSIGNED_LONG_LONG, 0);
87 
88 // using the local offset we can now assign the local unique indices to indsOut
89  index_t newInd = static_cast<index_t>(indOffset);
90  for(size_t i = 0; i < numLocalInds; ++i){
91  if(indsOut[i] != invalidIndex){
92  indsOut[i] = newInd;
93  ++newInd;
94  }
95  }
96 
97 // finally we have to copy the unique indices from hmasters to hslaves and
98 // afterwards from vslaves to vmasters.
99  pcl::InterfaceCommunicator<IndexLayout>& icom = layouts.comm();
100  ComPol_VecCopy<TIndVec> compolCopy(&indsOut);
101 
102  icom.send_data(layouts.master(), compolCopy);
103  icom.receive_data(layouts.slave(), compolCopy);
104  icom.communicate();
105 
106 // Todo: This method is mainly used on surface-layouts.
107 // Currently a bug in those layouts prevents the execution of the following code
108 // which would be required for level-matrices.
109 // Since ghosts (pure v-masters) are ignored in surface-vectors and matrices,
110 // but pure v-slaves are considered, v-master and v-slave interfaces do not
111 // match. Fortunately they are not required for surface-index-generation.
112 // The following block should however be used as soon as those layouts are fixed.
113  // icom.send_data(layouts.vertical_slave(), compolCopy);
114  // icom.receive_data(layouts.vertical_master(), compolCopy);
115  // icom.communicate();
116 }
