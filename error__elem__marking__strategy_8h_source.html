93 template<class TElem>
94 number CreateListOfElemWeights(
95  MultiGrid::AttachmentAccessor<TElem, ug::Attachment<number> > &aaError,
96  typename DoFDistribution::traits<TElem>::const_iterator iterBegin,
97  const typename DoFDistribution::traits<TElem>::const_iterator iterEnd,
98  std::vector<double> &eta)
99 {
100  number localErr=0;
101  typename DoFDistribution::traits<TElem>::const_iterator iter; // = iterBegin;
102  size_t i=0;
103  for (iter = iterBegin; iter != iterEnd; ++iter)
104  {
105  const double elemErr = aaError[*iter];
106 
107  // if no error value exists: ignore (might be newly added by refinement);
108  // newly added elements are supposed to have a negative error estimator
109  if (elemErr < 0) continue;
110 
111  eta[i++] = elemErr;
112  localErr += elemErr;
113  }
114 
115  // sort descending using default comparison
116  std::sort (eta.begin(), eta.end(), std::greater<double>());
117  return localErr;
118 };
