197 template<class TElem>
198 number CreateListOfElemWeights(
199  MultiGrid::AttachmentAccessor<TElem, ug::Attachment<number> > &aaError,
200  typename DoFDistribution::traits<TElem>::const_iterator iterBegin,
201  const typename DoFDistribution::traits<TElem>::const_iterator iterEnd,
202  std::vector<double> &eta)
203 {
204  number localErr=0;
205  typename DoFDistribution::traits<TElem>::const_iterator iter; // = iterBegin;
206  size_t i=0;
207  for (iter = iterBegin; iter != iterEnd; ++iter)
208  {
209  const double elemErr = aaError[*iter];
210 
211  // if no error value exists: ignore (might be newly added by refinement);
212  // newly added elements are supposed to have a negative error estimator
213  if (elemErr < 0) continue;
214 
215  eta[i++] = elemErr;
216  localErr += elemErr;
217  }
218 
219  // sort descending using default comparison
220  std::sort (eta.begin(), eta.end(), std::greater<double>());
221  return localErr;
222 };

245 void MaximumMarking<TDomain>::mark(typename base_type::elem_accessor_type& aaError,
246  IRefiner& refiner,
247  ConstSmartPtr<DoFDistribution> dd)
248 {
249  typedef typename base_type::elem_type TElem;
250  typedef typename DoFDistribution::traits<TElem>::const_iterator const_iterator;
251 
252  // compute minimal/maximal/ total error and number of elements
253 
254  number minElemErr, minElemErrLocal;
255  number maxElemErr, maxElemErrLocal;
256  number errTotal, errLocal;
257  size_t numElem, numElemLocal;
258 
259  ComputeMinMax(aaError, dd, minElemErr, maxElemErr, errTotal, numElem,
260  minElemErrLocal, maxElemErrLocal, errLocal, numElemLocal);
261 
262 
263  // init iterators
264  const_iterator iter;
265  const const_iterator iterEnd = dd->template end<TElem>();
266 
267  // determine (local) number of excess elements
268  const size_t ndiscard = (size_t) (numElemLocal*m_eps); // TODO: on every process?
269  UG_LOG(" +++ Found max "<< maxElemErr << " ndiscard="<<ndiscard<<".\n");
270 
271  // Verfuerths strategy for skipping excess
272 /* while (ndiscard)
273  {
274  // find elements that should be discarded
275  int discard = 0;
276  double newmax = 0.0;
277  for (iter = dd->template begin<TElem>(); iter != iterEnd; ++iter)
278  {
279  const double elemErr = aaError[*iter];
280 
281  if ( elemErr >= maxElemErr) { discard++; }
282  else { newmax = std::max(newmax, elemErr); }
283  }
284 
285  // quit, if enough elements have been found
286  if ( discard < ndiscard) maxElemErr = newmax;
287  else break;
288 
289  }*/
290 
291  if (numElemLocal > 0)
292  {
293  // create sorted array of elem weights
294  std::vector<double> eta;
295  eta.resize(numElemLocal);
296  CreateListOfElemWeights<TElem>(aaError,dd->template begin<TElem>(), iterEnd, eta);
297  UG_ASSERT(numElemLocal==eta.size(), "Huhh: number of elements does not match!");
298  UG_ASSERT(numElemLocal > ndiscard, "Huhh: number of elements does not match!");
299  maxElemErr = eta[ndiscard];
300 
301  }
302 
303  UG_LOG(" +++ MaximumMarking: Maximum for refinement: " << maxElemErr << " elements.\n");
304 
305  // compute parallel threshold
306 #ifdef UG_PARALLEL
307  if (pcl::NumProcs() > 1)
308  {
309  pcl::ProcessCommunicator com;
310  number maxElemErrLocal = maxElemErr;
311  maxElemErr = com.allreduce(maxElemErrLocal, PCL_RO_SUM);
312  UG_LOG(" +++ MaximumMarking: Maximum for refinement: " << maxElemErr << ".\n");
313  maxElemErr = maxElemErr / pcl::NumProcs();
314  UG_LOG(" +++ MaximumMarking: Average for refinement: " << maxElemErr << ".\n");
315  }
316 #else
317  UG_LOG(" +++ Skipping " << ndiscard << " elements; new max." << maxElemErr << ".\n");
318 #endif
319 
320  // refine all element above threshold
321  const number minErrToRefine = maxElemErr*m_theta;
322  UG_LOG(" +++ Refining elements if error greater " << maxElemErr << "*" << m_theta <<
323  " = " << minErrToRefine << ".\n");
324 
325  // reset counter
326  std::size_t numMarkedRefine = 0;
327 
328  // loop elements for marking
329  for (iter = dd->template begin<TElem>(); iter != iterEnd; ++iter)
330  {
331  // get element
332  TElem* elem = *iter;
333 
334  // if no error value exists: ignore (might be newly added by refinement);
335  // newly added elements are supposed to have a negative error estimator
336  if (aaError[elem] < 0) continue;
337 
338  // marks for refinement
339  if (aaError[elem] >= minErrToRefine)
340  if (dd->multi_grid()->get_level(elem) <= m_max_level)
341  {
342  refiner.mark(elem, RM_REFINE);
343  numMarkedRefine++;
344  }
345  }
346 
347 #ifdef UG_PARALLEL
348  if (pcl::NumProcs() > 1)
349  {
350  pcl::ProcessCommunicator com;
351  std::size_t numMarkedRefineLocal = numMarkedRefine;
352  numMarkedRefine = com.allreduce(numMarkedRefineLocal, PCL_RO_SUM);
353  UG_LOG(" +++ MaximumMarking: Marked for refinement: " << numMarkedRefine << " ("<< numMarkedRefineLocal << ") elements.\n");
354  }
355 #else
356  UG_LOG(" +++ MaximumMarking: Marked for refinement: " << numMarkedRefine << " elements.\n");

382 template <typename TDomain>
383 void VarianceMarking<TDomain>::mark(typename base_type::elem_accessor_type& aaError,
384  IRefiner& refiner,
385  ConstSmartPtr<DoFDistribution> dd)
386 {
387  typedef typename base_type::elem_type TElem;
388  typedef typename DoFDistribution::traits<TElem>::const_iterator const_iterator;
389 
390  // compute minimal/maximal/ total error and number of elements
391 
392  number minElemErr, minElemErrLocal;
393  number maxElemErr, maxElemErrLocal;
394  number errTotal, errLocal;

397  ComputeMinMax(aaError, dd, minElemErr, maxElemErr, errTotal, numElem,
398  minElemErrLocal, maxElemErrLocal, errLocal, numElemLocal);
399 
400  this->m_latest_error = sqrt(errTotal);
401  this->m_latest_error_per_elem_max = maxElemErr;
402  this->m_latest_error_per_elem_min = minElemErr;
403

405  number elemMean = errTotal / numElem;
406 
407  UG_LOG(" +++ VarianceMarking: Mean error : " << elemMean << " on "<< numElem << " elements.\n");
408

410  const_iterator iter;
411  const const_iterator iterEnd = dd->template end<TElem>();
412 
413  number elemVar = 0.0;
414  for (iter = dd->template begin<TElem>(); iter != iterEnd; ++iter)
415  {
416  TElem* elem = *iter;
417  number elemError = aaError[elem]; // eta_i^2
418 
419  if (elemError < 0) continue;
420  elemVar += (elemMean-elemError) * (elemMean-elemError);
421 
422 
423  }
424 
425 #ifdef UG_PARALLEL

432  }
433 #endif

435 
436 
437  elemVar /= (numElem-1.0);
438  UG_LOG(" +++ VarianceMarking: Est. variance (2): " << elemVar << " on "<< numElem << " elements.\n");
439 
440 
441  // refine all element above threshold
442  const number sigma = sqrt(elemVar);
443  const number maxError = elemMean + sigma*m_width;
444  UG_LOG(" +++ Refining elements if error greater " << sigma << "*" << m_width <<" + "<< elemMean <<
445  " = " << maxError << ".\n");
446 
447 
448 
449  const number minErrToRefine = maxError*m_theta;
450  UG_LOG(" +++ Refining elements if error greater " << maxError << "*" << m_theta <<
451  " = " << minErrToRefine << ".\n");
452 
453  // reset counter

455 
456  // loop elements for marking

459  // get element

461
