53 template<typename TGroupObj, typename TDomain, typename TAlgebra>
54 void ElementGaussSeidelStep(const typename TAlgebra::matrix_type& A,
55  GridFunction<TDomain, TAlgebra>& c,
56  const typename TAlgebra::vector_type& d,
57  number relax)
58 {
59  typedef typename TAlgebra::matrix_type::const_row_iterator const_row_iterator;
60  const static int blockSize = TAlgebra::blockSize;
61 
62  // memory for local algebra
63  DenseVector< VariableArray1<number> > s;
64  DenseVector< VariableArray1<number> > x;
65  DenseMatrix< VariableArray2<number> > mat;
66  std::vector<size_t> vInd;
67 
68  // set all vector entries to zero
69  c.set(0.0);
70 #ifdef UG_PARALLEL
71  c.set_storage_type(PST_ADDITIVE);
72 #endif
73  typedef typename GridFunction<TDomain, TAlgebra>::element_type Element;
74  std::vector<Element*> vElem;
75 
76  // loop all grouping objects
77  typedef typename GridFunction<TDomain, TAlgebra>::template traits<TGroupObj>::const_iterator GroupObjIter;
78  for(GroupObjIter iter = c.template begin<TGroupObj>();
79  iter != c.template end<TGroupObj>(); ++iter){
80 
81  // get grouping obj
82  TGroupObj* groupObj = *iter;
83 
84  // collect elems associated to grouping object
85  c.collect_associated(vElem, groupObj);
86 
87  // get all algebraic indices on element
88  vInd.clear();
89  for(size_t i = 0; i < vElem.size(); ++i)
90  c.algebra_indices(vElem[i], vInd, false);
91 
92  // check for doublicates
93  if(vElem.size() > 1){
94  std::sort(vInd.begin(), vInd.end());
95  vInd.erase(std::unique(vInd.begin(), vInd.end()), vInd.end());
96  }
97 
98  // get number of indices on patch
99  const size_t numIndex = vInd.size();
100 
101  // fill local block matrix
102  bool bFound;
103  mat.resize(numIndex, numIndex);
104  mat = 0.0;
105  for (size_t j = 0; j<numIndex; j++){
106  for (size_t k=0;k<numIndex;k++){
107  const_row_iterator it = A.get_connection(vInd[j],vInd[k], bFound);
108  if(bFound){
109  mat.subassign(j*blockSize,k*blockSize,it.value());
110  }
111  };
112  }
113 
114  // compute s[j] := d[j] - sum_k A(j,k)*c[k]
115  // note: the loop over k is the whole matrix row (not only selected indices)
116  s.resize(numIndex);
117  for (size_t j = 0; j<numIndex; j++){
118  typename TAlgebra::vector_type::value_type sj = d[vInd[j]];
119  for(const_row_iterator it = A.begin_row(vInd[j]);
120  it != A.end_row(vInd[j]) ; ++it){
121  MatMultAdd(sj, 1.0, sj, -1.0, it.value(), c[it.index()]);
122  };
123  s.subassign(j*blockSize,sj);
124  };
125 
126  // solve block
127  x.resize(numIndex);
128  InverseMatMult(x,1,mat,s);
129  for (size_t j=0;j<numIndex;j++){
130  c[vInd[j]] += relax*x[j];
131  };
132  }
133 }
