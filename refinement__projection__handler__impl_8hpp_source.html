63 {
64  UG_COND_THROW(subsetIndex < 0,
65  "Invalid subset specified during refinement projector registration");
66 
67  if(subsetIndex >= (int)m_callbacks.size())

82 template <class TParent>
83 inline void RefinementProjectionHandler<TAPosition>::
84 handle_new_vertex(Vertex* vrt, TParent* parent)
85 {
86  UG_ASSERT(parent, "A vertex can only be created from a parent element!");
87 
88  int si = m_sh->get_subset_index(parent);
89  if((si == -1) || (si >= (int)m_callbacks.size()) || (!m_callbacks[si].valid()))
90  m_defaultCallback->new_vertex(vrt, parent);
91  else
92  m_callbacks[si]->new_vertex(vrt, parent);
93 }
