214  void global_to_local(MathVector<dim>& locPos,
215  const MathVector<worldDim>& globPos,
216  const size_t maxIter = 1000,
217  const number tol = 1e-10) const
218  {
219  MathMatrix<worldDim, dim> J;
220  MathMatrix<dim, worldDim> JInv;
221  MathVector<worldDim> dist, compGlobPos;
222  number tolSq = sq(tol);
223 
224  for (size_t i = 0; i < maxIter; ++i) {
225 
226  // f(x) := \phi(x) - x_{glob}
227  getImpl().local_to_global(compGlobPos, locPos);
228  VecSubtract(dist, compGlobPos, globPos);
229 
230  // check if tol reached
231  if(VecTwoNormSq(dist) < tolSq) return;
232 
233  // compute jacobian df/dx = d \phi(x) / dx =: J
234  getImpl().jacobian(J, locPos);
235 
236  // solve x -= J^{-1} f
237  LeftInverse(JInv, J);
238  MatVecScaleMultAppend(locPos, -1.0, JInv, dist);
239  }
240 
241  // compiler does not know that maxIter will never be 0
242  // therefore it warns that dist may be uninit'ed;
243  // as we will throw here anyway, we may as well check that
244  UG_COND_THROW(!maxIter, "Without a single iteration, local-to-global "
245  "mapping can never converge.");
246 
247  UG_THROW("ReferenceMapping::global_to_local: Newton method did not"
248  " reach a tolerance "<<tol<<" after "<<maxIter<<" steps. "
249  "Global Pos: "<<globPos<<", dim: "<<dim<<", worldDim: "<<
250  worldDim<<", last newton defect: "<<fabs(VecTwoNorm(dist)));
251  }

254  void global_to_local(MathVector<dim>* vLocPos,
255  const MathVector<worldDim>* vGlobPos, size_t n,
256  const size_t maxIter = 1000,
257  const number tol = 1e-10) const
258  {
259  if(isLinear){
260  if(n == 0) return;
261 
262  MathMatrix<worldDim, dim> J;
263  MathMatrix<dim, worldDim> JInv;
264  MathVector<worldDim> dist, compGlobPos;
265 
266  // compute jacobian df/dx = d \phi(x) / dx =: J
267  getImpl().jacobian(J, vLocPos[0]);
268 
269  // solve x -= J^{-1} f
270  LeftInverse(JInv, J);
271 
272  for(size_t ip = 0; ip < n; ++ip)
273  {
274  // f(x) := \phi(x) - x_{glob}
275  getImpl().local_to_global(compGlobPos, vLocPos[ip]);
276  VecSubtract(dist, compGlobPos, vGlobPos[ip]);
277 
278  MatVecScaleMultAppend(vLocPos[ip], -1.0, JInv, dist);
279  }
280  }
281  else{
282  for(size_t ip = 0; ip < n; ++ip)
283  getImpl().global_to_local(vLocPos[ip], vGlobPos[ip], maxIter, tol);
284  }
