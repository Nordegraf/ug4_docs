256 // Loop all vertices of top level

283 void MoveVertexToSubdivisionVolumePosition( MultiGrid& mg, Vertex* vrt,
284  Grid::VertexAttachmentAccessor<APosition>& aaPos,
285  Grid::VertexAttachmentAccessor<APosition>& aaSmoothPos)
286 {
287 // Declare centroid coordinate vector
288  typedef APosition::ValueType pos_type;
289  pos_type p;
290 
291 // Declare vertex volume valence
292  size_t valence = 0;
293 
294 // Collect associated volumes
295  std::vector<Volume*> volumes;
296  CollectAssociated(volumes, mg, vrt);
297 
298 // Iterate over all associated volumes
299  for(Grid::AssociatedVolumeIterator vIter = mg.associated_volumes_begin(vrt); vIter != mg.associated_volumes_end(vrt); ++vIter)
300  {
301  VecSet(p, 0);
302  Volume* vol = *vIter;
303  ++valence;
304 
305  // TETRAHEDRON CASE
306  if(vol->reference_object_id() == ROID_TETRAHEDRON)
307  {
308  // Iterate over all associated vertices inside tetrahedron
309  int vrtIndex = 0;
310  for(size_t i = 0; i < vol->num_vertices(); ++i)
311  {
312  if(vrtIndex != GetVertexIndex(vol, vrt))
313  {
314  VecAdd(p, p, aaPos[vol->vertex(i)]);
315  }
316 
317  ++vrtIndex;
318  }
319 
320  // TODO: refer to subdivision rules object
321  number centerWgt = -1.0/16;
322  number nbrWgt = 17.0/48;
323 
324  VecScaleAppend(aaSmoothPos[vrt], centerWgt, aaPos[vrt], nbrWgt, p);
325  }
326 
327  // OCTAHEDRON CASE
328  else if(vol->reference_object_id() == ROID_OCTAHEDRON)
329  {
330  // Summate positions of all other vertices inside octahedron, first associate ones and last the opposing one
331  for(Grid::AssociatedEdgeIterator eIter = mg.associated_edges_begin(vol); eIter != mg.associated_edges_end(vol); ++eIter)
332  {
333  Edge* e = *eIter;
334  if(GetConnectedVertex(e, vrt) != NULL)
335  {
336  VecAdd(p, p, aaPos[GetConnectedVertex(e, vrt)]);
337  }
338  }
339 
340  Vertex* oppVrt = vol->vertex(vol->get_opposing_object(vrt).second);
341 
342  // TODO: refer to subdivision rules object
343  number centerWgt = 3.0/8;
344  number nbrWgt = 1.0/12;
345  number oppNbrWgt = 7.0/24;
346 
347  VecScaleAppend(aaSmoothPos[vrt], centerWgt, aaPos[vrt], nbrWgt, p, oppNbrWgt, aaPos[oppVrt]);
348  }
349 
350  // UNSUPPORTED VOLUME ELEMENT CASE
351  else
352  {
353  UG_THROW("Volume type not supported for subdivision volumes refinement.");
354  }
355  }
356 
357 // Scale vertex position by the number of associated volume elements
358  VecScale(aaSmoothPos[vrt], aaSmoothPos[vrt], 1.0/valence);
359 }

371 void MoveVertexToSubdivisionSurfacePosition(MultiGrid& mg, Vertex* vrt,
372  Grid::VertexAttachmentAccessor<APosition>& aaPos,
373  Grid::VertexAttachmentAccessor<APosition>& aaSmoothPos)
374 {
375 // Check, if volumes are included in input grid
376  bool volumesExist = mg.num<Volume>() > 0;
377 
378 // Declare centroid coordinate vector
379  typedef APosition::ValueType pos_type;
380  pos_type p;
381 
382 // Load subdivision surfaces rules
383  SubdivRules_PLoop& subdiv = SubdivRules_PLoop::inst();
384 
385 // Even vertex
386  if(mg.get_parent(vrt)->reference_object_id() == ROID_VERTEX)
387  {
388  //Vertex* parentVrt = dynamic_cast<Vertex*>(mg.get_parent(vrt));
389  Vertex* parentVrt = static_cast<Vertex*>(mg.get_parent(vrt));
390 
391  // perform loop subdivision on even surface vertices
392  // first get neighboured vertices
393  size_t valence = 0;
394  pos_type p;
395  VecSet(p, 0);
396 
397  for(Grid::AssociatedEdgeIterator iter = mg.associated_edges_begin(parentVrt); iter != mg.associated_edges_end(parentVrt); ++iter)
398  {
399  if((!volumesExist) || IsBoundaryEdge3D(mg, *iter))
400  {
401  VecAdd(p, p, aaPos[GetConnectedVertex(*iter, parentVrt)]);
402  ++valence;
403  }
404  }
405 
406  number centerWgt = subdiv.ref_even_inner_center_weight(valence);
407  number nbrWgt = subdiv.ref_even_inner_nbr_weight(valence);
408 
409  VecScaleAdd(aaSmoothPos[vrt], centerWgt, aaPos[parentVrt], nbrWgt, p);
410  }
411 
412 // Odd vertex
413  else if(mg.get_parent(vrt)->reference_object_id() == ROID_EDGE)
414  {
415  // Get parent edge
416  //Edge* parentEdge = dynamic_cast<Edge*>(mg.get_parent(vrt));
417  Edge* parentEdge = static_cast<Edge*>(mg.get_parent(vrt));
418 
419  // apply loop-subdivision on inner elements
420  // get the neighboured triangles
421  Face* f[2];
422  int numAssociatedBndFaces = 0;
423 
424  std::vector<Face*> faces;
425  CollectAssociated(faces, mg, parentEdge);
426 
427  for(size_t i = 0; i < faces.size(); ++i)
428  {
429  if(IsBoundaryFace3D(mg, faces[i]))
430  {
431  if(numAssociatedBndFaces < 2)
432  {
433  f[numAssociatedBndFaces] = faces[i];
434  }
435  ++numAssociatedBndFaces;
436  }
437  }
438 
439  if(numAssociatedBndFaces == 2)
440  {
441  if(f[0]->num_vertices() == 3 && f[1]->num_vertices() == 3)
442  {
443  // the 4 vertices that are important for the calculation
444  Vertex* v[4];
445  v[0] = parentEdge->vertex(0); v[1] = parentEdge->vertex(1);
446  v[2] = GetConnectedVertex(parentEdge, f[0]);
447  v[3] = GetConnectedVertex(parentEdge, f[1]);
448 
449  vector4 wghts;
450 
451  wghts = subdiv.ref_odd_inner_weights();
452 
453  VecScaleAdd(aaSmoothPos[vrt],
454  wghts.x(), aaPos[v[0]], wghts.y(), aaPos[v[1]],
455  wghts.z(), aaPos[v[2]], wghts.w(), aaPos[v[3]]);
456  }
457  else
458  UG_THROW("Non triangular faces included in grid.");
459  }
460  else
461  UG_THROW("numAssociatedBndFaces != 2.");
462  }
463 }

476 void SubdivisionVolumes(MultiGrid& mg, bool bPreserveBnd, bool bSubdivisionLoopBnd)
477 {
478 // Ensure, that hybrid tet-/oct refinement is used as refinement rule for tetrahedrons
479  if(tet_rules::GetRefinementRule() != tet_rules::HYBRID_TET_OCT)
480  UG_THROW("ERROR in SubdivisionVolumes: Set necessary refinement rule by SetTetRefinementRule(hybrid_tet_oct).");
481 
482 // Ensure correct user input
483  if(!bPreserveBnd)
484  bSubdivisionLoopBnd = false;
485 
486 // Position attachment management
487  Grid::VertexAttachmentAccessor<APosition> aaPos(mg, aPosition);
488 
489 // Smooth position attachment plus (default) initialization with zero
490  APosition aSmoothPos;
491  mg.attach_to_vertices_dv(aSmoothPos, vector3(0, 0, 0));
492  Grid::VertexAttachmentAccessor<APosition> aaSmoothPos(mg, aSmoothPos);
493 
494 // Load subdivision surfaces rules
495  //SubdivRules_PLoop& subdiv = SubdivRules_PLoop::inst();
496 
497 // Check, if volumes are included in input grid
498  bool volumesExist = mg.num<Volume>() > 0;
499  if(!volumesExist)
500  UG_THROW("SubdivisionVolumes: No volumes included in input grid for smooth TetGrid subdivision refinement.");
501 
502 // Loop all vertices of top_level
503  for(VertexIterator vrtIter = mg.begin<Vertex>(mg.top_level()); vrtIter != mg.end<Vertex>(mg.top_level()); ++vrtIter)
504  {
505  Vertex* vrt = *vrtIter;
506 
507  if(!bPreserveBnd)
508  {
509  MoveVertexToSubdivisionVolumePosition(mg, vrt, aaPos, aaSmoothPos);
510  }
511 
512  // Surface preservation
513  else
514  {
515  // BOUNDARY VERTEX
516  if(IsBoundaryVertex3D(mg, vrt))
517  {
518  if(bSubdivisionLoopBnd)
519  MoveVertexToSubdivisionSurfacePosition(mg, vrt, aaPos, aaSmoothPos);
520  else
521  aaSmoothPos[vrt] = aaPos[vrt];
522  }
523 
524  // INNER VERTEX
525  else
526  {
527  MoveVertexToSubdivisionVolumePosition(mg, vrt, aaPos, aaSmoothPos);
528  }
529  }
530  }
531 
532 // Move vertices to their smoothed position
533  for(VertexIterator vrtIter = mg.begin<Vertex>(mg.top_level()); vrtIter != mg.end<Vertex>(mg.top_level()); ++vrtIter)
534  {
535  Vertex* vrt = *vrtIter;
536  VecScale(aaPos[vrt], aaSmoothPos[vrt], 1.0);
537  }
538 
539 }
