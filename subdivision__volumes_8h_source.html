300 void TetrahedralizeHybridTetOctGrid(MultiGrid& mg, int bestDiag)
301 {
302  PROFILE_FUNC_GROUP("subdivision_volumes");
303 
304  if(bestDiag != 0 && bestDiag != 1 && bestDiag != 2)
305  {
306  bestDiag = -1;
307  }
308 
309 // Position attachment management
310  Grid::VertexAttachmentAccessor<APosition> aaPos(mg, aPosition);
311 
312  std::vector<Tetrahedron*> vTetsOut;
313 
314 // Loop over all levels and split octahedrons
315  for(size_t i = mg.num_levels()-1; i > 0; --i)
316  {
317  // Loop over all octahedrons on each level
318  for(VolumeIterator octIter = mg.begin<Octahedron>(i); octIter != mg.end<Octahedron>(i); ++octIter)
319  {
320  Octahedron* oct = dynamic_cast<Octahedron*>(*octIter);
321  Volume* parentVol = dynamic_cast<Volume*>(mg.get_parent(oct));
322 
323  SplitOctahedronToTetrahedrons(mg, oct, parentVol, vTetsOut, bestDiag);
324  }
325  }
326 
327 // Erase all octahedrons in multigrid
328  while(mg.begin<Octahedron>() != mg.end<Octahedron>())
329  {
330  Octahedron* oct = *mg.begin<Octahedron>();
331  mg.erase(oct);
332  }
333 }

343 {

351  }

355  ComPol_CopyAttachment<VertexLayout, AVector3> comPolCopyAPosition(mg, aPosition);

360  GridLayoutMap& glm = dgm.grid_layout_map();

408 void CalculateSmoothManifoldPosInParentLevelLoopScheme(MultiGrid& mg, MGSubsetHandler& markSH,
409  MGSubsetHandler& linearManifoldSH,
410  APosition& aSmoothBndPosEvenVrt,
411  APosition& aSmoothBndPosOddVrt,
412  AInt& aNumManifoldEdges)
413 {
414 // Catch use of procedure for MultiGrids with just one level
415  if(mg.num_levels() == 1)
416  {
417  UG_THROW("Error in CalculateSmoothManifoldPosInParentLevel: "
418  "Procedure only to be used for MultiGrids with more than one level.");
419  }
420 
421 // Define attachment accessors
422  Grid::VertexAttachmentAccessor<APosition> aaPos(mg, aPosition);
423  Grid::VertexAttachmentAccessor<APosition> aaSmoothBndPosEvenVrt(mg, aSmoothBndPosEvenVrt);
424  Grid::EdgeAttachmentAccessor<APosition> aaSmoothBndPosOddVrt(mg, aSmoothBndPosOddVrt);
425  Grid::VertexAttachmentAccessor<AInt> aaNumManifoldEdges(mg, aNumManifoldEdges);
426 
427  #ifdef UG_PARALLEL
428  DistributedGridManager& dgm = *mg.distributed_grid_manager();
429  #endif
430 
431 // Declare centroid coordinate vector
432  typedef APosition::ValueType pos_type;
433  pos_type p;
434  VecSet(p, 0);
435 
436 // Load subdivision surfaces rules
437  SubdivRules_PLoop& subdiv = SubdivRules_PLoop::inst();
438 
439 // Calculate smooth position for EVEN vertices
440  for(VertexIterator vrtIter = mg.begin<Vertex>(mg.top_level()-1); vrtIter != mg.end<Vertex>(mg.top_level()-1); ++vrtIter)
441  {
442  VecSet(p, 0);
443 
444  Vertex* vrt = *vrtIter;
445 
446  // Skip ghost vertices
447  #ifdef UG_PARALLEL
448  if(dgm.is_ghost(vrt))
449  continue;
450  #endif
451 
452  // In case of marked manifold vertices, which do not belong to the user-specified linear boundary manifold subsets,
453  // and activated subdivision Loop refinement calculate subdivision surfaces smooth position
454  if(markSH.get_subset_index(vrt) != -1 && linearManifoldSH.get_subset_index(vrt) == -1)
455  {
456  // perform loop subdivision on even manifold vertices
457  // first get neighbored manifold vertices
458  for(Grid::AssociatedEdgeIterator eIter = mg.associated_edges_begin(vrt); eIter != mg.associated_edges_end(vrt); ++eIter)
459  {
460  Edge* e = *eIter;
461 
462  // Only consider associated edges, which are marked as manifold edges
463  if(markSH.get_subset_index(e) != -1)
464  {
465  // Exclude ghost and horizontal slave neighbor vertices from contributing to centroid
466  #ifdef UG_PARALLEL
467  if(dgm.is_ghost(e))
468  continue;
469 
470  if(dgm.contains_status(e, ES_H_SLAVE))
471  continue;
472  #endif
473 
474  VecAdd(p, p, aaPos[GetConnectedVertex(e, vrt)]);
475  }
476  }
477 
478  number centerWgt = subdiv.ref_even_inner_center_weight(aaNumManifoldEdges[vrt]);
479  number nbrWgt = subdiv.ref_even_inner_nbr_weight(aaNumManifoldEdges[vrt]);
480 
481  // Exclude horizontal slaves of the currently smoothed vertex to avoid multiple contributions to centroid
482  #ifdef UG_PARALLEL
483  if(dgm.is_ghost(vrt))
484  continue;
485 
486  if(dgm.contains_status(vrt, ES_H_SLAVE))
487  {
488  VecScaleAppend(aaSmoothBndPosEvenVrt[vrt], nbrWgt, p);
489  continue;
490  }
491  #endif
492 
493  VecScaleAppend(aaSmoothBndPosEvenVrt[vrt], centerWgt, aaPos[vrt], nbrWgt, p);
494  }
495  }
496 
497  /*
498  * Smoothing of odd vertices x
499  *
500  * Weights of face adjacent vertices to x: 1/8
501  * Weights of edge adjacent vertices to x: 3/8
502  *
503  1/8
504  / \
505  3/8--x--3/8
506  \ /
507  1/8
508  *
509  */
510 
511 // Calculate smooth position for ODD vertices
512  for(EdgeIterator eIter = mg.begin<Edge>(mg.top_level()-1); eIter != mg.end<Edge>(mg.top_level()-1); ++eIter)
513  {
514  VecSet(p, 0);
515 
516  Edge* e = *eIter;
517 
518  // Skip ghost edges
519  #ifdef UG_PARALLEL
520  if(dgm.is_ghost(e))
521  continue;
522  #endif
523 
524  // In case of marked manifold edges, which do not belong to the user-specified linear boundary manifold subsets,
525  // and activated subdivision Loop refinement calculate subdivision surfaces smooth position
526  if(markSH.get_subset_index(e) != -1 && linearManifoldSH.get_subset_index(e) == -1)
527  {
528  // perform loop subdivision on odd manifold vertices
529  // get the neighbored manifold triangles
530  std::vector<Face*> associatedFaces;
531  std::vector<Face*> associatedManifoldFaces;
532 
533  CollectAssociated(associatedFaces, mg, e);
534 
535  for(size_t i = 0; i < associatedFaces.size(); ++i)
536  {
537 
538  // Only consider associated faces, which are marked as manifold faces
539  if(markSH.get_subset_index(associatedFaces[i]) != -1)
540  {
541  // Exclude ghost and horizontal slave manifold faces
542  #ifdef UG_PARALLEL
543  if(dgm.is_ghost(associatedFaces[i]))
544  continue;
545 
546  if(dgm.contains_status(associatedFaces[i], ES_H_SLAVE))
547  continue;
548  #endif
549 
550  if(associatedManifoldFaces.size() < 2)
551  {
552  associatedManifoldFaces.push_back(associatedFaces[i]);
553  }
554  }
555  }
556 
557  if(associatedManifoldFaces.size() <= 2)
558  {
559  // Check, if all faces are triangles
560  for(size_t i = 0; i < associatedManifoldFaces.size(); ++i)
561  {
562  if(associatedManifoldFaces[i]->num_vertices() != 3)
563  {
564  UG_THROW("ERROR in CalculateSmoothManifoldPosInParentLevel: Non triangular faces included in grid.");
565  }
566  }
567 
568  // Summate centroid of face adjacent vertices
569  for(size_t i = 0; i < associatedManifoldFaces.size(); ++i)
570  {
571  VecAdd(p, p, aaPos[GetConnectedVertex(e, associatedManifoldFaces[i])]);
572  }
573 
574  // Exclude ghost and horizontal slaves of the parent edge vertices of the currently smoothed vertex
575  // to avoid multiple contributions to the centroid of the edge adjacent vertices
576  #ifdef UG_PARALLEL
577  if(dgm.is_ghost(e))
578  {
579  continue;
580  }
581 
582  if(dgm.contains_status(e, ES_H_SLAVE))
583  {
584  VecScaleAppend(aaSmoothBndPosOddVrt[e], 0.125, p);
585  continue;
586  }
587  #endif
588 
589  VecScaleAppend(aaSmoothBndPosOddVrt[e], 0.375, aaPos[e->vertex(0)], 0.375, aaPos[e->vertex(1)], 0.125, p);
590  }
591  else
592  UG_THROW("ERROR in CalculateSmoothManifoldPosInParentLevel: numAssociatedManifoldFaces > 2.");
593  }
594  }
595 
596 // Manage vertex and edge attachment communication in parallel case -> COMMUNICATE aSmoothBndPosEvenVrt, aSmoothBndPosOddVrt
597  #ifdef UG_PARALLEL
598  // Reduce add operations:
599  // sum up h_slaves into h_masters
600 
601  // Copy operations:
602  // copy h_masters to h_slaves for consistency
603  AttachmentAllReduce<Vertex>(mg, aSmoothBndPosEvenVrt, PCL_RO_SUM);
604  AttachmentAllReduce<Edge>(mg, aSmoothBndPosOddVrt, PCL_RO_SUM);
605  #endif
606 }

618 void CalculateSmoothManifoldPosInTopLevelAveragingScheme(MultiGrid& mg, MGSubsetHandler& markSH,
619  MGSubsetHandler& linearManifoldSH,
620  APosition& aSmoothBndPos)
621 {
622 // Define attachment accessors
623  Grid::VertexAttachmentAccessor<APosition> aaPos(mg, aPosition);
624  Grid::VertexAttachmentAccessor<APosition> aaSmoothBndPos(mg, aSmoothBndPos);
625 
626  #ifdef UG_PARALLEL
627  DistributedGridManager& dgm = *mg.distributed_grid_manager();
628  #endif
629 
630 // Declare centroid coordinate vector
631  typedef APosition::ValueType pos_type;
632  pos_type p;
633  VecSet(p, 0);
634 
635 // Loop all manifold faces of top_level
636  for(FaceIterator fIter = mg.begin<Face>(mg.top_level()); fIter != mg.end<Face>(mg.top_level()); ++fIter)
637  {
638  Face* f = *fIter;
639 
640  // Skip ghost volumes
641  #ifdef UG_PARALLEL
642  if(dgm.is_ghost(f))
643  continue;
644  #endif
645 
646  // In case of marked manifold faces, which do not belong to the user-specified linear boundary manifold subsets,
647  // and activated Averaging scheme calculate subdivision surfaces smooth position
648  if(markSH.get_subset_index(f) != -1 && linearManifoldSH.get_subset_index(f) == -1)
649  {
650  if(f->num_vertices() != 3)
651  UG_THROW("ERROR in CalculateSmoothManifoldPosInTopLevelAveragingScheme: Non triangular faces included in grid.");
652 
653  // Iterate over all face vertices, calculate and apply local centroid masks
654  for(size_t i = 0; i < f->num_vertices(); ++i)
655  {
656  // Init
657  Vertex* vrt = f->vertex(i);
658  VecSet(p, 0);
659 
660  // Summate coordinates of neighbor vertices to vrt inside face
661  for(size_t j = 0; j < f->num_vertices(); ++j)
662  {
663  if(j != i)
664  {
665  VecAdd(p, p, aaPos[f->vertex(j)]);
666  }
667  }
668 
669  // Smooth vertex position
670  VecScaleAppend(aaSmoothBndPos[vrt], 2.0/8, aaPos[vrt], 3.0/8, p);
671  }
672  }
673  }
674 
675 // Manage vertex attachment communication in parallel case -> COMMUNICATE aaSmoothBndPos
676  #ifdef UG_PARALLEL
677  // Reduce add operations:
678  // sum up h_slaves into h_masters
679 
680  // Copy operations:
681  // copy h_masters to h_slaves for consistency
682  AttachmentAllReduce<Vertex>(mg, aSmoothBndPos, PCL_RO_SUM);
683  #endif
684 }
