222  this_type& operator=(number val)
223  {
224  for(size_t fct = 0; fct < m_vvValue.size(); ++fct)
225  for(size_t dof = 0; dof < m_vvValue[fct].size(); ++dof)
226  m_vvValue[fct][dof] = val;
227  return *this;
228  }
229 
231  this_type& operator*(number val)
232  {
233  return this->operator*=(val);
234  }

241  m_vvValue[fct][dof] *= val;
242  return *this;
243  }

248  UG_LOCALALGEBRA_ASSERT(m_pIndex==rhs.m_pIndex, "Not same indices.");
249  for(size_t fct = 0; fct < m_vvValue.size(); ++fct)
250  for(size_t dof = 0; dof < m_vvValue[fct].size(); ++dof)
251  m_vvValue[fct][dof] += rhs.m_vvValue[fct][dof];
252  return *this;
253  }
254 
256  this_type& operator-=(const this_type& rhs)
257  {
258  UG_LOCALALGEBRA_ASSERT(m_pIndex==rhs.m_pIndex, "Not same indices.");
259  for(size_t fct = 0; fct < m_vvValue.size(); ++fct)
260  for(size_t dof = 0; dof < m_vvValue[fct].size(); ++dof)
261  m_vvValue[fct][dof] -= rhs.m_vvValue[fct][dof];
262  return *this;
263  }
264 
266  this_type& scale_append(number s, const this_type& rhs)
267  {
268  UG_LOCALALGEBRA_ASSERT(m_pIndex==rhs.m_pIndex, "Not same indices.");
269  for(size_t fct = 0; fct < m_vvValue.size(); ++fct)

273  }

280  void access_by_map(const FunctionIndexMapping& funcMap)
281  {
282  m_pFuncMap = &funcMap;
283  for(size_t i = 0; i < funcMap.num_fct(); ++i)
284  {
285  const size_t mapFct = funcMap[i];
286  m_vvValueAcc[i] = &(m_vvValue[mapFct][0]);

289 
291  void access_all()
292  {
293  m_pFuncMap = NULL;
294 
295  if(m_pIndex==NULL) {m_vvValueAcc.clear(); return;}
296 
297  for(size_t i = 0; i < m_pIndex->num_fct(); ++i)
298  m_vvValueAcc[i] = &(m_vvValue[i][0]);
299  }
300 
302  size_t num_fct() const
303  {
304  if(m_pFuncMap == NULL) return m_vvValue.size();
305  return m_pFuncMap->num_fct();
306  }
307 
309  const LFEID& local_finite_element_id(size_t fct) const
310  {
311  UG_ASSERT(m_pIndex != NULL, "No indices present");
312  check_fct(fct);
313  if(m_pFuncMap == NULL) return m_pIndex->local_finite_element_id(fct);
314  else return m_pIndex->local_finite_element_id((*m_pFuncMap)[fct]);
315  }
316 
318  size_t num_dof(size_t fct) const
319  {
320  check_fct(fct);
321  if(m_pFuncMap == NULL) return m_vvValue[fct].size();
322  else return m_vvValue[ (*m_pFuncMap)[fct] ].size();
323  }
324 
326  number& operator()(size_t fct, size_t dof)
327  {
328  check_dof(fct,dof);
329  return m_vvValueAcc[fct][dof];
330  }
331 
333  number operator()(size_t fct, size_t dof) const
334  {
335  check_dof(fct,dof);
336  return m_vvValueAcc[fct][dof];
337  }
338 
340  // all DoF access
342 
344  size_t num_all_fct() const {return m_vvValue.size();}
345 
347  size_t num_all_dof(size_t fct) const {check_all_fct(fct); return m_vvValue[fct].size();}
348 
350  number& value(size_t fct, size_t dof){check_all_dof(fct,dof);return m_vvValue[fct][dof];}
351 
353  const number& value(size_t fct, size_t dof) const{check_all_dof(fct,dof);return m_vvValue[fct][dof];}
354 
355  protected:
357  inline void check_fct(size_t fct) const
358  {
359  UG_LOCALALGEBRA_ASSERT(fct < num_fct(), "Wrong index: fct: "<<fct<<
360  " of num_fct: "<<num_fct());
361  }
363  inline void check_dof(size_t fct, size_t dof) const
364  {
365  check_fct(fct);
366  UG_LOCALALGEBRA_ASSERT(dof < num_dof(fct), "Wrong index: dof: "
367  <<dof<<", num_dof: "<<num_dof(fct)<<" of fct: "<<fct);
368  }
370  inline void check_all_fct(size_t fct) const
371  {
372  UG_LOCALALGEBRA_ASSERT(fct < num_all_fct(), "Wrong index.");
373  }
375  inline void check_all_dof(size_t fct, size_t dof) const
376  {
377  check_all_fct(fct);
378  UG_LOCALALGEBRA_ASSERT(dof < num_all_dof(fct), "Wrong index.");
379  }
380 
381  protected:
383  const LocalIndices* m_pIndex;
384 
386  const FunctionIndexMapping* m_pFuncMap;
387 
389  std::vector<value_type*> m_vvValueAcc;
390 
392  std::vector<std::vector<value_type> > m_vvValue;
393 };

428  m_CplMatAcc.resize(rowInd.num_fct(), colInd.num_fct());

431  for(size_t fct2 = 0; fct2 < m_CplMat.num_cols(); ++fct2)

541 
542  for(size_t i = 0; i < m_pRowIndex->num_fct(); ++i)
