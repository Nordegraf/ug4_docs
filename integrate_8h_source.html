204 template <int WorldDim, int dim, typename TConstIterator>
205 number Integrate(TConstIterator iterBegin,
206  TConstIterator iterEnd,
207  typename domain_traits<WorldDim>::position_accessor_type& aaPos,
208  IIntegrand<number, WorldDim>& integrand,
209  int quadOrder, std::string quadType,
210  Grid::AttachmentAccessor<
211  typename domain_traits<dim>::grid_base_object, ANumber>
212  *paaElemContribs = NULL
213  )
214 {
215  PROFILE_FUNC();
216 
217 // reset the result
218  number integral = 0;
219 
220 // note: this iterator is for the base elements, e.g. Face and not
221 // for the special type, e.g. Triangle, Quadrilateral
222  TConstIterator iter = iterBegin;
223 
224 // this is the base element type (e.g. Face). This is the type when the
225 // iterators above are dereferenciated.
226  typedef typename domain_traits<dim>::grid_base_object grid_base_object;
227 
228 // get quad type
229  if(quadType.empty()) quadType = "best";
230  QuadType type = GetQuadratureType(quadType);
231 
232 // accessing without dereferencing a pointer first is simpler...
233  Grid::AttachmentAccessor<grid_base_object, ANumber> aaElemContribs;
234  if(paaElemContribs)
235  aaElemContribs = *paaElemContribs;
236 
237 // We'll reuse containers to avoid reallocations
238  std::vector<MathVector<WorldDim> > vCorner;
239  std::vector<MathVector<WorldDim> > vGlobIP;
240  std::vector<MathMatrix<dim, WorldDim> > vJT;
241  std::vector<number> vValue;
242 
243 // iterate over all elements
244  for(; iter != iterEnd; ++iter)
245  {
246  // get element
247  grid_base_object* pElem = *iter;
248 
249  // get reference object id (i.e. Triangle, Quadrilateral, Tetrahedron, ...)
250  ReferenceObjectID roid = (ReferenceObjectID) pElem->reference_object_id();
251 
252  // get quadrature Rule for reference object id and order
253  try{
254  const QuadratureRule<dim>& rQuadRule
255  = QuadratureRuleProvider<dim>::get(roid, quadOrder, type);
256 
257  // get reference element mapping by reference object id
258  DimReferenceMapping<dim, WorldDim>& mapping
259  = ReferenceMappingProvider::get<dim, WorldDim>(roid);
260 
261  // number of integration points
262  const size_t numIP = rQuadRule.size();
263 
264  // get all corner coordinates
265  CollectCornerCoordinates(vCorner, *pElem, aaPos, true);
266 
267  // update the reference mapping for the corners
268  mapping.update(&vCorner[0]);
269 
270  // compute global integration points
271  vGlobIP.resize(numIP);
272  mapping.local_to_global(&(vGlobIP[0]), rQuadRule.points(), numIP);
273 
274  // compute transformation matrices
275  vJT.resize(numIP);
276  mapping.jacobian_transposed(&(vJT[0]), rQuadRule.points(), numIP);
277 
278  // compute integrand values at integration points
279  vValue.resize(numIP);
280  try
281  {
282  integrand.values(&(vValue[0]), &(vGlobIP[0]),
283  pElem, &vCorner[0], rQuadRule.points(),
284  &(vJT[0]),
285  numIP);
286  }
287  UG_CATCH_THROW("Unable to compute values of integrand at integration point.");
288 
289  // reset contribution of this element
290  number intValElem = 0;
291 
292  // loop integration points
293  for(size_t ip = 0; ip < numIP; ++ip)
294  {
295  // get quadrature weight
296  const number weightIP = rQuadRule.weight(ip);
297 
298  // get determinate of mapping
299  const number det = SqrtGramDeterminant(vJT[ip]);
300 
301  // add contribution of integration point
302  intValElem += vValue[ip] * weightIP * det;
303  }
304 
305  // add to global sum
306  integral += intValElem;
307  if(aaElemContribs.valid())
308  aaElemContribs[pElem] = intValElem;
309 
310  }UG_CATCH_THROW("SumValuesOnElems failed.");
311  } // end elem
312 
313 // return the summed integral contributions of all elements
314  return integral;
315 }

337 template <typename TGridFunction>
338 number IntegrateSubsets(SmartPtr<IIntegrand<number, TGridFunction::dim> > spIntegrand,
339  SmartPtr<TGridFunction> spGridFct,
340  const char* subsets, int quadOrder,
341  std::string quadType = std::string())
342 {
343 // world dimensions
344  static const int dim = TGridFunction::dim;
345 
346 // read subsets
347  SubsetGroup ssGrp(spGridFct->domain()->subset_handler());
348  if(subsets != NULL)
349  {
350  ssGrp.add(TokenizeString(subsets));
351  if(!SameDimensionsInAllSubsets(ssGrp))
352  UG_THROW("IntegrateSubsets: Subsets '"<<subsets<<"' do not have same dimension."
353  "Can not integrate on subsets of different dimensions.");
354  }
355  else
356  {
357  // add all subsets and remove lower dim subsets afterwards
358  ssGrp.add_all();
359  RemoveLowerDimSubsets(ssGrp);
360  }
361 
362 // reset value
363  number value = 0;
364 
365 // loop subsets
366  for(size_t i = 0; i < ssGrp.size(); ++i)
367  {
368  // get subset index
369  const int si = ssGrp[i];
370 
371  // check dimension
372  if(ssGrp.dim(i) > dim)
373  UG_THROW("IntegrateSubsets: Dimension of subset is "<<ssGrp.dim(i)<<", but "
374  " World Dimension is "<<dim<<". Cannot integrate this.");
375 
376  // integrate elements of subset
377  try{
378  switch(ssGrp.dim(i))
379  {
380  case DIM_SUBSET_EMPTY_GRID: break;
381  case 1: value += IntegrateSubset<TGridFunction, 1>(spIntegrand, spGridFct, si, quadOrder, quadType); break;
382  case 2: value += IntegrateSubset<TGridFunction, 2>(spIntegrand, spGridFct, si, quadOrder, quadType); break;
383  case 3: value += IntegrateSubset<TGridFunction, 3>(spIntegrand, spGridFct, si, quadOrder, quadType); break;
384  default: UG_THROW("IntegrateSubsets: Dimension "<<ssGrp.dim(i)<<" not supported. "
385  " World dimension is "<<dim<<".");
386  }
387  }
388  UG_CATCH_THROW("IntegrateSubsets: Integration failed on subset "<<si);
389  }
390 
391 #ifdef UG_PARALLEL
392  // sum over processes
393  if(pcl::NumProcs() > 1)
394  {
395  pcl::ProcessCommunicator com;
396  number local = value;
397  com.allreduce(&local, &value, 1, PCL_DT_DOUBLE, PCL_RO_SUM);
398  }
399 #endif
400 
401 // return the result
402  return value;
403 }

657  template <int elemDim>
658  void evaluate(number vValue[],
659  const MathVector<worldDim> vGlobIP[],
660  GridObject* pElem,
661  const MathVector<worldDim> vCornerCoords[],
662  const MathVector<elemDim> vLocIP[],
663  const MathMatrix<elemDim, worldDim> vJT[],
664  const size_t numIP)
665  {
666  // get reference object id (i.e. Triangle, Quadrilateral, Tetrahedron, ...)
667  const ReferenceObjectID roid = pElem->reference_object_id();
668 
669  try{
670  // get trial space
671  const LocalShapeFunctionSet<elemDim>& rTrialSpace =
672  LocalFiniteElementProvider::get<elemDim>(roid, m_id);
673 
674  // number of dofs on element
675  const size_t num_sh = rTrialSpace.num_sh();
676 
677  // get multiindices of element
678  std::vector<DoFIndex> ind; // aux. index array
679  m_spGridFct->dof_indices(pElem, m_fct, ind);
680 
681  // check multi indices
682  if(ind.size() != num_sh)
683  UG_THROW("L2ErrorIntegrand::evaluate: Wrong number of"
684  " multi indices.");
685 
686  // loop all integration points
687  for(size_t ip = 0; ip < numIP; ++ip)
688  {
689  // compute exact solution at integration point
690  number exactSolIP;
691  (*m_spExactSolution)(exactSolIP, vGlobIP[ip], m_time, this->subset());
692 
693  // compute approximated solution at integration point
694  number approxSolIP = 0.0;
695  for(size_t sh = 0; sh < num_sh; ++sh)
696  {
697  // get value at shape point (e.g. corner for P1 fct)
698  const number valSH = DoFRef(*m_spGridFct, ind[sh]);
699 
700  // add shape fct at ip * value at shape
701  approxSolIP += valSH * rTrialSpace.shape(sh, vLocIP[ip]);
702  }
703 
704  // get squared of difference
705  vValue[ip] = (exactSolIP - approxSolIP);
706  vValue[ip] *= vValue[ip];
707  }
