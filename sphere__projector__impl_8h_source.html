103 template <class TAPosition>
104 template <class TElem>
105 void SphereProjector<TAPosition>::
106 perform_projection(Vertex* vrt, TElem* parent)
107 {
108  assert(m_aaPos.valid() && "make sure to initialise the refiner-callback correctly.");
109 
110 // first calculate the average distance of corners of parent and the parents center
111  typename TElem::ConstVertexArray vrts = parent->vertices();
112  size_t numVrts = parent->num_vertices();
113  number avDist = 0;
114  pos_type parentCenter;
115  VecSet(parentCenter, 0);
116 
117  for(size_t i = 0; i < numVrts; ++i){
118  const pos_type& p = m_aaPos[vrts[i]];
119  avDist += VecDistance(p, m_center);
120  VecAdd(parentCenter, parentCenter, p);
121  }
122 
123  avDist /= (number)numVrts;
124  VecScale(parentCenter, parentCenter, 1. / (number)numVrts);
125 
126  if(vrt->is_constrained()){
127  m_aaPos[vrt] = parentCenter;
128  return;
129  }
130 
131 // calculate projection
132  pos_type cylProj;
133  VecSubtract(cylProj, parentCenter, m_center);
134  number len = VecLength(cylProj);
135  if(len > SMALL * avDist){ // if avDist is very small, len may be small, too
136  VecScale(cylProj, cylProj, avDist / len);
137  VecAdd(m_aaPos[vrt], cylProj, m_center);
138  }
139  else
140  m_aaPos[vrt] = parentCenter;
141 }
