262  void replace_indices_in_layout(slice_desc_type type, IndexLayout &il) const
263  {
264  IndexLayout::iterator iter;
265  for (iter = il.begin(); iter!=il.end(); ++iter)
266  {
267  // iterate over interfaces
268  // i.e., pcl::OrderedInterface<size_t, std::vector>
269  IndexLayout::Interface &interf=il.interface(iter);
270 
271  IndexLayout::Interface::iterator eiter;
272  for (eiter = interf.begin(); eiter!=interf.end(); ++eiter)
273  {
274  // replace elements in interface
275  size_t &elem = interf.get_element(eiter);
276  int newind;
277 
278  bool found=find_index(type, elem, newind);
279  UG_COND_THROW(!found, "SlicingData:: Did not find index???");
280  interf.get_element(eiter) = newind;
281 
282  }
283  }
284  }
