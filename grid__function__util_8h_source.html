79 template <typename TBaseElem, typename TGridFunction>
80 static void ScaleGFOnElems
81 (
82  ConstSmartPtr<DoFDistribution> dd,
83  SmartPtr<TGridFunction> vecOut,
84  ConstSmartPtr<TGridFunction> vecIn,
85  const std::vector<number>& vScale
86 )
87 {
88  typename DoFDistribution::traits<TBaseElem>::const_iterator iter, iterEnd;
89  std::vector<DoFIndex> vInd;
90 
91  try
92  {
93  // iterate all elements (including SHADOW_RIM_COPY!)
94  iter = dd->template begin<TBaseElem>(SurfaceView::ALL);
95  iterEnd = dd->template end<TBaseElem>(SurfaceView::ALL);
96  for (; iter != iterEnd; ++iter)
97  {
98  for (size_t fi = 0; fi < dd->num_fct(); ++fi)
99  {
100  size_t nInd = dd->inner_dof_indices(*iter, fi, vInd);
101 
102  // remember multi indices
103  for (size_t dof = 0; dof < nInd; ++dof)
104  DoFRef(*vecOut, vInd[dof]) = vScale[fi] * DoFRef(*vecIn, vInd[dof]);
105  }
106  }
107  }
108  UG_CATCH_THROW("Error while scaling vector.")
109 }

121 template <typename TGridFunction>
122 void ScaleGF
123 (
124  SmartPtr<TGridFunction> scaledVecOut,
125  ConstSmartPtr<TGridFunction> vecIn,
126  const std::vector<number>& scalingFactors
127 )
128 {
129  // check that the correct numbers of scaling factors are given
130  size_t n = scalingFactors.size();
131  UG_COND_THROW(n != vecIn->num_fct(), "Number of scaling factors (" << n << ") "
132  "does not match number of functions given in dimless vector (" << vecIn->num_fct() << ").");
133 
134  // check that input and output vectors have the same number of components and dofs
135  UG_COND_THROW(n != scaledVecOut->num_fct(), "Input and output vectors do not have "
136  "the same number of functions (" << n << " vs. " << scaledVecOut->num_fct() << ").");
137  for (size_t fct = 0; fct < n; ++fct)
138  {
139  UG_COND_THROW(vecIn->num_dofs(fct) != scaledVecOut->num_dofs(fct),
140  "Input and output vectors do not have the same number of DoFs for function " << fct
141  << " (" << vecIn->num_dofs(fct) << " vs. " << scaledVecOut->num_dofs(fct) << ").");
142  }
143 
144  ConstSmartPtr<DoFDistribution> dd = vecIn->dof_distribution();
145 
146  if (dd->max_dofs(VERTEX))
147  ScaleGFOnElems<Vertex, TGridFunction>(dd, scaledVecOut, vecIn, scalingFactors);
148  if (dd->max_dofs(EDGE))
149  ScaleGFOnElems<Edge, TGridFunction>(dd, scaledVecOut, vecIn, scalingFactors);
150  if (dd->max_dofs(FACE))
151  ScaleGFOnElems<Face, TGridFunction>(dd, scaledVecOut, vecIn, scalingFactors);
152  if (dd->max_dofs(VOLUME))
153  ScaleGFOnElems<Volume, TGridFunction>(dd, scaledVecOut, vecIn, scalingFactors);
154 }

170 template<typename TGridFunction, typename TBaseElem>
171 void SubtractValueFromComponent(SmartPtr<TGridFunction> spGF, size_t fct, number sub)
172 {
173  typedef TGridFunction GF;
174  typedef typename GF::template traits<TBaseElem>::const_iterator iter_type;
175 
176  iter_type iter = spGF->template begin<TBaseElem>();
177  iter_type iterEnd = spGF->template end<TBaseElem>();
178 
179  PeriodicBoundaryManager* pbm = spGF->domain()->grid()->periodic_boundary_manager();
180 
181 // loop elems
182  std::vector<DoFIndex> vMultInd;
183  for(; iter != iterEnd; ++iter)
184  {
185  // get element
186  TBaseElem* elem = *iter;
187 
188  // skip periodic ghosts
189  if (pbm && pbm->is_slave(elem)) continue;
190 
191  // get global indices
192  spGF->inner_dof_indices(elem, fct, vMultInd);
193 
194  // sum up value
195  for(size_t i = 0; i < vMultInd.size(); ++i)
196  {
197  DoFRef(*spGF, vMultInd[i]) -= sub;
198  }
199  }
200 }

212 template<typename TGridFunction>
213 void AdjustMeanValue
214 (
215  SmartPtr<TGridFunction> spGF,
216  const std::vector<std::string>& vCmp,
217  number mean
218 )
219 {
220  typedef TGridFunction GF;
221  PROFILE_FUNC_GROUP("gmg");
222 
223  if(vCmp.empty())
224  return;
225 
226  if(spGF.invalid())
227  UG_THROW("AverageComponent: expects a valid GridFunction.");
228 
229  ConstSmartPtr<DoFDistributionInfo> ddinfo =
230  spGF->approx_space()->dof_distribution_info();
231 
232 // compute integral of components
233  const number area = Integral(1.0, spGF);
234  std::vector<number> vIntegral(vCmp.size(), 0.0);
235  for(size_t f = 0; f < vCmp.size(); f++){
236  const size_t fct = spGF->fct_id_by_name(vCmp[f].c_str());
237  vIntegral[f] = Integral(spGF, vCmp[f].c_str(), NULL, ddinfo->lfeid(fct).order());
238  }
239 
240 // subtract value
241  for(size_t f = 0; f < vCmp.size(); f++)
242  {
243  const number sub = (vIntegral[f] - mean) / area;
244  const size_t fct = spGF->fct_id_by_name(vCmp[f].c_str());
245 
246  if(ddinfo->max_fct_dofs(fct, VERTEX)) SubtractValueFromComponent<GF, Vertex>(spGF, fct, sub);
247  if(ddinfo->max_fct_dofs(fct, EDGE)) SubtractValueFromComponent<GF, Edge>(spGF, fct, sub);
248  if(ddinfo->max_fct_dofs(fct, FACE)) SubtractValueFromComponent<GF, Face>(spGF, fct, sub);
249  if(ddinfo->max_fct_dofs(fct, VOLUME)) SubtractValueFromComponent<GF, Volume>(spGF, fct, sub);
250  }
251 }

305 template <typename TGridFunction, typename TBaseElem>
306 number SumGFValuesAt
307 (
308  TGridFunction * u,
309  size_t fct,
310  SubsetGroup & ssGroup
311 )
312 {
313  typedef typename TGridFunction::template traits<TBaseElem>::const_iterator t_elem_iter;
314 
315  std::vector<DoFIndex> ind;
316  number sum (1);
317 
318 // Loop the selected subsets
319  sum = 0;
320  for (size_t i = 0; i < ssGroup.size (); i++)
321  {
322  int ssi = ssGroup [i];
323 
324  // Loop the elements in the subset
325  for (t_elem_iter vi = u->template begin<TBaseElem> (ssi);
326  vi != u->template end<TBaseElem> (ssi); ++vi)
327  {
328  TBaseElem * vert = *vi;
329 
330  // indices at this element
331  u->inner_dof_indices (vert, fct, ind);
332 
333  if (ind.size () != 1)
334  UG_THROW ("SumGFValuesAt: The function must be scalar!");
335 
336  // add the contribution
337  sum += DoFRef (*u, ind [0]);
338  }
339  }
340 
341 #ifdef UG_PARALLEL
342  pcl::ProcessCommunicator procComm;
343  sum = procComm.allreduce (sum, PCL_RO_SUM);
344 #endif
345 
346  return sum;
347 }

393 template <typename TGridFunction, typename TBaseElem>
394 bool CheckGFforNaN
395 (
396  const TGridFunction * u,
397  size_t fct
398 )
399 {
400  typedef typename TGridFunction::template traits<TBaseElem>::const_iterator t_elem_iter;
401 
402  std::vector<DoFIndex> ind;
403 
404 // Loop the elements in the subset
405  for (t_elem_iter vi = u->template begin<TBaseElem> ();
406  vi != u->template end<TBaseElem> (); ++vi)
407  {
408  TBaseElem * vert = *vi;
409 
410  // indices at this element
411  u->inner_dof_indices (vert, fct, ind);
412 
413  if (ind.size () != 1)
414  UG_THROW ("CheckGFforNaN: The function must be scalar!");
415 
416  // check the value
417  number value = DoFRef (*u, ind [0]);
418  if (isnan (value))
419  {
420  int si = u->domain()->subset_handler()->get_subset_index (vert);
421  UG_LOG ("nan at index " << ind [0] << ", grid data idx " << vert->grid_data_index ());
422  if (si >= 0)
423  UG_LOG (", subset " << u->domain()->subset_handler()->get_subset_name (si))
424  UG_LOG ('\n');
425  return true;
426  }
427  if (isinf (value))
428  {
429  int si = u->domain()->subset_handler()->get_subset_index (vert);
430  UG_LOG ("inf at index " << ind [0] << ", grid data idx " << vert->grid_data_index ());
431  if (si >= 0)
432  UG_LOG (", subset " << u->domain()->subset_handler()->get_subset_name (si))
433  UG_LOG ('\n');
434  return true;
435  }
436  }
437 
438  return false;
439 }

507 template <typename TGridFunction, typename TBaseElem>
508 bool CheckGFValuesWithinBounds
509 (
510  ConstSmartPtr<TGridFunction> u,
511  size_t cmp,
512  number lowerBnd,
513  number upperBnd
514 )
515 {
516  typedef typename TGridFunction::template traits<TBaseElem>::const_iterator elem_it;
517 
518  bool ret = true;
519 
520  // loop the elements in the subset
521  std::vector<DoFIndex> vDI;
522  elem_it it = u->template begin<TBaseElem>();
523  elem_it itEnd = u->template end<TBaseElem>();
524  for (; it != itEnd; ++it)
525  {
526  TBaseElem* elem = *it;
527 
528  // loop indices at this element
529  const size_t nInd = u->inner_dof_indices(elem, cmp, vDI, true);
530  for (size_t i = 0; i < nInd; ++i)
531  {
532  const number& val = DoFRef(*u, vDI[i]);
533  if (val < lowerBnd || val > upperBnd)
534  {
535  UG_LOG_ALL_PROCS("Function value for component " << cmp << " (" << val << ") "
536  "is outside the specified range [" << lowerBnd << ", " << upperBnd << "] "
537  "at " << ElementDebugInfo(*u->domain()->grid(), elem) << std::endl);
538  ret = false;
539  }
540  }
541  }
542 
543 #ifdef UG_PARALLEL
544  if (pcl::NumProcs() > 1)
545  {
546  pcl::ProcessCommunicator pc;
547  int retInt = ret;
548  ret = pc.allreduce(retInt, PCL_RO_BAND);
549  }
550 #endif
551 
552  return ret;
553 }

573 
574  return ret;
