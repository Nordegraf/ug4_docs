42 template <class TIterator>
43 number AreaFaceQuality(TIterator facesBegin, TIterator facesEnd,
44  Grid::VertexAttachmentAccessor<APosition>& aaPos)
45 {
46 // if the area is empty return 0 (bad)
47  if(facesBegin == facesEnd)
48  return 0;
49 
50 // get the first
51  number q = FaceQuality(*facesBegin, aaPos);
52  ++facesBegin;
53 
54 // iterate over the others and find a worse one
55  for(; facesBegin != facesEnd; ++facesBegin){
56  number tq = FaceQuality(*facesBegin, aaPos);
57  if(tq < q)
58  q = tq;
59  }
60 
61 // return the quality
62  return q;
63 }

75 template <class TFaceIterator>
76 void FixFaceOrientation(Grid& grid, TFaceIterator facesBegin,
77  TFaceIterator facesEnd)
78 {
79  using namespace std;
80 // we use marks to differentiate between processed and unprocessed faces
81  grid.begin_marking();
82 
83 // we have to mark all faces between facesBegin and facesEnd initially,
84 // to differentiate them from the other faces in the grid.
85  for(TFaceIterator iter = facesBegin; iter != facesEnd; ++iter)
86  grid.mark(*iter);
87 
88 // this edge descriptor will be used multiple times
89  EdgeDescriptor ed;
90 
91 // containers to store neighbour elements.
92  vector<Face*> vNeighbours;
93 
94 // stack that stores candidates
95  stack<Face*> stkFaces;
96 
97 // we'll iterate through all faces
98  while(facesBegin != facesEnd)
99  {
100  // candidates are empty at this point.
101  // if the face is unprocessed it is a new candidate
102  if(grid.is_marked(*facesBegin))
103  {
104  // mark it as candidate (by removing the mark)
105  grid.unmark(*facesBegin);
106  stkFaces.push(*facesBegin);
107 
108  // while the stack is not empty
109  while(!stkFaces.empty())
110  {
111  // get the candidate
112  Face* f = stkFaces.top();
113  stkFaces.pop();
114 
115  // get the neighbours for each side
116  for(size_t i = 0; i < f->num_edges(); ++i)
117  {
118  f->edge_desc(i, ed);
119  GetNeighbours(vNeighbours, grid, f, i);
120 
121  // fix orientation of unprocessed neighbours.
122  for(size_t j = 0; j < vNeighbours.size(); ++j)
123  {
124  Face* fn = vNeighbours[j];
125  if(grid.is_marked(fn))
126  {
127  // check whether the orientation of f and fn differs.
128  if(EdgeOrientationMatches(&ed, fn))
129  {
130  // the orientation of ed is the same as the orientation
131  // of an edge in fn.
132  // the faces thus have different orientation.
133  grid.flip_orientation(fn);
134  }
135 
136  // mark the face as processed and add it to the stack
137  grid.unmark(fn);
138  stkFaces.push(fn);
139  }
140  }
141  }
142  }
143  }
144 
145  // check the next face
146  ++facesBegin;
147  }
148 
149  grid.end_marking();
150 }

164 template<class TVertexPositionAttachmentAccessor>
165 typename TVertexPositionAttachmentAccessor::ValueType
166 CalculateCenter(const FaceVertices* f, TVertexPositionAttachmentAccessor& aaPosVRT)
167 {
168  uint numVrts = f->num_vertices();
169  typename TVertexPositionAttachmentAccessor::ValueType v;
170 // init v with 0.
171  VecSet(v, 0);
172 
173  FaceVertices::ConstVertexArray vrts = f->vertices();
174 
175 // sum up
176  for(uint i = 0; i < numVrts; ++i)
177  {
178  VecAdd(v, v, aaPosVRT[vrts[i]]);
179  }
180 
181 // average
182  if(numVrts > 0)
183  VecScale(v, v, 1./(number)numVrts);
184 
185  return v;
186 }

190 template<class TAAPosVRT, class TAAWeightVRT>
191 UG_API
192 typename TAAPosVRT::ValueType
193 CalculateCenter(const FaceVertices* f, TAAPosVRT& aaPos, TAAWeightVRT& aaWeight)
194 {
195  uint numVrts = f->num_vertices();
196  typename TAAPosVRT::ValueType v;
197  typedef typename TAAWeightVRT::ValueType weight_t;
198 // init v with 0.
199  VecSet(v, 0);
200 
201  FaceVertices::ConstVertexArray vrts = f->vertices();
202 
203 // sum up
204  weight_t totalWeight = 0;
205  for(uint i = 0; i < numVrts; ++i)
206  {
207  weight_t w = aaWeight[vrts[i]];
208  VecScaleAppend(v, w, aaPos[vrts[i]]);
209  totalWeight += w;
210  }
211 
212 // average
213  if(totalWeight != 0)
214  VecScale(v, v, 1./(number)totalWeight);
215 
216  return v;
217 }

241 template <class TTriangleIterator, class TAAPosVRT>
242 bool ProjectPointToSurface(vector3& vOut, const vector3& v, const vector3& n,
243  TTriangleIterator trisBegin, TTriangleIterator trisEnd,
244  TAAPosVRT& aaPos, bool compareNormals)
245 {
246  bool gotOne = false;
247  number bestDist = 0; // value doesn't matter - will be overwritten later on.
248 
249 // iterate through all triangles and find the closest intersection
250  for(TTriangleIterator iter = trisBegin; iter != trisEnd; ++iter)
251  {
252  Triangle* tri = *iter;
253 
254  const vector3& p1 = aaPos[tri->vertex(0)];
255  const vector3& p2 = aaPos[tri->vertex(1)];
256  const vector3& p3 = aaPos[tri->vertex(2)];
257 
258  vector3 tn;
259  CalculateTriangleNormalNoNormalize(tn, p1, p2, p3);
260 
261  // if normal-check is enabled, we have to make sure, that the points
262  // normal points into the same direction as the triangles normal.
263  if(compareNormals){
264  if(VecDot(tn, n) <= 0)
265  continue;
266  }
267 
268  number bc1, bc2;
269  vector3 vTmp;
270  number distance = DistancePointToTriangle(vTmp, bc1, bc2,
271  v, p1, p2, p3, tn);
272 
273  if(gotOne){
274  if(distance < bestDist){
275  bestDist = distance;
276  vOut = vTmp;
277  }
278  }
279  else{
280  gotOne = true;
281  vOut = vTmp;
282  bestDist = distance;
283  }
284  }
285 
286  return gotOne;
287 }

348 template <class TIterator, class TAAPosVRT>
349 Face* FindSmallestFace(TIterator facesBegin, TIterator facesEnd, TAAPosVRT& aaPos)
350 {
351  // if facesBegin equals facesEnd, then the list is empty and we can
352  // immediately return NULL
353  if(facesBegin == facesEnd)
354  return NULL;
355 
356  // the first face is the first candidate for the smallest face.
357  Face* smallestFace = *facesBegin;
358  number smallestArea = FaceArea(smallestFace, aaPos);
359  ++facesBegin;
360 
361  for(; facesBegin != facesEnd; ++facesBegin){
362  Face* curFace = *facesBegin;
363  number curArea = FaceArea(curFace, aaPos);
364  if(curArea < smallestArea){
365  smallestFace = curFace;
366  smallestArea = curArea;
367  }
368  }
369 
370  return smallestFace;
371 }
