368  // i = number of side
369  void compute_side_midpoints(size_t i,
370  MathVector<dim>& locSideMid,
371  MathVector<worldDim>& gloSideMid)
372  {
373  if(dim>1)
374  {
375  const size_t coID0 = m_rRefElem.id(2, i, 0, 0);
376  locSideMid = m_locMid[0][coID0];
377  gloSideMid = m_gloMid[0][coID0];
378 
379  // add corner coordinates of the corners of the geometric object
380  for(size_t j = 1; j < m_rRefElem.num(2, i, 0); ++j)
381  {
382  const size_t coID = m_rRefElem.id(2, i, 0, j);
383  locSideMid += m_locMid[0][coID];
384  gloSideMid += m_gloMid[0][coID];
385  }
386 
387  // scale for correct averaging
388  locSideMid *= 1./(m_rRefElem.num(2, i, 0));
389  gloSideMid *= 1./(m_rRefElem.num(2, i, 0));
390  }
391  }

418  // returns edgeID of child edge, that has corner co. i is the natural edge index
419  size_t get_child_edge_of_corner(size_t i, size_t co)
420  {
421  for(size_t e = 0; e < m_vNatEdgeInfo[i].num_child_edges(); ++e)
422  {
423  const size_t childId = m_vNatEdgeInfo[i].child_edge(e);
424  if(m_vNewEdgeInfo[childId].from() == co)
425  return childId;
426  if(m_vNewEdgeInfo[childId].to() == co)
427  return childId;
428  }
429  return -1;
430  }
