119 template <class TAPosition>
120 template <class TElem>
121 void CylindricalFalloffProjector<TAPosition>::
122 perform_projection(Vertex* vrt, TElem* parent)
123 {
124  assert(m_aaPos.valid() && "make sure to initialise the refiner-callback correctly.");
125 
126 // calculate the new position by linear interpolation and project that point
127 // onto the cylinder.
128  typename TElem::ConstVertexArray vrts = parent->vertices();
129  size_t numVrts = parent->num_vertices();
130  number avDist = 0;
131  pos_type parentCenter;
132  VecSet(parentCenter, 0);
133 
134  for(size_t i = 0; i < numVrts; ++i){
135  const pos_type& p = m_aaPos[vrts[i]];
136  avDist += DistancePointToRay(p, m_center, m_axis);
137  VecAdd(parentCenter, parentCenter, p);
138  }
139 
140  avDist /= (number)numVrts;
141  VecScale(parentCenter, parentCenter, 1. / (number)numVrts);
142 
143  number ia = (avDist - m_innerRadius) / (m_outerRadius - m_innerRadius);
144 
145  if(ia > 1)
146  m_aaPos[vrt] = parentCenter;
147  else{
148  pos_type proj, v;
149  ProjectPointToRay(proj, parentCenter, m_center, m_axis);
150  VecSubtract(v, parentCenter, proj);
151  number len = VecLength(v);
152  if(len > SMALL * avDist){ // if avDist is very small, len may be small, too
153  VecScale(v, v, avDist / len);
154  VecAdd(v, proj, v);
155 
156  if(ia <= 0)
157  m_aaPos[vrt] = v;
158  else
159  VecScaleAdd(m_aaPos[vrt], 1.-ia, v, ia, parentCenter);
160  }
161  else
162  m_aaPos[vrt] = parentCenter;
163  }
164 }
