56 template <typename TAAPos>
57 bool is_anisotropic
58 (
59  Face* elem,
60  Grid& grid,
61  const TAAPos& aaPos,
62  number thresholdRatio,
63  std::vector<Edge*>* nb
64 )
65 {
66  Quadrilateral* q = dynamic_cast<Quadrilateral*>(elem);
67  if (!q)
68  return false;
69 
70  // check whether elem is anisotropic
71  number sideLength02 = VertexDistance(q->vertex(0), q->vertex(1), aaPos)
72  + VertexDistance(q->vertex(2), q->vertex(3), aaPos);
73 
74  number sideLength13 = VertexDistance(q->vertex(1), q->vertex(2), aaPos)
75  + VertexDistance(q->vertex(3), q->vertex(0), aaPos);
76 
77  number ratio = 0.0;
78  if (sideLength02 > sideLength13)
79  ratio = sideLength13 / sideLength02;
80  else if (sideLength13 > sideLength02)
81  ratio = sideLength02 / sideLength13;
82  else
83  ratio = 1.0;
84 
85  if (ratio >= thresholdRatio)
86  return false;
87 
88 
89  // get elem neighbors at larger sides
90  typedef typename Grid::traits<Face>::secure_container elem_list;
91  typedef typename Grid::traits<Edge>::secure_container side_list;
92 
93  side_list sl;
94  EdgeDescriptor largeSide;
95  if (sideLength02 > sideLength13)
96  largeSide = elem->edge_desc(0);
97  else
98  largeSide = elem->edge_desc(1);
99 
100  grid.associated_elements(sl, elem);
101  size_t slSz = sl.size();
102  for (size_t s = 0; s < slSz; ++s)
103  {
104  if (CompareVertices(sl[s], &largeSide))
105  {
106  Edge* opp = GetOpposingSide(grid, elem, sl[s]);
107  if (nb)
108  {
109  nb->push_back(sl[s]);
110  nb->push_back(opp);
111  }
112 
113  return true;
114  }
115  }
116 
117  return true;
118 }

122 template <typename TAAPos>
123 bool is_anisotropic
124 (
125  Volume* elem,
126  Grid& grid,
127  const TAAPos& aaPos,
128  number thresholdRatio,
129  std::vector<Face*>* nb
130 )
131 {
132  Prism* prism = dynamic_cast<Prism*>(elem);
133 
134  // treat prism case
135  if (prism)
136  {
137  // check whether elem is anisotropic
138  number length = VertexDistance(prism->vertex(0), prism->vertex(3), aaPos)
139  + VertexDistance(prism->vertex(1), prism->vertex(4), aaPos)
140  + VertexDistance(prism->vertex(2), prism->vertex(5), aaPos);
141  number width = VertexDistance(prism->vertex(0), prism->vertex(1), aaPos)
142  + VertexDistance(prism->vertex(1), prism->vertex(2), aaPos)
143  + VertexDistance(prism->vertex(2), prism->vertex(0), aaPos)
144  + VertexDistance(prism->vertex(3), prism->vertex(4), aaPos)
145  + VertexDistance(prism->vertex(4), prism->vertex(5), aaPos)
146  + VertexDistance(prism->vertex(5), prism->vertex(3), aaPos);
147 
148  number ratio = width ? 2.0*length/width : std::numeric_limits<number>::max();
149 
150  if (ratio >= thresholdRatio)
151  return false;
152 
153 
154  // get elem neighbors at larger sides
155  typedef typename Grid::traits<Volume>::secure_container elem_list;
156  typedef typename Grid::traits<Face>::secure_container side_list;
157 
158  side_list sl;
159  FaceDescriptor largeSide = prism->face_desc(0);
160 
161  grid.associated_elements(sl, prism);
162  size_t slSz = sl.size();
163  for (size_t s = 0; s < slSz; ++s)
164  {
165  if (CompareVertices(sl[s], &largeSide))
166  {
167  Face* opp = GetOpposingSide(grid, elem, sl[s]);
168  if (nb)
169  {
170  nb->push_back(sl[s]);
171  nb->push_back(opp);
172  }
173 
174  return true;
175  }
176  }
177 
178  return true;
179  }
180 
181  Hexahedron* hex = dynamic_cast<Hexahedron*>(elem);
182  if (!hex)
183  return false;
184 
185  // treat hexahedron case
186 
187  // check whether elem is anisotropic
188  number length1 = VertexDistance(hex->vertex(0), hex->vertex(1), aaPos)
189  + VertexDistance(hex->vertex(2), hex->vertex(3), aaPos)
190  + VertexDistance(hex->vertex(4), hex->vertex(5), aaPos)
191  + VertexDistance(hex->vertex(6), hex->vertex(7), aaPos);
192  number length2 = VertexDistance(hex->vertex(0), hex->vertex(3), aaPos)
193  + VertexDistance(hex->vertex(1), hex->vertex(2), aaPos)
194  + VertexDistance(hex->vertex(4), hex->vertex(7), aaPos)
195  + VertexDistance(hex->vertex(5), hex->vertex(6), aaPos);
196  number length3 = VertexDistance(hex->vertex(0), hex->vertex(4), aaPos)
197  + VertexDistance(hex->vertex(1), hex->vertex(5), aaPos)
198  + VertexDistance(hex->vertex(2), hex->vertex(6), aaPos)
199  + VertexDistance(hex->vertex(3), hex->vertex(7), aaPos);
200 
201  number ratio;
202  number largestLength = std::max(length1, std::max(length2, length3));
203  number smallestLength = std::min(length1, std::min(length2, length3));
204  if (largestLength > smallestLength)
205  ratio = smallestLength / largestLength;
206  else
207  ratio = 1.0;
208 
209  if (ratio >= thresholdRatio)
210  return false;
211 
212  // get elem neighbors at larger sides
213  typedef typename Grid::traits<Volume>::secure_container elem_list;
214  typedef typename Grid::traits<Face>::secure_container side_list;
215 
216  size_t smallestDim = length1 < length2 ? (length1 < length3 ? 1 : 3) : (length2 < length3 ? 2 : 3);
217  FaceDescriptor largeSide = hex->face_desc(3-smallestDim);
218 
219  side_list sl;
220  grid.associated_elements(sl, hex);
221  size_t slSz = sl.size();
222  for (size_t s = 0; s < slSz; ++s)
223  {
224  if (CompareVertices(sl[s], &largeSide))
225  {
226  Face* opp = GetOpposingSide(grid, elem, sl[s]);
227  if (nb)
228  {
229  nb->push_back(sl[s]);
230  nb->push_back(opp);
231  }
232 
233  return true;
234  }
235  }
236 
237  return true;
238 }
