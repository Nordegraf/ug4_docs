200  write_debug(vec, filename.c_str());
201  }
203  virtual void write_debug(const vector_type& vec, const char* filename)
204  {
205  PROFILE_FUNC_GROUP("algebra debug");
206  // if no debug writer set, we're done
207  if(m_spVectorDebugWriter.invalid()) return;
208 
209  // check ending
210  std::string name(filename);
211  size_t iExtPos = name.find_last_of(".");
212  if(iExtPos != std::string::npos && name.substr(iExtPos).compare(".vec") != 0)
213  UG_THROW("Only '.vec' format supported for vectors, but"
214  " filename is '"<<name<<"'.");
215 
216  if(iExtPos == std::string::npos)
217  name.append(".vec");
218 
219  // write
220  m_spVectorDebugWriter->write_vector(vec, name.c_str());
221  }
222 
223  protected:
225  SmartPtr<IVectorDebugWriter<vector_type> > m_spVectorDebugWriter;
226 };
