507 template <typename TMatrix, typename TVector>
508 void MatExtractDiagOnLayout( TVector* pDiagVector,
509  const TMatrix* pMatrix,
510  const IndexLayout& Layout)
511 {
512  PROFILE_FUNC_GROUP("algebra parallelization");
513 // interface iterator
514  typename IndexLayout::const_iterator iter = Layout.begin();
515  typename IndexLayout::const_iterator end = Layout.end();
516 
517  for(; iter != end; ++iter)
518  {
519  // get interface
520  const typename IndexLayout::Interface& interface = Layout.interface(iter);
521 
522  for(typename IndexLayout::Interface::const_iterator iter = interface.begin();
523  iter != interface.end(); ++iter)
524  {
525  // get index
526  const size_t index = interface.get_element(iter);
527 
528  // copy values
529  const typename TMatrix::value_type& block = (*pMatrix)(index, index);
530  for(size_t beta = 0; beta < (size_t) GetCols(block); ++beta)
531  {
532  BlockRef((*pDiagVector)[index], beta) = BlockRef(block, beta, beta);
533  }
534  }
535  }
536 }

590 void MatAdditiveToConsistentOnDiag( typename TAlgebra::matrix_type* pMat,
591  const IndexLayout& masterLayout, const IndexLayout& slaveLayout,
592  pcl::InterfaceCommunicator<IndexLayout>* pCom = NULL)
593 {
594  PROFILE_FUNC_GROUP("algebra parallelization");

599
