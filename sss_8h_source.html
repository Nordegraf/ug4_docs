103  void addps(const std::vector<number>& _point,
104  const std::vector<number>& _data)
105  {
106  if (_point.size() != dim)
107  UG_THROW("Expected vector of dimension size (dim: " << dim << ").");
108 
109  MathVector<dim> point;
110  for(size_t i = 0; i < dim; i++)
111  point[i] = _point[i];
112 
113  if (_data.size() != non)
114  UG_THROW("Expected vector of data size (non: " << non << ").");
115 
116  MathVector<non> data;
117  for(size_t i = 0; i < non; i++)
118  data[i] = _data[i];
119 
120  ListP.push_back(PointSourceOrSink<dim, non>(point, data));
121  }

188  template<typename TElem,
189  typename TAAPos,
190  typename TFVGeom>
191  number
192  get_contrib_of_scv(TElem* elem,
193  Grid& grid,
194  TAAPos aaPos,
195  TFVGeom& geo,
196  size_t scv,
197  number time,
198  MathVector<non>& out)
199  {
200  typename std::list<PointSourceOrSink<dim, non> >::iterator p;
201  AABox<MathVector<dim> > bbox = CalculateBoundingBox(elem, aaPos);
202  size_t scv_id = (size_t)elem + scv;
203  for (p = ListP.begin(); p != ListP.end(); p++) {
204  bool in = true;
205  if (p->marker != 0) {
206  if (p->marker != scv_id)
207  continue;
208  }
209  else {
210  // quick bbox test
211  if (!bbox.contains_point(p->point))
212  continue;
213 
214  // restrict point to element
215  if (!ContainsPoint(elem, p->point, aaPos))
216  continue;
217 
218  // restrict point to scv
219  for (size_t i = 0; i < geo.num_scvf(); i++) {
220  const typename TFVGeom::SCVF& scvf = geo.scvf(i);
221  MathVector<dim> d;
222  if (scvf.from() == scv) {
223  VecSubtract(d, p->point, scvf.global_ip());
224  if (VecDot(d, scvf.normal()) > 0.0) {
225  in = false;
226  break;
227  }
228  }
229  else if (scvf.to() == scv) {
230  VecSubtract(d, p->point, scvf.global_ip());
231  if (VecDot(d, scvf.normal()) < 0.0) {
232  in = false;
233  break;
234  }
235  }
236  }
237  }
238  if (in) {
239  p->marker = scv_id;
240  if (p->spData.invalid())
241  out = p->data;
242  else
243  (*p->spData)(out, p->point, time, 0);
244  return 1.0;
245  }
246  }
247 
248  typename std::list<LineSourceOrSink<dim, non> >::iterator l;
249  for (l = ListL.begin(); l != ListL.end(); l++)
250  {
251  // quick bbox test
252  if (!bbox.overlaps_line(l->point1, l->point2))
253  continue;
254 
255  // restrict line segment to element
256  MathVector<dim> ls = l->point1;
257  MathVector<dim> le = l->point2;
258  MathVector<dim> dir;
259  number lambda_min, lambda_max;
260  VecSubtract(dir, le, ls);
261  if (!RayElementIntersection(lambda_min, lambda_max, ls, dir, elem, grid, aaPos))
262  continue;
263  if (ContainsPoint(elem, ls, aaPos))
264  lambda_min = 0.0;
265  if (ContainsPoint(elem, le, aaPos))
266  lambda_max = 1.0;
267  if (lambda_min < 0.0 || lambda_max > 1.0)
268  continue;
269  VecScaleAdd(le, 1.0, ls, lambda_max, dir);
270  VecScaleAdd(ls, 1.0, ls, lambda_min, dir);
271 
272  // restrict line segment to scv
273  MathVector<dim> p;
274  bool in = true;
275  for (size_t i = 0; i < geo.num_scvf(); i++) {
276  const typename TFVGeom::SCVF& scvf = geo.scvf(i);
277  number d_min, d_max, lambda;
278  if (scvf.from() == scv) {
279  VecSubtract(p, ls, scvf.global_ip());
280  d_min = VecDot(p, scvf.normal());
281  VecSubtract(p, le, scvf.global_ip());
282  d_max = VecDot(p, scvf.normal());
283  if (d_min*d_max < 0.0) {
284  lambda = fabs(d_min)/(fabs(d_min)+fabs(d_max));
285  VecScaleAdd(p, 1.0-lambda, ls, lambda, le);
286  if (d_max > 0.0)
287  le = p;
288  else
289  ls = p;
290  }
291  else if (d_min > 0.0) {
292  in = false;
293  break;
294  }
295  }
296  else if (scvf.to() == scv) {
297  VecSubtract(p, ls, scvf.global_ip());
298  d_min = VecDot(p, scvf.normal());
299  VecSubtract(p, le, scvf.global_ip());
300  d_max = VecDot(p, scvf.normal());
301  if (d_min*d_max < 0.0) {
302  lambda = fabs(d_min)/(fabs(d_min)+fabs(d_max));
303  VecScaleAdd(p, 1.0-lambda, ls, lambda, le);
304  if (d_max < 0.0)
305  le = p;
306  else
307  ls = p;
308  }
309  else if (d_min < 0.0) {
310  in = false;
311  break;
312  }
313  }
314  }
315  if (in) {
316  VecSubtract(p, le, ls);
317  number len = VecLength(p);
318  if (l->spData.invalid())
319  out = l->data;
320  else
321  (*l->spData)(out, ls, time, 0);
322  return len;
323  }
324  }
325  return 0.0;
326  }
