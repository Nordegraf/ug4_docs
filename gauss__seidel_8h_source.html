46 template<typename TAlgebra>
47 class GaussSeidelBase : public IPreconditioner<TAlgebra>
48 {
49  public:
50  // Algebra type
51  typedef TAlgebra algebra_type;
52 
53  // Vector type
54  typedef typename TAlgebra::vector_type vector_type;
55 
56  // Matrix type
57  typedef typename TAlgebra::matrix_type matrix_type;
58 
60  typedef typename IPreconditioner<TAlgebra>::matrix_operator_type matrix_operator_type;
61 
63  typedef IPreconditioner<TAlgebra> base_type;
64 
65  protected:
66  using base_type::set_debug;
67  using base_type::debug_writer;
68  using base_type::write_debug;
69 
70  public:
71  // Constructor
72  GaussSeidelBase() { m_relax = 1.0; };
73 
75  GaussSeidelBase( const GaussSeidelBase<TAlgebra> &parent )
76  : base_type(parent)
77  {
78  set_sor_relax(parent.m_relax);
79  }
80 
81  // set relaxation parameter to define a SOR-method
82  void set_sor_relax(number relaxFactor){ m_relax = relaxFactor;}
83 
84  virtual const char* name() const = 0;
85  protected:
86 
87  virtual bool supports_parallel() const {return true;}
88 
89  // Preprocess routine
90  virtual bool preprocess(SmartPtr<MatrixOperator<matrix_type, vector_type> > pOp)
91  {
92  PROFILE_BEGIN_GROUP(GaussSeidel_preprocess, "algebra gaussseidel");
93  matrix_type *pA;
94 #ifdef UG_PARALLEL
95  if(pcl::NumProcs() > 1)
96  {
97  // copy original matrix
98  MakeConsistent(*pOp, m_A);
99  // set zero on slaves
100  std::vector<IndexLayout::Element> vIndex;
101  CollectUniqueElements(vIndex, m_A.layouts()->slave());
102  SetDirichletRow(m_A, vIndex);
103  pA = &m_A;
104  }
105  else
106  pA = &(*pOp);
107 #else
108  pA = &(*pOp);
109 #endif
110  THROW_IF_NOT_EQUAL(pA->num_rows(), pA->num_cols());
111 // UG_ASSERT(CheckDiagonalInvertible(A), "GS: A has noninvertible diagonal");
112  UG_COND_THROW(CheckDiagonalInvertible(*pA) == false, name() << ": A has noninvertible diagonal");
113  return true;
114  }
115 
116  // Postprocess routine
117  virtual bool postprocess() {return true;}
118 
119  virtual void step(const matrix_type &A, vector_type &c, const vector_type &d, const number relax) = 0;
120 
121  // Stepping routine
122  virtual bool step(SmartPtr<MatrixOperator<matrix_type, vector_type> > pOp, vector_type& c, const vector_type& d)
123  {
124  PROFILE_BEGIN_GROUP(GaussSeidel_step, "algebra gaussseidel");
125 
126 #ifdef UG_PARALLEL
127  if(pcl::NumProcs() > 1)
128  {
129  // todo: do not clone every time
130  // make defect unique
131  SmartPtr<vector_type> spDtmp = d.clone();
132  spDtmp->change_storage_type(PST_UNIQUE);
133 
134  THROW_IF_NOT_EQUAL_3(c.size(), spDtmp->size(), m_A.num_rows());
135  step(m_A, c, *spDtmp, m_relax);
136  c.set_storage_type(PST_UNIQUE);
137  return true;
138  }
139  else
140 #endif
141  {
142  matrix_type &A = *pOp;
143  THROW_IF_NOT_EQUAL_4(c.size(), d.size(), A.num_rows(), A.num_cols());
144  step(A, c, d, m_relax);
145 #ifdef UG_PARALLEL
146  c.set_storage_type(PST_UNIQUE);
147 #endif
148  return true;
149  }
150  }
151 
152  protected:
153 #ifdef UG_PARALLEL
154  matrix_type m_A;
155 #endif
156 
157  private:
158  // relaxation parameter
159  number m_relax;
160 };
