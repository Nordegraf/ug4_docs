216  virtual void refinement_begins (const ISubGrid* psg)
217  {
218  RefinementProjector::refinement_begins(psg);
219 
220  UG_COND_THROW(!m_sh, "Please set a valid SubsetHandler to "
221  "'ProjectionHandler' before using it during refinement.");
222 
223  // make sure that the internal vector of projectors is at least as large
224  // as there are subsets in the associated subset handler
225  projector_required((int)m_sh->num_subsets()-1);
226 
227  // this selector will be used if we have to pass sub-grids to individual projectors
228  Selector sel;
229 
230  for(size_t i = 0;
231  i < std::min<size_t>(m_projectors.size(), m_sh->num_subsets() + 1);
232  ++i)
233  {
234  RefinementProjector& proj = *m_projectors[i];
235  if(!proj.geometry().valid())
236  proj.set_geometry(geometry());
237 
238  //todo: move this somewhere where it is only triggered if required.
239  // e.g. to 'set_subset_handler' and 'add_projector'
240  proj.set_concerned_elements(make_sp(new IsInSubset(*m_sh, i-1)));
241 
242  if(!proj.refinement_begins_requires_subgrid()) {
243  proj.refinement_begins(NULL);
244  }
245  else{
246  const ISubGrid& sg = *psg;
247 
248  if(!sel.grid())
249  sel.assign_grid(geom().grid());
250 
251  sel.clear();
252  bool selectionPerformed = false;
253 
254  // this is a small optimization. We can either iterate over the elements
255  // of sg or of subset (i-1), i>0. Choose the one with less elements.
256  GridObjectCollection subsetGoc;
257  if(i > 0){
258  subsetGoc = m_sh->get_grid_objects_in_subset(i - 1);
259  // todo: this check would be more precise if only those element-types
260  // were considered which exist in both goc's
261  if(subsetGoc.num_vertices() + subsetGoc.num_edges()
262  + subsetGoc.num_faces() + subsetGoc.num_volumes()
263  < sg.goc().num_vertices() + sg.goc().num_edges()
264  + sg.goc().num_faces() + sg.goc().num_volumes())
265  {
266  if(sg.goc().num_vertices() > 0){
267  lg_for_each_const(Vertex, e, subsetGoc){
268  if(sg.is_contained(e))
269  sel.select(e);
270  }lg_end_for;
271  }
272  if(sg.goc().num_edges() > 0){
273  lg_for_each_const(Edge, e, subsetGoc){
274  if(sg.is_contained(e))
275  sel.select(e);
276  }lg_end_for;
277  }
278  if(sg.goc().num_faces() > 0){
279  lg_for_each_const(Face, e, subsetGoc){
280  if(sg.is_contained(e))
281  sel.select(e);
282  }lg_end_for;
283  }
284  if(sg.goc().num_volumes() > 0){
285  lg_for_each_const(Volume, e, subsetGoc){
286  if(sg.is_contained(e))
287  sel.select(e);
288  }lg_end_for;
289  }
290 
291  selectionPerformed = true;
292  }
293  }
294 
295  if(!selectionPerformed){
296  if(i == 0 || subsetGoc.num_vertices() > 0){
297  lg_for_each_const(Vertex, e, sg.goc()){
298  if(m_sh->get_subset_index(e) + 1 == (int)i)
299  sel.select(e);
300  }lg_end_for;
301  }
302  if(i == 0 || subsetGoc.num_edges() > 0){
303  lg_for_each_const(Edge, e, sg.goc()){
304  if(m_sh->get_subset_index(e) + 1 == (int)i)
305  sel.select(e);
306  }lg_end_for;
307  }
308  if(i == 0 || subsetGoc.num_faces() > 0){
309  lg_for_each_const(Face, e, sg.goc()){
310  if(m_sh->get_subset_index(e) + 1 == (int)i)
311  sel.select(e);
312  }lg_end_for;
313  }
314  if(i == 0 || subsetGoc.num_volumes() > 0){
315  lg_for_each_const(Volume, e, sg.goc()){
316  if(m_sh->get_subset_index(e) + 1 == (int)i)
317  sel.select(e);
318  }lg_end_for;
319  }
320  selectionPerformed = true;
321  }
322 
323  SubGrid<IsSelected> tsg(sel.get_grid_objects(), IsSelected(sel));
324  proj.refinement_begins(&tsg);
325  }
326  }
327  }

365  // in order to automatically treat subset '-1' correctly, internally we add 1

368  m_projectors.resize(index + 2, m_defaultProjector);

371  template <class TParent>
372  number handle_new_vertex (Vertex* vrt, TParent* parent)
373  {
374  int si = m_sh->get_subset_index(parent);
375 
376  UG_ASSERT (si + 1 < (int) m_projectors.size(),
377  "Please make sure to call 'refinement_begins' before calling "
378  "'new_vertex' for the first time for a given subset-assignment.");
379 
380  number ia = m_projectors [si + 1]->new_vertex (vrt, parent);
381  if(ia < 1){
382  vector3 p = pos (vrt);
383  m_linearProjector.new_vertex (vrt, parent);
384  vector3 lp = pos (vrt);
385  p *= ia;
386  lp *= (1. - ia);
387  p += lp;
388  set_pos(vrt, p);
389  }
390  return 1;
391  }
