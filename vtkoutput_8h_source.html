51 template <typename T>
52 struct IteratorProvider
53 {
54  template <typename TElem>
55  struct traits
56  {
57  typedef typename T::template traits<TElem>::iterator iterator;
58  typedef typename T::template traits<TElem>::const_iterator const_iterator;
59  };
60 
61  template <typename TElem>
62  static typename traits<TElem>::const_iterator begin(const T& provider, int si)
63  {
64  if(si < 0) return provider.template begin<TElem>();
65  else return provider.template begin<TElem>(si);
66  }
67 
68  template <typename TElem>
69  static typename traits<TElem>::const_iterator end(const T& provider, int si)
70  {
71  if(si < 0) return provider.template end<TElem>();
72  else return provider.template end<TElem>(si);
73  }
74 };

84  struct traits

87  typedef typename T::template traits<TElem>::const_iterator const_iterator;
88  };
89 
90  template <typename TElem>
91  static typename traits<TElem>::const_iterator begin(const T& provider, int si)
92  {
93  const int lev = provider.num_levels() - 1;
94  if(si < 0) return provider.multi_grid()->template begin<TElem>(lev);
95  else return provider.template begin<TElem>(si, lev);
96  }
97 
98  template <typename TElem>
99  static typename traits<TElem>::const_iterator end(const T& provider, int si)
100  {
101  const int lev = provider.num_levels() - 1;
102  if(si < 0) return provider.multi_grid()->template end<TElem>(lev);
103  else return provider.template end<TElem>(si, lev);
104  }
105 };
