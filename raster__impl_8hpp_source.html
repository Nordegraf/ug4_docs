417 template <class T, int TDIM>
418 T Raster<T, TDIM>::
419 interpolate (const Coordinate& coord, int order) const
420 {
421  switch(order){
422  case 0: {
423  MultiIndex mi;
424  for(size_t d = 0; d < TDIM; ++d)
425  {
426  mi[d] = static_cast<int>(0.5 + (coord[d] - m_minCorner[d]) / m_cellExtension[d]);
427  if(mi[d] < 0) mi[d] = 0;
428  else if(mi[d] >= num_nodes(d)) mi[d] = num_nodes(d) - 1;
429  }
430  return node_value(mi);
431  } break;
432 
433  case 1: {
434  MultiIndex mi;
435  Coordinate lc;
436  for(size_t d = 0; d < TDIM; ++d)
437  {
438  mi[d] = static_cast<int>((coord[d] - m_minCorner[d]) / m_cellExtension[d]);
439  if(mi[d] < 0){
440  mi[d] = 0;
441  lc[d] = 0;
442  }
443  else if(mi[d] + 1 >= num_nodes(d)){
444  mi[d] = num_nodes(d) - 2;
445  lc[d] = 1;
446  }
447  else{
448  lc[d] = ( coord[d]
449  - ((number)mi[d] * m_cellExtension[d]
450  + m_minCorner[d]))
451  / m_cellExtension[d];
452  }
453  }
454 
455  return interpolate_linear(mi, lc);
456  } break;
457 
458  default:
459  UG_THROW("Raster::interpolate(): Unsupported interpolation order: " << order);
460  }
461  return m_noDataValue;
462 }

539 template <class T, int TDIM>
540 void Raster<T, TDIM>::
541 load_from_asc (const char* filename)
542 {
543  using namespace std;
544 
545  #define LFA_ERR_WHERE "Error in Raster::load_from_asc('" << filename << "'): "
546 // this macro helps with error-checks for bad dimensions
547  #define LFA_CHECK_DIM(d, line)\
548  UG_COND_THROW(d >= TDIM, LFA_ERR_WHERE << "Bad dimension '" << d <<\
549  "' in line " << line << " of file " << filename << "," <<\
550  "while trying to read a " << TDIM << "d raster.");
551 
552  std::string fullFileName = FindFileInStandardPaths(filename);
553  UG_COND_THROW(fullFileName.empty(),
554  LFA_ERR_WHERE << "Couldn't find the specified file in any of the standard paths.");
555 
556  ifstream in(fullFileName.c_str());
557  UG_COND_THROW(!in, LFA_ERR_WHERE << "Couldn't access file.");
558 
559  MultiIndex numNodes(0);
560 
561 // indicate whether minCoord was specified as cell-center
562  MultiIndex minCoordIsCenter(0);
563 
564  Coordinate minCoord(0);
565  Coordinate cellSize(0);
566 
567  T noDataValue = T();
568 
569 // parse header
570 // the header lenght varies between different asc files. This depends on the
571 // dimension and whether equlateral cells are specified or not, i.e., whether
572 // 'cellsize' or whether 'xcellsize', 'ycellsize', ... was specified.
573 // We're trying to guess the correct length here.
574  int headerLen = 0;
575  if(TDIM == 1) headerLen = 4;
576  else if(TDIM == 2) headerLen = 6;
577  else if(TDIM == 3) headerLen = 8;
578 
579  for(int i = 0; i < headerLen; ++i){
580  string name;
581  double value;
582  in >> name >> value;
583  UG_COND_THROW(!in, LFA_ERR_WHERE <<
584  "Couldn't parse expected name-value pair in row " << i);
585 
586  name = ToLower(name);
587 
588  if(name.compare("ncols") == 0){
589  LFA_CHECK_DIM(0, i);
590  numNodes[0] = (int)value;
591  }
592  else if(name.compare("nrows") == 0){
593  LFA_CHECK_DIM(1, i);
594  numNodes[1] = (int)value;
595  }
596  else if(name.compare("nstacks") == 0){
597  LFA_CHECK_DIM(2, i);
598  numNodes[2] = (int)value;
599  }
600 
601  else if(name.compare("xllcenter") == 0){
602  LFA_CHECK_DIM(0, i);
603  minCoord[0] = value;
604  minCoordIsCenter[0] = 1;
605  }
606  else if(name.compare("yllcenter") == 0){
607  LFA_CHECK_DIM(1, i);
608  minCoord[1] = value;
609  minCoordIsCenter[1] = 1;
610  }
611  else if(name.compare("zllcenter") == 0){
612  LFA_CHECK_DIM(2, i);
613  minCoord[2] = value;
614  minCoordIsCenter[2] = 1;
615  }
616  else if(name.compare("xllcorner") == 0){
617  LFA_CHECK_DIM(0, i);
618  minCoord[0] = value;
619  }
620  else if(name.compare("yllcorner") == 0){
621  LFA_CHECK_DIM(1, i);
622  minCoord[1] = value;
623  }
624  else if(name.compare("zllcorner") == 0){
625  LFA_CHECK_DIM(2, i);
626  minCoord[2] = value;
627  }
628 
629  else if(name.compare("cellsize") == 0){
630  for(int d = 0; d < TDIM; ++d)
631  cellSize[d] = value;
632  }
633 
634  else if(name.compare("xcellsize") == 0){
635  LFA_CHECK_DIM(0, i);
636  // we have to read additional cell-sizes for the other dimensions
637  headerLen += (TDIM - 1);
638  cellSize[0] = value;
639  }
640  else if(name.compare("ycellsize") == 0){
641  LFA_CHECK_DIM(1, i);
642  cellSize[1] = value;
643  }
644  else if(name.compare("zcellsize") == 0){
645  LFA_CHECK_DIM(2, i);
646  cellSize[2] = value;
647  }
648 
649  else if(name.compare("nodata_value") == 0){
650  noDataValue = value;
651  }
652 
653  else{
654  UG_THROW(LFA_ERR_WHERE << "unknown identifier in header: " << name);
655  }
656  }
657 
658  for(int d = 0; d < TDIM; ++d){
659  if(minCoordIsCenter[d])
660  minCoord[d] -= 0.5 * cellSize[d];
661  }
662 
663 // check validity
664  for(int d = 0; d < TDIM; ++d){
665  UG_COND_THROW(numNodes[d] == 0, LFA_ERR_WHERE << "Num nodes may not be 0 for dim " << d);
666  UG_COND_THROW(cellSize[d] <= 0, LFA_ERR_WHERE << "cell-size must be bigger than 0 for dim " << d);
667  }
668 
669  set_num_nodes(numNodes);
670  set_min_corner(minCoord);
671  Coordinate extension = cellSize;
672  for(int d = 0; d < TDIM; ++d)
673  extension[d] *= (number)(numNodes[d] - 1);
674  set_extension(extension);
675  set_no_data_value(noDataValue);
676 
677  create();
678 
679 // parse values
680  const size_t numNodesTotal = num_nodes_total();
681  for(size_t i = 0; i < numNodesTotal; ++i){
682  in >> m_data[i];
683  UG_COND_THROW(!in, LFA_ERR_WHERE << "Couldn't read value for node " << i);
684  }
685 }

687 template <class T, int TDIM>
688 void Raster<T, TDIM>::
689 save_to_asc (const char* filename) const
690 {
691  using namespace std;
692  #define STA_ERR_WHERE "Error in Raster::save_to_asc('" << filename << "'): "
693 
694  UG_COND_THROW(!m_data, STA_ERR_WHERE << "Can't write an unitinialized raster."
695  "Please call 'create' or 'load_from_asc' first.");
696 
697  ofstream out(filename);
698  UG_COND_THROW(!out, STA_ERR_WHERE << "Couldn't open file for writing.");
699 
700  out << "ncols " << num_nodes(0) << endl;
701  if(TDIM > 1)
702  out << "nrows " << num_nodes(1) << endl;
703  if(TDIM > 2)
704  out << "nstacks " << num_nodes(2) << endl;
705 
706  out << "xllcorner " << setprecision(16) << min_corner(0) << endl;
707  if(TDIM > 1)
708  out << "yllcorner " << setprecision(16) << min_corner(1) << endl;
709  if(TDIM > 2)
710  out << "zllcorner " << setprecision(16) << min_corner(2) << endl;
711 
712  bool equlateralCells = true;
713  for(int d = 1; d < TDIM; ++d){
714  if(m_cellExtension[d] != m_cellExtension[0]){
715  equlateralCells = false;
716  break;
717  }
718  }
719 
720  if(equlateralCells)
721  out << "cellsize " << m_cellExtension[0] << endl;
722  else{
723  out << "xcellsize " << m_cellExtension[0] << endl;
724  if(TDIM > 1)
725  out << "ycellsize " << m_cellExtension[1] << endl;
726  if(TDIM > 2)
727  out << "zcellsize " << m_cellExtension[2] << endl;
728  }
729 
730  out << "NODATA_value " << no_data_value() << endl;
731 
732  const size_t numNodesTotal = num_nodes_total();
733  const size_t numNodes0 = num_nodes(0);
734  for(size_t i = 0; i < numNodesTotal; ++i){
735  out << m_data[i];
736  if((i+1) % numNodes0 != 0)
737  out << " ";
738  else
739  out << endl;
740  }
741 }
