52 inline bool LoadMesh(Mesh* obj, const char* filename)
53 {
54  const char* pSuffix = strrchr(filename, '.');
55  if(!pSuffix)
56  return false;
57 
58  if(strcmp(pSuffix, ".ugx") == 0)
59  {
60  // load from ugx
61  std::string fullFileName = FindFileInStandardPaths(filename);
62  GridReaderUGX ugxReader;
63  if(!ugxReader.parse_file(fullFileName.c_str())){
64  UG_LOG("ERROR in LoadGridFromUGX: Couldn't read file: " << fullFileName << std::endl);
65  return false;
66  }
67  else{
68  if(ugxReader.num_grids() < 1){
69  UG_LOG("ERROR in LoadGridFromUGX: File contains no grid.\n");
70  return false;
71  }
72  else{
73  ugxReader.grid(obj->grid(), 0, obj->position_attachment());
74 
75  if(ugxReader.num_subset_handlers(0) > 0)
76  ugxReader.subset_handler(obj->subset_handler(), 0, 0);
77 
78  if(ugxReader.num_subset_handlers(0) > 1)
79  ugxReader.subset_handler(obj->crease_handler(), 1, 0);
80 
81  if(ugxReader.num_selectors(0) > 0)
82  ugxReader.selector(obj->selector(), 0, 0);
83 
84  return true;
85  }
86  }
87  }
88  else{
89  return LoadGridFromFile(obj->grid(), obj->subset_handler(), filename,
90  obj->position_attachment());
91  }
92  return false;
93 }

95 inline bool SaveMesh(Mesh* obj, const char* filename)
96 {
97  const char* pSuffix = strrchr(filename, '.');
98  if(!pSuffix)
99  return false;
100 
101  if(strcmp(pSuffix, ".ugx") == 0){
102  GridWriterUGX ugxWriter;
103  ugxWriter.add_grid(obj->grid(), "defGrid", obj->position_attachment());
104  ugxWriter.add_subset_handler(obj->subset_handler(), "defSH", 0);
105  ugxWriter.add_subset_handler(obj->crease_handler(), "markSH", 0);
106  ugxWriter.add_selector(obj->selector(), "defSel", 0);
107 
108  return ugxWriter.write_to_file(filename);
109  }
110  else
111  return SaveGridToFile(obj->grid(), obj->subset_handler(), filename,
112  obj->position_attachment());
113 }

115 inline bool ExportToUG3(Mesh* obj, const char* filenamePrefix, const char* lgmName,
116  const char* problemName)
117 {
118  bool saveOk = false;
119 
120  Grid& g = obj->grid();
121  SubsetHandler& sh = obj->subset_handler();
122 
123  if(g.num_volumes() > 0){
124  // Create the subset-handlers
125  SubsetHandler shFaces(g, SHE_FACE);
126  SubsetHandler shVolumes(g, SHE_VOLUME);
127 
128  for(int i = 0; i < sh.num_subsets(); ++i){
129  shFaces.assign_subset(sh.begin<Face>(i), sh.end<Face>(i), i);
130  shVolumes.assign_subset(sh.begin<Volume>(i), sh.end<Volume>(i), i);
131  }
132 
133  saveOk = ExportGridToUG(g, shFaces, shVolumes, filenamePrefix,
134  lgmName, problemName, 0);
135  }
136  else{
137  saveOk = ExportGridToUG_2D(g, filenamePrefix, lgmName, problemName, 0, &sh);
138  }
139  return saveOk;
140 }
