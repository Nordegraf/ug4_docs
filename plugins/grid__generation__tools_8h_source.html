57 inline void CopySelectedElements(Mesh* srcMesh, Mesh* destMesh, AVertex aNewVrt)
58 {
59  Grid& srcGrid = srcMesh->grid();
60  Selector& srcSel = srcMesh->selector();
61  SubsetHandler& srcSH = srcMesh->subset_handler();
62  SubsetHandler& srcCreaseSH = srcMesh->crease_handler();
63 
64  Grid& destGrid = destMesh->grid();
65  SubsetHandler& destSH = destMesh->subset_handler();
66  SubsetHandler& destCreaseSH = destMesh->crease_handler();
67 
68  Grid::VertexAttachmentAccessor<AVertex> aaNewVrt(srcGrid, aNewVrt);
69 
70  CustomVertexGroup vrts;
71  typedef typename Grid::traits<TElem>::iterator iter_t;
72 
73  for(iter_t eiter = srcSel.begin<TElem>();
74  eiter != srcSel.end<TElem>(); ++eiter)
75  {
76  TElem* e = *eiter;
77  vrts.resize(e->num_vertices());
78  for(size_t iv = 0; iv < e->num_vertices(); ++iv)
79  vrts.set_vertex(iv, aaNewVrt[e->vertex(iv)]);
80 
81  TElem* ne = *destGrid.create_by_cloning(e, vrts);
82  destSH.assign_subset(ne, srcSH.get_subset_index(e));
83 
84  if(TElem::dim < 2)
85  destCreaseSH.assign_subset(ne, srcCreaseSH.get_subset_index(e));
86  }
87 }

92  SubsetHandler& srcSH = srcMesh->subset_handler();

94  Mesh::position_accessor_t aaPosSrc = srcMesh->position_accessor();
95 
96  Grid& destGrid = destMesh->grid();
97  SubsetHandler& destSH = destMesh->subset_handler();
98  SubsetHandler& destCreaseSH = destMesh->crease_handler();
99  Mesh::position_accessor_t aaPosDest = destMesh->position_accessor();
100 
101  AVertex aNewVrt;
102  srcGrid.attach_to_vertices(aNewVrt);
103  Grid::VertexAttachmentAccessor<AVertex> aaNewVrt(srcGrid, aNewVrt);
104 
105  for(int si = destSH.num_subsets(); si < srcSH.num_subsets(); ++si)
106  destSH.subset_info(si) = srcSH.subset_info(si);
107 
108  SelectAssociatedGridObjects(srcSel);
109 // create new vertices in destGrid
110  for(VertexIterator viter = srcSel.begin<Vertex>();
111  viter != srcSel.end<Vertex>(); ++viter)
112  {
113  Vertex* v = *viter;
114  Vertex* nv = *destGrid.create_by_cloning(v);
115  aaNewVrt[v] = nv;
116  aaPosDest[nv] = aaPosSrc[v];
117  destSH.assign_subset(nv, srcSH.get_subset_index(v));
118  destCreaseSH.assign_subset(nv, srcCreaseSH.get_subset_index(v));
119  }
120 
121  CopySelectedElements<Edge>(srcMesh, destMesh, aNewVrt);
122  CopySelectedElements<Face>(srcMesh, destMesh, aNewVrt);
123  CopySelectedElements<Volume>(srcMesh, destMesh, aNewVrt);
124 
125  srcGrid.detach_from_vertices(aNewVrt);
126 }

170  }break;

172  default:
173  UG_LOG("Can't create edge: Bad number of vertices. 2 are required.\n");

179  sh.assign_subset(e, subsetInd);

185 inline Face* CreateFace(Mesh* obj, int subsetInd)
186 {
187  using namespace std;
188  Selector& sel = obj->selector();
189  Grid& grid = obj->grid();
190  SubsetHandler& sh = obj->subset_handler();
191 
192  size_t numVrts = sel.num<Vertex>();
193 
194  if(numVrts < 3 || numVrts > 4){
195  UG_LOG("Bad number of vertices! Can't create a face element from " << numVrts << " vertices.");
196  return NULL;
197  }
198 
199  vector<Vertex*> vrts;
200  vrts.reserve(numVrts);
201  vrts.assign(sel.begin<Vertex>(), sel.end<Vertex>());
202 
203  FaceDescriptor fd(numVrts);
204  for(size_t i = 0; i < numVrts; ++i)
205  fd.set_vertex(i, vrts[i]);
206 
207  if(grid.get_face(fd)){
208  UG_LOG("A face connecting the given vertices already exists. Won't create a new one!");
209  return NULL;
210  }
211 
212  Face* f = NULL;
213  switch(numVrts){
214  case 3:{// create triangle
215  f = *grid.create<Triangle>(TriangleDescriptor(vrts[0], vrts[1], vrts[2]));
216  }break;
217 
218  case 4:{// create quadrilateral
219  f = *grid.create<Quadrilateral>(QuadrilateralDescriptor(vrts[0], vrts[1],
220  vrts[2], vrts[3]));
221  }break;
222 
223  default:
224  UG_LOG("Can't create face: Bad number of vertices. 3 or 4 are supported.\n");
225  break;
226  }
227 
228  if(f)
229  sh.assign_subset(f, subsetInd);
230 
231  return f;
232 }

235 inline Volume* CreateVolume(Mesh* obj, int subsetInd)
236 {
237  using namespace std;
238  Selector& sel = obj->selector();
239  Grid& grid = obj->grid();
240  SubsetHandler& sh = obj->subset_handler();
241 
242  size_t numVrts = sel.num<Vertex>();
243 
244  if(numVrts < 4 || numVrts > 8){
245  UG_LOG("Bad number of vertices! Can't create a volume element from " << numVrts << " vertices.");
246  return NULL;
247  }
248 
249  vector<Vertex*> vrts;
250  vrts.reserve(numVrts);
251  vrts.assign(sel.begin<Vertex>(), sel.end<Vertex>());
252 
253  VolumeDescriptor vd(numVrts);
254  for(size_t i = 0; i < numVrts; ++i)
255  vd.set_vertex(i, vrts[i]);
256 
257  if(grid.get_volume(vd)){
258  UG_LOG("A volume connecting the given vertices already exists. Won't create a new one!");
259  return NULL;
260  }
261 
262  Volume* v = NULL;
263  switch(numVrts){
264  case 4:{// create tetrahedron
265  v = *grid.create<Tetrahedron>(TetrahedronDescriptor(vrts[0], vrts[1],
266  vrts[2], vrts[3]));
267  }break;
268 
269  case 5:{// create pyramid
270  v = *grid.create<Pyramid>(PyramidDescriptor(vrts[0], vrts[1],
271  vrts[2], vrts[3], vrts[4]));
272  }break;
273 
274  case 6:{// create prism
275  v = *grid.create<Prism>(PrismDescriptor(vrts[0], vrts[1], vrts[2],
276  vrts[3], vrts[4], vrts[5]));
277  }break;
278 
279  case 8:{// create hexahedron
280  v = *grid.create<Hexahedron>(HexahedronDescriptor(vrts[0], vrts[1], vrts[2], vrts[3],
281  vrts[4], vrts[5], vrts[6], vrts[7]));
282  }break;
283 
284  default:
285  UG_LOG("Can't create volume: Bad number of vertices. 4, 5, 6, and 8 are supported.\n");
286  break;
287  }
288 
289  if(v){
290  // check and fix orientation
291  Grid::VertexAttachmentAccessor<APosition> aaPos(grid, aPosition);
292  if(!CheckOrientation(v, aaPos)){
293  grid.flip_orientation(v);
294  }
295 
296  sh.assign_subset(v, subsetInd);
297  }
298 
299  return v;
300 }

303 inline void CreatePlane(Mesh* obj, const vector3& upLeft, const vector3& upRight,
304  const vector3& lowLeft, const vector3& lowRight, int subsetInd,

306 {

309  Selector& sel = obj->selector();
310  SubsetHandler& sh = obj->subset_handler();
311 
312  sel.clear();
313  bool autoselEnabled = sel.autoselection_enabled();
314  sel.enable_autoselection(true);
315 
316 // create the vertices
317  Vertex* vrts[4];
318  for(size_t i = 0; i < 4; ++i)
319  vrts[i] = *grid.create<RegularVertex>();
320 
321  aaPos[vrts[0]] = upLeft;
322  aaPos[vrts[1]] = lowLeft;
323  aaPos[vrts[2]] = lowRight;
324  aaPos[vrts[3]] = upRight;
325 
326 // create the plane
327  if(fill)
328  grid.create<Quadrilateral>(QuadrilateralDescriptor(vrts[0], vrts[1], vrts[2], vrts[3]));
329  else{
330  for(size_t i = 0; i < 4; ++i){
331  int i0 = i;
332  int i1 = (i + 1) % 4;
333  grid.create<RegularEdge>(EdgeDescriptor(vrts[i0], vrts[i1]));
334  }
335  }
336 
337 // assign subset
338  sh.assign_subset(sel.begin<Vertex>(), sel.end<Vertex>(), subsetInd);
339  sh.assign_subset(sel.begin<Edge>(), sel.end<Edge>(), subsetInd);
340  sh.assign_subset(sel.begin<Face>(), sel.end<Face>(), subsetInd);
341 
342 // restore selector
343  sel.enable_autoselection(autoselEnabled);

347 inline void CreateCircle(Mesh* obj, const vector3& center, number radius,
348  int numRimVertices, int subsetInd, bool fill)
349 {
350  Grid& grid = obj->grid();
351  Mesh::position_accessor_t& aaPos = obj->position_accessor();
352  Selector& sel = obj->selector();
353  SubsetHandler& sh = obj->subset_handler();
354 
355  sel.clear();
356  bool autoselEnabled = sel.autoselection_enabled();
357  sel.enable_autoselection(true);
358 
359 // create the vertices
360 // create one upfront, the others in a loop
361  Vertex* centerVrt = NULL;
362  if(fill){
363  centerVrt = *grid.create<RegularVertex>();
364  aaPos[centerVrt] = center;
365  }
366  Vertex* firstVrt = *grid.create<RegularVertex>();
367  aaPos[firstVrt] = vector3(0, radius, 0);
368  VecAdd(aaPos[firstVrt], aaPos[firstVrt], center);
369  Vertex* lastVrt = firstVrt;
370  for(int i = 1; i < numRimVertices; ++i){
371  // create a new vertex
372  number ia = (float)i / (float)numRimVertices;
373  Vertex* vNew = *grid.create<RegularVertex>();
374  aaPos[vNew] = vector3(sin(2. * PI * ia), cos(2. * PI * ia), 0);
375  VecScale(aaPos[vNew], aaPos[vNew], radius);
376  VecAdd(aaPos[vNew], aaPos[vNew], center);
377 
378  // create a new triangle or a new edge
379  if(fill)
380  grid.create<Triangle>(TriangleDescriptor(centerVrt, vNew, lastVrt));
381  else
382  grid.create<RegularEdge>(EdgeDescriptor(lastVrt, vNew));
383 
384  // prepare the next iteration
385  lastVrt = vNew;
386  }
387 
388 // one triangle / edge is still missing
389  if(fill)
390  grid.create<Triangle>(TriangleDescriptor(centerVrt, firstVrt, lastVrt));
391  else
392  grid.create<RegularEdge>(EdgeDescriptor(lastVrt, firstVrt));
393 
394 // assign subset
395  sh.assign_subset(sel.begin<Vertex>(), sel.end<Vertex>(), subsetInd);
396  sh.assign_subset(sel.begin<Edge>(), sel.end<Edge>(), subsetInd);
397  sh.assign_subset(sel.begin<Face>(), sel.end<Face>(), subsetInd);
398 
399 // restore selector
400  sel.enable_autoselection(autoselEnabled);
401 }
