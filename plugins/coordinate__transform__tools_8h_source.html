114 inline void MoveAlongNormal(Mesh* obj, number offset, bool usePrecalculatedNormals)
115 {
116  Grid& grid = obj->grid();
117  Selector& sel = obj->selector();
118  Mesh::position_accessor_t& aaPos = obj->position_accessor();
119  Mesh::normal_accessor_t& aaNorm = obj->normal_accessor();
120 
121  Grid::face_traits::secure_container faces;
122  std::vector<Vertex*> vrts;
123  CollectVerticesTouchingSelection(vrts, sel);
124 
125  for(std::vector<Vertex*>::iterator iter = vrts.begin(); iter != vrts.end(); ++iter)
126  {
127  vector3& p = aaPos[*iter];
128  // calculate vertex normal by averaging face normals
129  vector3 n(0, 0, 0);
130  grid.associated_elements(faces, *iter);
131  if(usePrecalculatedNormals){
132  for(size_t i = 0; i < faces.size(); ++i)
133  VecAdd(n, n, aaNorm[faces[i]]);
134  }
135  else{
136  for(size_t i = 0; i < faces.size(); ++i){
137  vector3 fn;
138  CalculateNormal(fn, faces[i], aaPos);
139  VecAdd(n, n, fn);
140  }
141  }
142 
143  VecNormalize(n, n);
144 
145  VecScaleAdd(p, 1., p, offset, n);
146  }
147 }

154 inline void MoveVerticesAlongEdges(Mesh* obj, number relVal)
155 {
156  Grid& grid = obj->grid();
157  Selector& sel = obj->selector();
158  Mesh::position_accessor_t& aaPos = obj->position_accessor();
159  Grid::edge_traits::secure_container edges;
160 
161 // calculate offsets (do not apply, so that we correctly support cases in which
162 // both corners of an edge are selected)
163  std::vector<vector3> offsets;
164  offsets.reserve(sel.num<Vertex>());
165  lg_for_each(Vertex, vrt, sel){
166  vector3 offset(0, 0, 0);
167  number numSelEdges = 0;
168  grid.associated_elements(edges, vrt);
169  for_each_in_vec(Edge* e, edges){
170  if(!sel.is_selected(e))
171  continue;
172  vector3 dir;
173  if(e->vertex(0) == vrt)
174  VecSubtract(dir, aaPos[e->vertex(1)], aaPos[e->vertex(0)]);
175  else
176  VecSubtract(dir, aaPos[e->vertex(0)], aaPos[e->vertex(1)]);
177 
178  VecScaleAppend(offset, relVal, dir);
179  ++numSelEdges;
180  }end_for;
181 
182  if(numSelEdges > 0)
183  offset *= 1. / numSelEdges;
184  offsets.push_back(offset);
185  }lg_end_for;
186 
187 // apply offsets
188  size_t index = 0;
189  lg_for_each(Vertex, vrt, sel){
190  VecAppend(aaPos[vrt], offsets[index++]);
191  }lg_end_for;
192 }

279 inline void ConeTransform(Mesh* obj, const vector3& base, const vector3& axis,
280  number scaleAtTip)
281 {
282  Mesh::position_accessor_t& aaPos = obj->position_accessor();
283 
284  std::vector<Vertex*> vrts;
285  CollectVerticesTouchingSelection(vrts, obj->selector());
286 
287  for(std::vector<Vertex*>::iterator iter = vrts.begin(); iter != vrts.end(); ++iter)
288  {
289  vector3 v = aaPos[*iter];
290  vector3 proj(0, 0, 0);
291 
292  // project the vertex onto the ray from src along dir
293  number s = ProjectPointToRay(proj, v, base, axis);
294 
295  // s determines the scaling amount:
296  // s = 0 => scale with 1
297  // s = 1 => scale with scaleAtTip
298  // else interpolate linear
299  number scaling = (1. - s) + s * scaleAtTip;
300 
301  vector3 vtmp;
302  VecSubtract(vtmp, v, proj);
303  VecScale(vtmp, vtmp, scaling);
304  VecAdd(aaPos[*iter], proj, vtmp);
305  }
306 }

396 inline void FlattenBentQuadrilaterals(Mesh* obj, number stepSize, int numIterations)
397 {
398  Selector& sel = obj->selector();
399  Mesh::position_accessor_t& aaPos = obj->position_accessor();
400 
401  for(int i = 0; i < numIterations; ++i){
402  // iterate over all quadrilaterals
403  for(QuadrilateralIterator iter = sel.begin<Quadrilateral>();
404  iter != sel.end<Quadrilateral>(); ++iter)
405  {
406  Quadrilateral* q = *iter;
407 
408  // find the plane that minimizes the distance to the corner points
409  // of the quadrilateral.
410  vector3 p, n;
411  vector3 points[4] = {aaPos[q->vertex(0)], aaPos[q->vertex(1)],
412  aaPos[q->vertex(2)], aaPos[q->vertex(3)]};
413 
414  if(FindClosestPlane(p, n, points, 4)){
415  // find the projections of the corners and move the corner
416  // vertices a little into that direction.
417  for(size_t i = 0; i < 4; ++i){
418  vector3 proj;
419  ProjectPointToPlane(proj, points[i], p, n);
420  VecScaleAdd(aaPos[q->vertex(i)], 1. - stepSize, points[i],
421  stepSize, proj);
422  }
423  }
424  }
425  }
426 }
