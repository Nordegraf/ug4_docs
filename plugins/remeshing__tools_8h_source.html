87 inline void ExtrudeFacesWithTets(Mesh* obj, int fromSi, int toSi, const number factor)
88 {
89  using namespace std;
90  Grid& grid = obj->grid();
91  Mesh::position_accessor_t& aaPos = obj->position_accessor();
92  Selector& sel = obj->selector();
93  SubsetHandler& sh = obj->subset_handler();
94 
95  sel.clear();
96  bool autoselEnabled = sel.autoselection_enabled();
97  sel.enable_autoselection(true);
98 
99  vector<Face*> faces;
100  faces.assign(sh.begin<Face>(fromSi), sh.end<Face>(fromSi));
101  for (size_t i = 0; i < faces.size(); i++)
102  {
103  Face* f = faces.at(i);
104  vector3 p1, p2, p3;
105  vector3 pnormal, center, top;
106  number scale;
107  if (f->num_vertices() == 3)
108  {
109  p1 = aaPos[f->vertex(0)];
110  p2 = aaPos[f->vertex(1)];
111  p3 = aaPos[f->vertex(2)];
112  CalculateTriangleNormal(pnormal, p1, p2, p3);
113  RegularVertex* n = *grid.create<RegularVertex>();
114  center = CalculateCenter(f, aaPos);
115  scale = FaceArea(f, aaPos);
116  pnormal *= scale * factor;
117  VecAdd(top, center, pnormal);
118  aaPos[n] = top;
119  Tetrahedron tet(f->vertex(0), f->vertex(1), f->vertex(2), n);
120  // assumption: the first face in the tet is the base one
121  for (size_t i = 1; i < tet.num_faces(); ++i)
122  {
123  grid.register_element(tet.create_face(i));
124  }
125  }
126  else
127  {
128  UG_LOG("Face with more than 3 vertices. Doesn't work yet.");
129  }
130  }
131  sh.assign_subset(sel.begin<Vertex>(), sel.end<Vertex>(), toSi);
132  sh.assign_subset(sel.begin<Edge>(), sel.end<Edge>(), toSi);
133  sh.assign_subset(sel.begin<Face>(), sel.end<Face>(), toSi);
134  sh.assign_subset(sel.begin<Volume>(), sel.end<Volume>(), toSi);
135 
136  // restore selector
137  sel.enable_autoselection(autoselEnabled);
138  // remove doubles
139 }

389 inline void ExtrudeAndMove(Mesh* obj, const vector3& totalDir, int numSteps,
390  bool createFaces, bool createVolumes)
391 {
392  using namespace std;
393  if(numSteps < 1)
394  return;
395 
396  vector3 stepDir;
397  VecScale(stepDir, totalDir, 1./(float)numSteps);
398 
399  Grid& grid = obj->grid();
400  Selector& sel = obj->selector();
401  SubsetHandler& sh = obj->subset_handler();
402 
403  SelectAssociatedGridObjects(sel);
404 
405  vector<Vertex*> vrts;
406  vrts.assign(sel.vertices_begin(), sel.vertices_end());
407  vector<Edge*> edges;
408  edges.assign(sel.edges_begin(), sel.edges_end());
409  vector<Face*> faces;
410  faces.assign(sel.faces_begin(), sel.faces_end());
411 
412  uint extrusionOptions = 0;
413  if(createFaces)
414  extrusionOptions |= EO_CREATE_FACES;
415  if(createVolumes)
416  extrusionOptions |= EO_CREATE_VOLUMES;
417 
418 // we use sel to collect the newly created volumes
419  bool strictInheritanceEnabled = sh.strict_inheritance_enabled();
420  sh.enable_strict_inheritance(false);
421 // mark all elements that were already in the selector.
422  for(int i = 0; i < numSteps; ++i)
423  {
424  Extrude(grid, &vrts, &edges, &faces, stepDir,
425  extrusionOptions, obj->position_attachment());
426  }
427 
428  sh.enable_strict_inheritance(strictInheritanceEnabled);
429 
430 // select faces, edges and vertices from the new top-layer.
431  sel.clear<Vertex>();
432  sel.clear<Edge>();
433  sel.clear<Face>();
434  sel.clear<Volume>();
435  sel.select(vrts.begin(), vrts.end());
436  sel.select(edges.begin(), edges.end());
437  sel.select(faces.begin(), faces.end());
438 }

440 inline void ExtrudeAndScale(Mesh* obj, number totalScale, bool scaleAroundPivot,
441  int numSteps, bool createFaces, bool createVolumes)
442 {
443  using namespace std;
444  if(numSteps < 1)
445  return;
446 
447  Grid& grid = obj->grid();
448  Selector& sel = obj->selector();
449  SubsetHandler& sh = obj->subset_handler();
450 
451  SelectAssociatedGridObjects(sel);
452 
453  vector<Vertex*> vrts;
454  vrts.assign(sel.vertices_begin(), sel.vertices_end());
455  vector<Edge*> edges;
456  edges.assign(sel.edges_begin(), sel.edges_end());
457  vector<Face*> faces;
458  faces.assign(sel.faces_begin(), sel.faces_end());
459 
460  uint extrusionOptions = 0;
461  if(createFaces)
462  extrusionOptions |= EO_CREATE_FACES;
463  if(createVolumes)
464  extrusionOptions |= EO_CREATE_VOLUMES;
465 
466  Mesh::position_accessor_t aaPos = obj->position_accessor();
467  vector3 center;
468  if(scaleAroundPivot)
469  center = obj->pivot();
470  else
471  center = CalculateCenter(vrts.begin(), vrts.end(), aaPos);
472 
473  vector<vector3> from;
474  vector<vector3> stepOffsets;
475  from.reserve(vrts.size());
476  stepOffsets.reserve(vrts.size());
477  for(size_t ivrt = 0; ivrt < vrts.size(); ++ivrt){
478  from.push_back(aaPos[vrts[ivrt]]);
479  vector3 dir;
480  VecSubtract(dir, from.back(), center);
481  dir *= totalScale;
482  // create a directional vector relative to from
483  dir += center;
484  VecSubtract(dir, dir, from.back());
485  dir *= 1. / (number)numSteps;
486  stepOffsets.push_back(dir);
487  }
488 
489 // we use sel to collect the newly created volumes
490  bool strictInheritanceEnabled = sh.strict_inheritance_enabled();
491  sh.enable_strict_inheritance(false);
492 // mark all elements that were already in the selector.
493  for(int i = 0; i < numSteps; ++i)
494  {
495  Extrude(grid, &vrts, &edges, &faces, vector3(0, 0, 0),
496  extrusionOptions, obj->position_attachment());
497 
498  for(size_t ivrt = 0; ivrt < vrts.size(); ++ivrt){
499  VecScaleAdd(aaPos[vrts[ivrt]], 1, from[ivrt], i+1, stepOffsets[ivrt]);
500  }
501  }
502 
503  sh.enable_strict_inheritance(strictInheritanceEnabled);
504 
505 // select faces, edges and vertices from the new top-layer.
506  sel.clear<Vertex>();
507  sel.clear<Edge>();
508  sel.clear<Face>();
509  sel.clear<Volume>();
510  sel.select(vrts.begin(), vrts.end());
511  sel.select(edges.begin(), edges.end());
512  sel.select(faces.begin(), faces.end());
513 }

516 inline void ExtrudeAlongNormal(Mesh* obj, number totalLength,
517  int numSteps, bool createFaces, bool createVolumes)
518 {
519  using namespace std;
520  if(numSteps < 1)
521  return;
522 
523  Grid& grid = obj->grid();
524  Selector& sel = obj->selector();
525  SubsetHandler& sh = obj->subset_handler();
526 
527  SelectAssociatedGridObjects(sel);
528 
529  vector<Vertex*> vrts;
530  vrts.assign(sel.vertices_begin(), sel.vertices_end());
531  vector<Edge*> edges;
532  edges.assign(sel.edges_begin(), sel.edges_end());
533  vector<Face*> faces;
534  faces.assign(sel.faces_begin(), sel.faces_end());
535 
536  uint extrusionOptions = 0;
537  if(createFaces)
538  extrusionOptions |= EO_CREATE_FACES;
539  if(createVolumes)
540  extrusionOptions |= EO_CREATE_VOLUMES;
541 
542  Mesh::position_accessor_t aaPos = obj->position_accessor();
543  Mesh::normal_accessor_t aaNorm = obj->normal_accessor();
544 
545  vector<vector3> from;
546  vector<vector3> stepOffsets;
547  from.reserve(vrts.size());
548  stepOffsets.reserve(vrts.size());
549 
550  Grid::face_traits::secure_container assFaces;
551  Grid::edge_traits::secure_container assEdges;
552 
553  for(size_t ivrt = 0; ivrt < vrts.size(); ++ivrt){
554  Vertex* vrt = vrts[ivrt];
555  from.push_back(aaPos[vrt]);
556  vector3 n(0, 0, 0);
557  grid.associated_elements(assFaces, vrt);
558  if(assFaces.size() > 0){
559  vector3 nSel(0, 0, 0);
560  for(size_t iface = 0; iface < assFaces.size(); ++iface){
561  Face* f = assFaces[iface];
562  n += aaNorm[f];
563  if(sel.is_selected(f))
564  nSel += aaNorm[f];
565  }
566  if(VecLengthSq(nSel) > 0)
567  n = nSel;
568  }
569  else{
570  grid.associated_elements(assEdges, vrt);
571  vector3 nSel(0, 0, 0);
572  for(size_t iedge = 0; iedge < assEdges.size(); ++iedge){
573  Edge* e = assEdges[iedge];
574  vector3 tmpN;
575  VecSubtract(tmpN, aaPos[e->vertex(1)], aaPos[e->vertex(0)]);
576  number t = tmpN.x();
577  tmpN.x() = tmpN.y();
578  tmpN.y() = -t;
579  tmpN.z() = 0;
580  n += tmpN;
581  if(sel.is_selected(e))
582  nSel += tmpN;
583  }
584  if(VecLengthSq(nSel) > 0)
585  n = nSel;
586  }
587 
588  VecNormalize(n, n);
589 
590  n *= totalLength / (number)numSteps;
591  stepOffsets.push_back(n);
592  }
593 
594 // we use sel to collect the newly created volumes
595  bool strictInheritanceEnabled = sh.strict_inheritance_enabled();
596  sh.enable_strict_inheritance(false);
597 // mark all elements that were already in the selector.
598  for(int i = 0; i < numSteps; ++i)
599  {
600  Extrude(grid, &vrts, &edges, &faces, vector3(0, 0, 0),
601  extrusionOptions, obj->position_attachment());
602 
603  for(size_t ivrt = 0; ivrt < vrts.size(); ++ivrt){
604  VecScaleAdd(aaPos[vrts[ivrt]], 1, from[ivrt], i+1, stepOffsets[ivrt]);
605  }
606  }
607 
608  sh.enable_strict_inheritance(strictInheritanceEnabled);
609 
610 // select faces, edges and vertices from the new top-layer.
611  sel.clear<Vertex>();
612  sel.clear<Edge>();
613  sel.clear<Face>();
614  sel.clear<Volume>();
615  sel.select(vrts.begin(), vrts.end());
616  sel.select(edges.begin(), edges.end());
617  sel.select(faces.begin(), faces.end());
618 }
