69 inline void ScaleAroundPoint(Mesh* obj, const vector3& scale, const vector3& point)
70 {
71  Mesh::position_accessor_t& aaPos = obj->position_accessor();
72 
73  std::vector<Vertex*> vrts;
74  CollectVerticesTouchingSelection(vrts, obj->selector());
75 
76  for(std::vector<Vertex*>::iterator iter = vrts.begin(); iter != vrts.end(); ++iter)
77  {
78  vector3& v = aaPos[*iter];
79  VecSubtract(v, v, point);
80  v.x() *= scale.x();
81  v.y() *= scale.y();
82  v.z() *= scale.z();
83  VecAdd(v, v, point);
84  }
85 }

112 template <class TElem>
113 inline void SelectElementsInCylinder(Mesh* obj, const vector3& cylBase,
114  const vector3& cylTop, number radius)
115 {
116  Mesh::position_accessor_t& aaPos = obj->position_accessor();
117 
118  Grid& grid = obj->grid();
119  Selector& sel = obj->selector();
120 
121  vector3 from = cylBase;
122  vector3 dir;
123  VecSubtract(dir, cylTop, cylBase);
124 
125  for(typename Grid::traits<TElem>::iterator iter = grid.begin<TElem>();
126  iter != grid.end<TElem>(); ++iter)
127  {
128  vector3 c = CalculateCenter(*iter, aaPos);
129  vector3 p;
130  number s = ProjectPointToRay(p, c, from, dir);
131  if((s > -SMALL) && (s < (1. + SMALL))){
132  if(VecDistanceSq(p, c) <= sq(radius))
133  sel.select(*iter);
134  }
135  }
136 }
