175 inline void SelectSubset(Mesh* obj, int si, bool selVrts, bool selEdges,
176  bool selFaces, bool selVols)
177 {
178  SubsetHandler& sh = obj->subset_handler();
179  Selector& sel = obj->selector();
180 
181  if(si >= 0){
182  if(selVrts)
183  sel.select(sh.begin<Vertex>(si), sh.end<Vertex>(si));
184  if(selEdges)
185  sel.select(sh.begin<Edge>(si), sh.end<Edge>(si));
186  if(selFaces)
187  sel.select(sh.begin<Face>(si), sh.end<Face>(si));
188  if(selVols)
189  sel.select(sh.begin<Volume>(si), sh.end<Volume>(si));
190  }
191  else{
192  Grid& grid = obj->grid();
193  // subset -1 has to be selected. Those are not directly accessible.
194  if(selVrts){
195  for(VertexIterator iter = grid.vertices_begin();
196  iter != grid.vertices_end(); ++iter)
197  {
198  if(sh.get_subset_index(*iter) == -1)
199  sel.select(*iter);
200  }
201  }
202  if(selEdges){
203  for(EdgeIterator iter = grid.edges_begin();
204  iter != grid.edges_end(); ++iter)
205  {
206  if(sh.get_subset_index(*iter) == -1)
207  sel.select(*iter);
208  }
209  }
210  if(selFaces){
211  for(FaceIterator iter = grid.faces_begin();
212  iter != grid.faces_end(); ++iter)
213  {
214  if(sh.get_subset_index(*iter) == -1)
215  sel.select(*iter);
216  }
217  }
218  if(selVols){
219  for(VolumeIterator iter = grid.volumes_begin();
220  iter != grid.volumes_end(); ++iter)
221  {
222  if(sh.get_subset_index(*iter) == -1)
223  sel.select(*iter);
224  }
225  }
226  }
227 }

450 inline size_t SelectSubsetKinkVertices(Mesh* obj, int subsetIndex,
451  number thresholdAngle, bool selectDarts)
452 {
453  Grid& grid = obj->grid();
454  Selector& sel = obj->selector();
455  SubsetHandler& sh = obj->subset_handler();
456 
457  std::vector<Vertex*> candidates;
458  grid.begin_marking();
459  for(EdgeIterator eiter = sh.begin<Edge>(subsetIndex);
460  eiter != sh.end<Edge>(subsetIndex); ++eiter)
461  {
462  Edge* e = *eiter;
463  for(size_t i = 0; i < 2; ++i){
464  if(!grid.is_marked(e->vertex(i))){
465  candidates.push_back(e->vertex(i));
466  grid.mark(e->vertex(i));
467  }
468  }
469  }
470  grid.end_marking();
471 
472  size_t numSel = sel.num<Vertex>();
473 
474  SelectKinkVertices(sel, candidates.begin(), candidates.end(),
475  thresholdAngle, selectDarts, obj->position_accessor(),
476  IsInSubset(obj->subset_handler(), subsetIndex));
477 
478  return sel.num<Vertex>() - numSel;
479 }

784 inline void SelectIntersectingTriangles(Mesh* obj)
785 {
786  using namespace ug::node_tree;
787 
788  Grid& grid = obj->grid();
789  Selector& sel = obj->selector();
790 
791 // make sure that triangles are present.
792  if(grid.num<Triangle>() == 0){
793  UG_LOG("Given grid does not contain any triangles. Aborting 'Select Intersecting Triangles'.\n");
794  return;
795  }
796 
797  if(grid.num<Quadrilateral>() > 0){
798  UG_LOG("WARNING: Quadrilateral intersections are ignored during 'Select Intersecting Triangles'.\n");
799  }
800 
801 // create an octree
802 // sort the triangles of grid into an octree to speed-up projection performance
803  SPOctree octree;
804  octree = CreateOctree(grid, grid.begin<Triangle>(),
805  grid.end<Triangle>(),
806  10, 30, false, obj->position_attachment());
807 
808  if(!octree.valid()){
809  UG_LOG(" Octree creation failed in 'Select Intersecting Triangles'. Aborting.\n");
810  return;
811  }
812 
813 // access the position attachment of the grid
814  Mesh::position_accessor_t& aaPos = obj->position_accessor();
815 
816 // we'll use a traverser to find the intersections.
817  Traverser_IntersectFaces intersectionTraverser;
818 
819 // for each face the face itself and direct neighbors shall be ignored
820  std::vector<Face*> ignoreList;
821 
822  size_t triCount = 0;
823 // now iterate over all triangles of the grid and find intersections
824  for(TriangleIterator iter = grid.begin<Triangle>();
825  iter != grid.end<Triangle>(); ++iter, ++triCount)
826  {
827  Triangle* t = *iter;
828 
829  // add neighbors and the face itself to the ignore list
830  CollectNeighbors(ignoreList, t, grid, NHT_VERTEX_NEIGHBORS);
831  intersectionTraverser.clear_ignore_list();
832  for(size_t i = 0; i < ignoreList.size(); ++i){
833  intersectionTraverser.ignore_element(ignoreList[i]);
834  }
835  intersectionTraverser.ignore_element(t);
836 
837  if(intersectionTraverser.intersect_tri(aaPos[t->vertex(0)], aaPos[t->vertex(1)],
838  aaPos[t->vertex(2)], octree))
839  {
840  // check the intersecting face:
841  /*
842  const std::vector<CollisionElementID>& faces =
843  intersectionTraverser.get_intersected_element_ids();
844  */
845  // an intersection occurred. Log the index and select the triangle.
846  sel.select(t);
847  }
848  }
