81 
82  bool operator < (const ProMeshFunction& f) const
83  {
84  if(m_priority != f.priority())
85  return m_priority < f.priority();
86  if(m_exportedFunction->name() != f.exported_function()->name())
87  return m_exportedFunction->name() < f.exported_function()->name();
88  return m_exportedFunction->num_parameter() < f.exported_function()->num_parameter();
89  }
90 
91  private:
92  bridge::ExportedFunction* m_exportedFunction;
93  int m_priority;
94  unsigned int m_target;
