137 class line_sss_marker
138 {
139 // All the members are used in the fractures only!
140 
142  struct t_fract_elem
143  {
144  IVertexGroup * fract_face;
145  size_t fract_co;
146 
147  t_fract_elem (IVertexGroup * face, size_t co) : fract_face (face), fract_co (co) {};
148  };
149 
151  std::vector<t_fract_elem> m_intersections;
152 
153 public:
154 
156  line_sss_marker () {};
157 
159  void init () {m_intersections.clear ();}
160 
162  bool marked_for (IVertexGroup * elem, size_t co)
163  {
164  // is the fracture already processed?
165  for (size_t i = 0; i < m_intersections.size (); i++)
166  {
167  // check if we have already registered this mark
168  t_fract_elem& intersection = m_intersections[i];
169  if (intersection.fract_face == elem)
170  return intersection.fract_co == co;
171 
172  // check if a different corner is meant (i.e. this is the same fracture)
173  Vertex * vrt = elem->vertex (co);
174  for (size_t j = 0; j < intersection.fract_face->num_vertices (); j++)
175  if (vrt == intersection.fract_face->vertex (j))
176  return false; // in this fracture, we use a different corner
177  }
178  // no, register this fracture, too
179  m_intersections.push_back (t_fract_elem (elem, co));
180  return true;
181  }
182 };
