51  bool eraseUnusedEdges, bool eraseUnusedFaces)

57  SelectAssociatedVolumes(sel, sel.begin<Face>(), sel.end<Face>());

61 
62  if(eraseUnusedEdges)

66  SelectInnerSelectionVertices(sel);
67 
68 // erase selected elements
69  Grid& grid = obj->grid();
70  grid.erase(sel.begin<Volume>(), sel.end<Volume>());
71  grid.erase(sel.begin<Face>(), sel.end<Face>());
72  grid.erase(sel.begin<Edge>(), sel.end<Edge>());
73  grid.erase(sel.begin<Vertex>(), sel.end<Vertex>());
74 }

156 inline void CollapseEdge(Mesh* obj)
157 {
158  using namespace std;
159  Grid& grid = obj->grid();
160  Selector& sel = obj->selector();
161  Mesh::position_accessor_t& aaPos = obj->position_accessor();
162 
163  vector<Face*> vFaces;
164  vector<Edge*> vEdges;
165  while(sel.num<Edge>() > 0){
166  Edge* e = *sel.begin<Edge>();
167  // to make sure that all selected edges are collapsed,
168  // we have to check the adjacent triangles
169  CollectFaces(vFaces, grid, e);
170  for(size_t i = 0; i < vFaces.size(); ++i){
171  Face* f = vFaces[i];
172  if(f->num_edges() == 3){
173  CollectEdges(vEdges, grid, f);
174  int counter = 0;
175  for(size_t j = 0; j < 3; ++j){
176  if(sel.is_selected(vEdges[j]))
177  ++counter;
178  }
179  // if two edges are selected, we have
180  // to mark the unselected edge, too (since we
181  // don't know which will be removed).
182  if(counter == 2){
183  for(size_t j = 0; j < 3; ++j)
184  sel.select(vEdges[j]);
185  }
186  }
187  }
188 
189  // calculate the center
190  VecAdd(aaPos[e->vertex(0)], aaPos[e->vertex(0)], aaPos[e->vertex(1)]);
191  VecScale(aaPos[e->vertex(0)], aaPos[e->vertex(0)], 0.5);
192 
193  // perform collapse
194  CollapseEdge(grid, e, e->vertex(0));
195  }
196 }

198 inline void SplitEdge(Mesh* obj)
199 {
200  using namespace std;
201 // collect all edges that shall be splitted in a vector
202 // since new edges will be automatically selected again.
203  Grid& grid = obj->grid();
204  Selector& sel = obj->selector();
205  Mesh::position_accessor_t& aaPos = obj->position_accessor();
206 
207  vector<Edge*> vEdges;
208  for(EdgeIterator iter = sel.begin<Edge>();
209  iter != sel.end<Edge>(); ++iter)
210  {
211  vEdges.push_back(*iter);
212  }
213 
214 // iterate through all edges in the vector and split them
215  for(size_t i = 0; i < vEdges.size(); ++i){
216  vector3 center = CalculateCenter(vEdges[i], aaPos);
217  RegularVertex* vrt = ug::SplitEdge<RegularVertex>(grid, vEdges[i]);
218  aaPos[vrt] = center;
219  }
220 }

222 inline void SwapEdge(Mesh* obj)
223 {
224  using namespace std;
225 // collect all edges that shall be swapped in a vector
226 // since new edges will be automatically selected again.
227  Grid& grid = obj->grid();
228  Selector& sel = obj->selector();
229  vector<Edge*> vEdges;
230  for(EdgeIterator iter = sel.begin<Edge>();
231  iter != sel.end<Edge>(); ++iter)
232  {
233  vEdges.push_back(*iter);
234  }
235 
236 // iterate through all edges in the vector and swap them
237 // if they are adjacent to two triangles
238  Face* faces[2];
239  for(size_t i = 0; i < vEdges.size(); ++i){
240  int numFaces = GetAssociatedFaces(faces, grid, vEdges[i], 2);
241  if(numFaces == 2){
242  if(faces[0]->num_vertices() == 3 && faces[1]->num_vertices() == 3){
243  SwapEdge(grid, vEdges[i]);
244  }
245  }
246  }

301 inline void ResolveEdgeIntersection(Mesh* obj, number snapThreshold)
302 {
303  Grid& grid = obj->grid();
304  Selector& sel = obj->selector();
305  Mesh::position_accessor_t& aaPos = obj->position_accessor();
306 
307 // the grid may now contain some degenerated triangles. We'll try to
308 // remove most of them by projecting vertices onto close edges
309  SelectAssociatedVertices(sel, sel.begin<Edge>(), sel.end<Edge>());
310 
311 
312  ProjectVerticesToCloseEdges(grid, sel.get_grid_objects(),
313  obj->position_attachment(), snapThreshold);
314 
315  IntersectCloseEdges(grid, sel, aaPos, snapThreshold);
316 
317 // remove doubles now
318  ug::RemoveDoubles<3>(grid, sel.begin<Vertex>(), sel.end<Vertex>(),
319  obj->position_attachment(), snapThreshold);

383 inline void ResolveSelfIntersections(Mesh* obj, number snapThreshold)
384 {
385  Grid& grid = obj->grid();
386  Selector& sel = obj->selector();
387  Mesh::position_accessor_t& aaPos = obj->position_accessor();
388 
389  Triangulate(grid, sel.begin<Quadrilateral>(), sel.end<Quadrilateral>(), &aaPos);
390 
391  bool intersectFaces = sel.num<Face>() > 0;
392 
393  SelectAssociatedEdges(sel, sel.begin<Face>(), sel.end<Face>());
394  SelectAssociatedVertices(sel, sel.begin<Face>(), sel.end<Face>());
395 
396  // remove doubles again
397  ug::RemoveDoubles<3>(grid, sel.begin<Vertex>(), sel.end<Vertex>(),
398  obj->position_attachment(), snapThreshold);
399 
400  if(intersectFaces)
401  ProjectVerticesToCloseFaces(grid, sel,
402  obj->position_attachment(), snapThreshold);
403 
404  ProjectVerticesToCloseEdges(grid, sel.get_grid_objects(),
405  obj->position_attachment(), snapThreshold);
406 
407 // resolve face intersections
408  if(intersectFaces){
409  ug::ResolveTriangleIntersections(grid, sel.begin<Triangle>(),
410  sel.end<Triangle>(), snapThreshold,
411  obj->position_attachment());
412 
413  // the grid may now contain some degenerated Faces. We'll try to
414  // remove most of them by projecting vertices onto close edges
415  SelectAssociatedEdges(sel, sel.begin<Face>(), sel.end<Face>());
416  SelectAssociatedVertices(sel, sel.begin<Face>(), sel.end<Face>());
417  ProjectVerticesToCloseEdges(grid, sel.get_grid_objects(),
418  obj->position_attachment(), snapThreshold);
419  }
420  else{
421  Mesh::position_accessor_t& aaPos = obj->position_accessor();
422  IntersectCloseEdges(grid, sel, aaPos, snapThreshold);
423  }
424 
425 
426 // remove doubles again
427  ug::RemoveDoubles<3>(grid, sel.begin<Vertex>(), sel.end<Vertex>(),
428  obj->position_attachment(), snapThreshold);
429 }
