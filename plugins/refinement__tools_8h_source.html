91 inline void RefineSmooth(Mesh* obj, bool strictSubsetInheritance)
92 {
93  Grid& grid = obj->grid();
94  Selector& sel = obj->selector();
95  SubsetHandler& sh = obj->subset_handler();
96  bool siEnabled = sh.strict_inheritance_enabled();
97 
98  sh.enable_strict_inheritance(strictSubsetInheritance);
99 
100 // currently only triangles are supported in smooth refinement.
101 // convert all selected quads to triangles first.
102  Grid::VertexAttachmentAccessor<APosition> aaPos(grid, aPosition);
103  Triangulate(grid, sel.begin<Quadrilateral>(), sel.end<Quadrilateral>(), &aaPos);
104 
105 // since we use a flat hierarchy, a temporary position attachment is required
106  APosition aTmpPos;
107  grid.attach_to_vertices(aTmpPos);
108 
109  CopyAttachments(grid, sel.begin<Vertex>(),
110  sel.end<Vertex>(),
111  aPosition, aTmpPos);
112 
113  CopyAttachments(grid, grid.begin<Vertex>(), grid.end<Vertex>(), aPosition, aTmpPos);
114 
115  SubdivisionLoopProjector<APosition>
116  refCallbackLoop(grid, aPosition, aTmpPos);
117 
118  refCallbackLoop.consider_as_crease_edge(IsInSubset(obj->crease_handler(), REM_CREASE));
119  Refine(grid, sel, &refCallbackLoop);
120 
121 // copy position data of selected vertices
122  CopyAttachments(grid, sel.begin<Vertex>(),
123  sel.end<Vertex>(),
124  aTmpPos, aPosition);
125 
126  grid.detach_from_vertices(aTmpPos);
127 
128  sh.enable_strict_inheritance(siEnabled);
129 }

159 inline void CreateFractal(Mesh* obj, size_t numIterations, number scaleFac)
160 {
161  Grid& grid = obj->grid();
162 
163 // we'll use a hanging-node refiner
164  FractalProjector refCallback(grid, scaleFac);
165  HangingNodeRefiner_Grid href(grid);
166  href.set_refinement_callback(&refCallback);
167 
168 // iterate for the specified number of times
169  for(size_t i = 0; i < numIterations; ++i){
170  if(grid.num_volumes() > 0){
171  // iterate over all faces and mark them for refinement, if they are boundary faces.
172  for(FaceIterator iter = grid.faces_begin();
173  iter != grid.faces_end(); ++iter)
174  {
175  if(IsVolumeBoundaryFace(grid, *iter)){
176  href.mark(*iter);
177  }
178 
179  }
180  }
181  else if(grid.num_faces() > 0){
182  // markall faces
183  href.mark(grid.faces_begin(), grid.faces_end());
184  }
185  else{
186  // mark all edges
187  href.mark(grid.edges_begin(), grid.edges_end());
188  }
189 
190  // refine them
191  href.refine();
192 
193  // change the scalefac
194  refCallback.set_scale_fac(-0.5 * refCallback.get_scale_fac());
195  //refCallback.set_scale_fac(refCallback.get_scale_fac() * refCallback.get_scale_fac());
196  }
197 }

199 inline void InsertCenter(Mesh* obj, bool strictSubsetInheritance)
200 {
201  Grid& grid = obj->grid();
202  Selector& sel = obj->selector();
203  SubsetHandler& sh = obj->subset_handler();
204  bool siEnabled = sh.strict_inheritance_enabled();
205 
206  sh.enable_strict_inheritance(strictSubsetInheritance);
207 
208 // access position attachment
209  Grid::VertexAttachmentAccessor<APosition> aaPos(grid, aPosition);
210 
211  std::vector<Edge*> edges;
212  std::vector<Face*> faces;
213  std::vector<Volume*> vols;
214  vols.assign(sel.begin<Volume>(), sel.end<Volume>());
215 
216 //todo: support insert center for all selections
217  if(grid.num<Volume>()){
218  if(sel.num<Face>() > 0){
219  UG_LOG("InsertCenter for faces is currently not supported if"
220  " volumes are present.\n");
221  }
222  }
223  else
224  faces.assign(sel.begin<Face>(), sel.end<Face>());
225 
226 //todo: support insert center for all selections
227  if(grid.num<Face>() > 0){
228  if(sel.num<Edge>() > 0){
229  UG_LOG("InsertCenter for edges is currently not supported if"
230  " faces are present.\n");
231  }
232  }
233  else
234  edges.assign(sel.begin<Edge>(), sel.end<Edge>());
235 
236 // insert centers
237  for(size_t i = 0; i < vols.size(); ++i){
238  Volume* vol = vols[i];
239  RegularVertex* vrt = *grid.create<RegularVertex>(vol);
240  aaPos[vrt] = CalculateCenter(vol, aaPos);
241  InsertCenterVertex(grid, vol, vrt, true);
242  }
243 
244 // insert centers
245  for(size_t i = 0; i < faces.size(); ++i){
246  Face* f = faces[i];
247  RegularVertex* vrt = *grid.create<RegularVertex>(f);
248  aaPos[vrt] = CalculateCenter(f, aaPos);
249  InsertCenterVertex(grid, f, vrt, true);
250  }
251 
252 // split edges
253  for(size_t i = 0; i < edges.size(); ++i){
254  Edge* e = edges[i];
255  vector3 center = CalculateCenter(e, aaPos);
256  RegularVertex* vrt = SplitEdge<RegularVertex>(grid, e);
257  aaPos[vrt] = center;
258  }
259 
260  sh.enable_strict_inheritance(siEnabled);
261 }
