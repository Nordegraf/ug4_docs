94 inline void RefineSmooth(Mesh* obj, bool strictSubsetInheritance)
95 {
96  Grid& grid = obj->grid();
97  Selector& sel = obj->selector();
98  SubsetHandler& sh = obj->subset_handler();
99  bool siEnabled = sh.strict_inheritance_enabled();
100 
101  sh.enable_strict_inheritance(strictSubsetInheritance);
102 
103 // currently only triangles are supported in smooth refinement.
104 // convert all selected quads to triangles first.
105  Grid::VertexAttachmentAccessor<APosition> aaPos(grid, aPosition);
106  Triangulate(grid, sel.begin<Quadrilateral>(), sel.end<Quadrilateral>(), &aaPos);
107 
108  SubdivisionProjector refProj(MakeGeometry3d(grid, aPosition),
109  Grid::edge_traits::callback(
110  IsInSubset(obj->crease_handler(), REM_CREASE)));
111  Refine(grid, sel, &refProj);
112 
113  sh.enable_strict_inheritance(siEnabled);
114 }

116 inline void InsertCenter(Mesh* obj, bool strictSubsetInheritance)
117 {
118  Grid& grid = obj->grid();
119  Selector& sel = obj->selector();
120  SubsetHandler& sh = obj->subset_handler();
121  bool siEnabled = sh.strict_inheritance_enabled();
122 
123  sh.enable_strict_inheritance(strictSubsetInheritance);
124 
125 // access position attachment
126  Grid::VertexAttachmentAccessor<APosition> aaPos(grid, aPosition);
127 
128  std::vector<Edge*> edges;
129  std::vector<Face*> faces;
130  std::vector<Volume*> vols;
131  vols.assign(sel.begin<Volume>(), sel.end<Volume>());
132 
133 //todo: support insert center for all selections
134  if(grid.num<Volume>()){
135  if(sel.num<Face>() > 0){
136  UG_LOG("InsertCenter for faces is currently not supported if"
137  " volumes are present.\n");
138  }
139  }
140  else
141  faces.assign(sel.begin<Face>(), sel.end<Face>());
142 
143 //todo: support insert center for all selections
144  if(grid.num<Face>() > 0){
145  if(sel.num<Edge>() > 0){
146  UG_LOG("InsertCenter for edges is currently not supported if"
147  " faces are present.\n");
148  }
149  }
150  else
151  edges.assign(sel.begin<Edge>(), sel.end<Edge>());
152 
153 // insert centers
154  for(size_t i = 0; i < vols.size(); ++i){
155  Volume* vol = vols[i];
156  RegularVertex* vrt = *grid.create<RegularVertex>(vol);
157  aaPos[vrt] = CalculateCenter(vol, aaPos);
158  InsertCenterVertex(grid, vol, vrt, true);
159  }
160 
161 // insert centers
162  for(size_t i = 0; i < faces.size(); ++i){
163  Face* f = faces[i];
164  RegularVertex* vrt = *grid.create<RegularVertex>(f);
165  aaPos[vrt] = CalculateCenter(f, aaPos);
166  InsertCenterVertex(grid, f, vrt, true);
167  }
168 
169 // split edges
170  for(size_t i = 0; i < edges.size(); ++i){
171  Edge* e = edges[i];
172  vector3 center = CalculateCenter(e, aaPos);
173  RegularVertex* vrt = SplitEdge<RegularVertex>(grid, e);
174  aaPos[vrt] = center;
175  }
176 
177  sh.enable_strict_inheritance(siEnabled);
178 }
