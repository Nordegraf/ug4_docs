66 template <typename TDomain>
67 void
68 HookeLaw<TDomain>::
69 stressTensor(MathMatrix<dim,dim>& stressTens, const size_t ip,
70  const MathMatrix<dim, dim>& GradU)
71 {
72  HOOKE_PROFILE_BEGIN(HookeLaw_stressTensor);
73 
74  // get linearized strain tensor (eps) at ip
75  MathMatrix<dim, dim> strainTens;
76  strainTensor(strainTens, GradU);
77 
78  // TODO: replace this with general implementation of TensContractMat
79  for(size_t i = 0; i < (size_t) dim; ++i)
80  for(size_t j = 0; j < (size_t) dim; ++j)
81  {
82  stressTens[i][j] = 0.0;
83 
84  for(size_t k = 0; k < (size_t) dim; ++k)
85  for(size_t l = 0; l < (size_t) dim; ++l)
86  stressTens[i][j] += (*m_spElastTensorFunct)[i][j][k][l]
87  * strainTens[k][l];
88 
89  }
90 
91 }

225 template <typename TDomain>
226 void
227 HookeLaw<TDomain>::
228 set_hooke_elasticity_tensor(const number lambda, const number mu)
229 {
230  // sets the 'Hooke'-elasticity tensor for isotropic materials
231  // in 2D this tensor formulation corresponds to the
232  // plane strain assumption for Hookes`s law
233 
234  MathTensor4<dim,dim,dim,dim> elastTensorFunct;
235 
236  // filling the constant elasticity tensor
237  for (size_t i = 0; i < (size_t) dim; ++i)
238  for (size_t j = 0; j < (size_t) dim; ++j)
239  for (size_t k = 0; k < (size_t) dim; ++k)
240  for (size_t l = 0; l < (size_t) dim; ++l)
241  {
242  elastTensorFunct[i][j][k][l] = 0.0;
243 
244  if ((i == j) && (k == l))
245  elastTensorFunct[i][j][k][l] += lambda;
246 
247  if ((i == k) && (j == l))
248  elastTensorFunct[i][j][k][l] += mu;
249 
250  if ((i == l) && (j == k))
251  elastTensorFunct[i][j][k][l] += mu;
252 
253  } //end (l)
254 
255  // remembering the elasticity tensor
256  SmartPtr<MathTensor4<dim,dim,dim,dim> > spElastTens(new MathTensor4<dim,dim,dim,dim>(elastTensorFunct));
257  m_spElastTensorFunct = spElastTens;
258 
259  number E = mu * (3.0 * lambda + 2.0 * mu)/(lambda + mu);
260  number v = 0.5 * lambda/(lambda + mu);
261  number kappa = lambda + 2.0/3.0 * mu;
262 
263  std::stringstream ss;
264  ss << "Hooke Elasticity Tensor: \n";
265  ss << " Lame`s first constant lambda: " << lambda << "\n";
266  ss << " Lame`s second constant mue (sometimes 'G', shear modulus) (Schubmodul): " << mu << "\n";
267  ss << " This setting equals: \n";
268  ss << " young modulus (Elastizitaetsmodul): " << E << "\n";
269  ss << " poisson ratio (Querkontraktionszahl) v: " << v << "\n";
270  ss << " bulk modulus (Kompressionsmodul): " << kappa << "\n";
271  ss << " Elasticity Tensor = " << elastTensorFunct << "\n";
272  m_materialConfiguration = ss.str();
273 }
