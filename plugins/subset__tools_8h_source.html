203 inline void AssignSubsetsByQuality(Mesh* obj, int numSections)
204 {
205  using namespace std;
206  std::vector<number> intervals;
207  intervals.push_back(0);
208  for(int i = 1; i < numSections; ++i)
209  intervals.push_back((number)i / (number)numSections);
210  intervals.push_back(1.);
211 
212  Grid& grid = obj->grid();

214  SubsetHandler& sh = obj->subset_handler();
215 
216  ug::AssignSubsetsByQuality(grid, sh, sel.begin<Face>(),
217  sel.end<Face>(), intervals);
218 
219 // log how many faces were assigned to the different subsets.
220 // since potentially only a subset of faces has been considered,
221 // we may not simply output the subset sizes.
222 
223  UG_LOG("Assigned faces to subsets:\n");
224  for(size_t i = 0; i < intervals.size() - 1; ++i){
225  // count the number of selected faces in this section
226  size_t counter = 0;
227  for(FaceIterator iter = sel.begin<Face>(); iter != sel.end<Face>(); ++iter)
228  {
229  if(sh.get_subset_index(*iter) == (int)i)
230  ++counter;
231  }
232 
233  UG_LOG(" quality " << intervals[i] << " - " << intervals[i+1] << ": \t" << counter << "\n");
234  }

236  UG_LOG(endl);
237 }

239 inline void SeparateDegeneratedBoundaryFaceSubsets(Mesh* obj, number angle)
240 {
241  using namespace std;
242  number thresholdDot = cos(deg_to_rad(angle));
243 
244  Grid& g = obj->grid();
245  SubsetHandler& sh = obj->subset_handler();
246  Mesh::position_accessor_t& aaPos = obj->position_accessor();
247 
248  vector<Edge*> edges;
249  vector<Edge*> edges2;
250  vector<Face*> faces;
251  vector<Face*> assembledSubset;
252  queue<Face*> queFaces;
253 
254 // we'll use this vector to check whether we have to assign faces which
255 // we assembled to a subset to a new subset or whether it can stay where it
256 // is. The first assembled face-pack can always stay in its subset.
257  vector<bool> vAssignNewSubset(sh.num_subsets(), false);
258 
259 // the index at which we'll add new subsets (increases during the algorithm)
260  int newSI = GetMaxSubsetIndex<Face>(sh) + 1;
261 
262 // we use marks to mark all processed elements
263  g.begin_marking();
264 
265 // iterate over all faces and search for a degenerated boundary face.
266  for(FaceIterator iter = g.begin<Face>(); iter != g.end<Face>(); ++iter){
267  Face* f = *iter;
268  // the face may have been marked during subset assembly below
269  if(g.is_marked(f))
270  continue;
271  g.mark(f);
272 
273  if(IsDegenerated(f, aaPos)){
274  if(IsVolumeBoundaryFace(g, f)){
275  // the face is a candidate. Get subset index and push it to the queue
276  int origSI = sh.get_subset_index(f);
277  queFaces.push(f);
278  assembledSubset.clear();
279 
280  while(!queFaces.empty()){
281  Face* curFace = queFaces.front();
282  queFaces.pop();
283  assembledSubset.push_back(curFace);
284 
285  // check all degenerated neighbor faces
286  CollectAssociated(edges, g, curFace);
287 
288  vector3 dir(0, 0, 0);
289  bool gotOne = false;
290  // the first non-degenerated edge defines the direction of the face
291  for(size_t i_edge = 0; i_edge < edges.size(); ++i_edge){
292  Edge* e = edges[i_edge];
293  if(VecDistanceSq(aaPos[e->vertex(0)], aaPos[e->vertex(1)]) >= SMALL*SMALL){
294  VecSubtract(dir, aaPos[e->vertex(1)], aaPos[e->vertex(0)]);
295  VecNormalize(dir, dir);
296  gotOne = true;
297  break;
298  }
299  }
300 
301  // if we haven't found a non-degenerated edge, we won't continue.
302  if(!gotOne)
303  continue;
304 
305  // now find associated degenerated faces
306  for(size_t i_edge = 0; i_edge < edges.size(); ++i_edge){
307  Edge* e = edges[i_edge];
308 
309  // we have to know whether the edge is degenerated or not.
310  bool bDegEdge = (VecDistanceSq(aaPos[e->vertex(0)], aaPos[e->vertex(1)]) < SMALL*SMALL);
311 
312  CollectAssociated(faces, g, e);
313 
314  for(size_t i_face = 0; i_face < faces.size(); ++i_face){
315  Face* nbrFace = faces[i_face];
316  if(!g.is_marked(nbrFace) && sh.get_subset_index(nbrFace) == origSI){
317  if(IsVolumeBoundaryFace(g, nbrFace)){
318  if(IsDegenerated(nbrFace, aaPos)){
319  // if the edge was non-degenerated, it is automatically part of the
320  // assembled subset.
321  if(!bDegEdge){
322  g.mark(nbrFace);
323  queFaces.push(nbrFace);
324  }
325  else{
326  // we have to compare the directions of the faces
327  CollectAssociated(edges2, g, nbrFace);
328 
329  vector3 dir2(0, 0, 0);
330  bool gotOne2 = false;
331  // the first non-degenerated edge defines the direction of the face
332  for(size_t i_edge = 0; i_edge < edges2.size(); ++i_edge){
333  Edge* e = edges2[i_edge];
334  if(VecDistanceSq(aaPos[e->vertex(0)], aaPos[e->vertex(1)]) >= SMALL*SMALL){
335  VecSubtract(dir2, aaPos[e->vertex(1)], aaPos[e->vertex(0)]);
336  VecNormalize(dir2, dir2);
337  gotOne2 = true;
338  break;
339  }
340  }
341 
342  if(gotOne2){
343  // we now got both directions. compare the angle.
344  if(fabs(VecDot(dir, dir2)) >= thresholdDot){
345  // both are in the same subset
346  g.mark(nbrFace);
347  queFaces.push(nbrFace);
348  }
349  }
350  }
351  }
352  }
353  }
354  }
355  }
356 
357  }
358 
359  // now add the assembledSubset to its new destination
360  if(vAssignNewSubset[origSI]){
361  sh.assign_subset(assembledSubset.begin(), assembledSubset.end(), newSI);
362  ++newSI;
363  }
364  else{
365  // if more degenerated faces are contained in this subset, they shall be
366  // assigned to another subset.
367  vAssignNewSubset[origSI] = true;
368  }
369  }
370  }
371  }
372 
373  g.end_marking();
374 }
