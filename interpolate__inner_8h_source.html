66 template <typename TElem, typename TGridFunction>
67 void InterpolateOnElementsInner(
68  SmartPtr<UserData<number, TGridFunction::dim> > spInterpolFunction,
69  SmartPtr<TGridFunction> spGridFct, size_t fct, int si, number time)
70 {
71  // do nothing if no DoFs here
72  if (!spGridFct->max_fct_dofs(fct, ROID_TETRAHEDRON, si))
73  return;
74 
75  // domain type and position_type
76  typedef typename TGridFunction::domain_type domain_type;
77  typedef typename domain_type::position_type position_type;
78 
79  // get iterators
80  typename TGridFunction::template traits<TElem>::const_iterator iterEnd, iter;
81  iterEnd = spGridFct->template end<TElem>(si);
82  iter = spGridFct->template begin<TElem>(si);
83 
84  // check if something to do:
85  if (iter == iterEnd) return;
86 
87  // id of shape functions used
88  LFEID id = spGridFct->local_finite_element_id(fct);
89 
90  // iterate over all elements
91  for (; iter != iterEnd; ++iter)
92  {
93  // get element
94  TElem* elem = *iter;
95 
96  // get multiindices of element
97  std::vector<DoFIndex> ind;
98  spGridFct->inner_dof_indices(elem, fct, ind);
99 
100  // global positions of DoFs
101  std::vector<position_type> glob_pos;
102  InnerDoFPosition<domain_type>(glob_pos, elem, *spGridFct->domain(), id);
103 
104  // check global positions size
105  size_t ind_sz = ind.size();
106  size_t gp_sz = glob_pos.size();
107  if (ind_sz != gp_sz)
108  UG_THROW("InterpolateOnElem: On subset " << si << ": Number of DoFs is "
109  << ind_sz << ", but number of DoF positions is "
110  << gp_sz << "." << std::endl);
111 
112  // loop all dofs
113  for (size_t i = 0; i < ind_sz && i < gp_sz; ++i)
114  {
115  // value at position
116  number val;
117  (*spInterpolFunction)(val, glob_pos[i], time, si);
118 
119  // set value
120  DoFRef(*spGridFct, ind[i]) = val;
121  }
122  }
123 }

137 template <typename TGridFunction>
138 void InterpolateOnElementsInner
139 (
140  SmartPtr<UserData<number, TGridFunction::dim> > spInterpolFunction,
141  SmartPtr<TGridFunction> spGridFct,
142  size_t fct,
143  number time,
144  const SubsetGroup& ssGrp
145 )
146 {
147 // loop subsets
148  for(size_t i = 0; i < ssGrp.size(); ++i)
149  {
150  // get subset index
151  const int si = ssGrp[i];
152 
153  // skip if function is not defined in subset
154  if (!spGridFct->is_def_in_subset(fct, si)) continue;
155 
156  // switch dimensions
157  try
158  {
159  const int dim = ssGrp.dim(i);
160 
161  if (dim > TGridFunction::dim)
162  UG_THROW("InterpolateOnElements: Dimension of subset is " << dim << ", but "
163  " World Dimension is " << TGridFunction::dim << ". Cannot interpolate this.");
164 
165  // In a parallel scenario, the distribution CAN cause elements of of lower
166  // dimension than the rest of their subset to be located disconnected from
167  // the rest of the subset on a processor. For example, in 2D, think of a
168  // (1D) boundary subset and a distribution where the boundary of a proc's
169  // domain only touches the boundary subset in a vertex, but intersects with
170  // the boundary subset in another place.
171  // Therefore, we need to interpolate on all dimensions < dim and we will
172  // only consider inner indices then, thus we simply switch-case backwards
173  // and without breaks.
174  switch (dim)
175  {
176  case DIM_SUBSET_EMPTY_GRID: break;
177  case 3:
178  if (spGridFct->max_fct_dofs(fct, VOLUME, si))
179  {
180  InterpolateOnElementsInner<Tetrahedron, TGridFunction>
181  (spInterpolFunction, spGridFct, fct, si, time);
182  InterpolateOnElementsInner<Hexahedron, TGridFunction>
183  (spInterpolFunction, spGridFct, fct, si, time);
184  InterpolateOnElementsInner<Prism, TGridFunction>
185  (spInterpolFunction, spGridFct, fct, si, time);
186  InterpolateOnElementsInner<Pyramid, TGridFunction>
187  (spInterpolFunction, spGridFct, fct, si, time);
188  InterpolateOnElementsInner<Octahedron, TGridFunction>
189  (spInterpolFunction, spGridFct, fct, si, time);
190  }
191  case 2:
192  if (spGridFct->max_fct_dofs(fct, FACE, si))
193  {
194  InterpolateOnElementsInner<Triangle, TGridFunction>
195  (spInterpolFunction, spGridFct, fct, si, time);
196  InterpolateOnElementsInner<Quadrilateral, TGridFunction>
197  (spInterpolFunction, spGridFct, fct, si, time);
198  InterpolateOnElementsInner<ConstrainingTriangle, TGridFunction>
199  (spInterpolFunction, spGridFct, fct, si, time);
200  InterpolateOnElementsInner<ConstrainingQuadrilateral, TGridFunction>
201  (spInterpolFunction, spGridFct, fct, si, time);
202  InterpolateOnElementsInner<ConstrainedTriangle, TGridFunction>
203  (spInterpolFunction, spGridFct, fct, si, time);
204  InterpolateOnElementsInner<ConstrainedQuadrilateral, TGridFunction>
205  (spInterpolFunction, spGridFct, fct, si, time);
206  }
207  case 1:
208  if (spGridFct->max_fct_dofs(fct, EDGE, si))
209  {
210  InterpolateOnElementsInner<RegularEdge, TGridFunction>
211  (spInterpolFunction, spGridFct, fct, si, time);
212  InterpolateOnElementsInner<ConstrainingEdge, TGridFunction>
213  (spInterpolFunction, spGridFct, fct, si, time);
214  InterpolateOnElementsInner<ConstrainedEdge, TGridFunction>
215  (spInterpolFunction, spGridFct, fct, si, time);
216  }
217  case 0:
218  if (spGridFct->max_fct_dofs(fct, VERTEX, si))
219  {
220  InterpolateOnElementsInner<RegularVertex, TGridFunction>
221  (spInterpolFunction, spGridFct, fct, si, time);
222  InterpolateOnElementsInner<ConstrainedVertex, TGridFunction>
223  (spInterpolFunction, spGridFct, fct, si, time);
224  }
225  break;
226  default: UG_THROW("InterpolateOnElements: Dimension " <<dim<<
227  " not possible for world dim "<<3<<".");
228  }
229  }
230  UG_CATCH_THROW("InterpolateOnElements: Cannot interpolate on elements.");
231  }
232 }

257 template <typename TGridFunction>
258 void InterpolateInner(SmartPtr<UserData<number, TGridFunction::dim> > spInterpolFunction,
259  SmartPtr<TGridFunction> spGridFct, const char* cmp,
260  const char* subsets, number time)
261 {
262  // check, that values do not depend on a solution
263  if (spInterpolFunction->requires_grid_fct())
264  UG_THROW("Interpolate: The interpolation values depend on a grid function."
265  " This is not allowed in the current implementation. Use constant,"
266  " lua-callback or vrl-callback user data only (even within linkers).");
267 
268  // get function id of name
269  const size_t fct = spGridFct->fct_id_by_name(cmp);
270 
271  // check that function found
272  if (fct > spGridFct->num_fct())
273  UG_THROW("Interpolate: Name of component '"<< cmp <<"' not found.");
274 
275  // check that type is Lagrange
276  if (spGridFct->local_finite_element_id(fct).type() != LFEID::LAGRANGE)
277  UG_THROW("This interpolation only allows Lagrange-type elements.\n"
278  "Feel free to add functionality for other types as needed.");
279 
280  // check if fast P1 interpolation can be used
281  const bool bUseP1Interpolation
282  = spGridFct->local_finite_element_id(fct).order() == 1;
283 
284  // create subset group
285  SubsetGroup ssGrp(spGridFct->domain()->subset_handler());
286  if (subsets != NULL)
287  ssGrp.add(TokenizeString(subsets));
288  else
289  ssGrp.add_all();
290 
291  // forward
292  if (bUseP1Interpolation)
293  InterpolateOnVertices<TGridFunction>(spInterpolFunction, spGridFct, fct, time, ssGrp);
294  else
295  InterpolateOnElementsInner<TGridFunction>(spInterpolFunction, spGridFct, fct, time, ssGrp);
296 
297  // adjust parallel storage state
298 #ifdef UG_PARALLEL
299  spGridFct->set_storage_type(PST_CONSISTENT);
300 #endif
301 }
