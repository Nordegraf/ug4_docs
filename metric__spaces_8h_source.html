98  *delta -= y;
99  return norm(*delta);

212  ss << this->m_fctNames << ", " << this->m_ssNames << ", " << this->m_quadorder
213  << ", type=" << std::endl;
214 
215  else
216  ss << this->m_fctNames << ", (no name), " << this->m_quadorder
217  << ", type=" << std::endl;
218  return ss.str();
