144 template <typename TAlgebra>
145 bool NewtonSolver<TAlgebra>::apply(vector_type& u)
146 {
147  NEWTON_PROFILE_BEGIN(NewtonSolver_apply);
148 // increase call count
149  m_dgbCall++;
150 
151 // Check for linear solver
152  if(m_spLinearSolver.invalid())
153  UG_THROW("NewtonSolver::apply: Linear Solver not set.");
154 
155 // Jacobian
156  if(m_J.invalid() || m_J->discretization() != m_spAss) {
157  m_J = make_sp(new AssembledLinearOperator<TAlgebra>(m_spAss));
158  }
159  m_J->set_level(m_N->level());
160 
161 // create tmp vectors
162  SmartPtr<vector_type> spD = u.clone_without_values();
163  SmartPtr<vector_type> spC = u.clone_without_values();
164 
165 // Set dirichlet values
166  try{
167  m_N->prepare(u);
168  }
169  UG_CATCH_THROW("NewtonSolver::prepare: Prepare of Operator failed.");
170 
171 // Compute first Defect
172  try{
173  NEWTON_PROFILE_BEGIN(NewtonComputeDefect1);
174  m_N->apply(*spD, u);
175  NEWTON_PROFILE_END();
176  }UG_CATCH_THROW("NewtonSolver::apply: Computation of Start-Defect failed.");
177 
178 // write start defect for debug
179  int loopCnt = 0;
180  m_lastNumSteps = 0;
181  char ext[20]; sprintf(ext, "_iter%03d", loopCnt);
182  std::string name("NEWTON_Defect");
183  name.append(ext);
184  write_debug(*spD, name.c_str());
185  write_debug(u, "NEWTON_StartSolution");
186 
187 // increase offset of output for linear solver
188  const int stdLinOffset = m_spLinearSolver->standard_offset();
189  m_spLinearSolver->convergence_check()->set_offset(stdLinOffset + 3);
190 
191 // set info string indicating the used linear solver
192  std::stringstream ss; ss << "(Linear Solver: " << m_spLinearSolver->name() << ")";
193  m_spConvCheck->set_info(ss.str());
194 
195 // start convergence check
196  m_spConvCheck->start(*spD);
197 
198  for(size_t i = 0; i < m_stepUpdate.size(); ++i)
199  m_stepUpdate[i]->update();
200 
201 // loop iteration
202  while(!m_spConvCheck->iteration_ended())
203  {
204  m_lastNumSteps = loopCnt;
205 
206  // set c = 0
207  NEWTON_PROFILE_BEGIN(NewtonSetCorretionZero);
208  spC->set(0.0);
209  NEWTON_PROFILE_END();
210 
211  for(size_t i = 0; i < m_innerStepUpdate.size(); ++i)
212  m_innerStepUpdate[i]->update();
213 
214  // Compute Jacobian
215  try{
216  NEWTON_PROFILE_BEGIN(NewtonComputeJacobian);
217  m_J->init(u);
218  NEWTON_PROFILE_END();
219  }UG_CATCH_THROW("NewtonSolver::apply: Initialization of Jacobian failed.");
220 
221  // Write Jacobian for debug
222  std::string matname("NEWTON_Jacobian");
223  matname.append(ext);
224  write_debug(m_J->get_matrix(), matname.c_str());
225 
226  // Init Jacobi Inverse
227  try{
228  NEWTON_PROFILE_BEGIN(NewtonPrepareLinSolver);
229  if(!m_spLinearSolver->init(m_J, u))
230  {
231  UG_LOG("ERROR in 'NewtonSolver::apply': Cannot init Inverse Linear "
232  "Operator for Jacobi-Operator.\n");
233  return false;
234  }
235  NEWTON_PROFILE_END();
236  }UG_CATCH_THROW("NewtonSolver::apply: Initialization of Linear Solver failed.");
237 
238  // Solve Linearized System
239  try{
240  NEWTON_PROFILE_BEGIN(NewtonApplyLinSolver);
241  if(!m_spLinearSolver->apply(*spC, *spD))
242  {
243  UG_LOG("ERROR in 'NewtonSolver::apply': Cannot apply Inverse Linear "
244  "Operator for Jacobi-Operator.\n");
245  return false;
246  }
247  NEWTON_PROFILE_END();
248  }UG_CATCH_THROW("NewtonSolver::apply: Application of Linear Solver failed.");
249 
250  // store convergence history
251  const int numSteps = m_spLinearSolver->step();
252  if(loopCnt >= (int)m_vTotalLinSolverSteps.size()) m_vTotalLinSolverSteps.resize(loopCnt+1);
253  if(loopCnt >= (int)m_vLinSolverCalls.size()) m_vLinSolverCalls.resize(loopCnt+1, 0);
254  if(loopCnt >= (int)m_vLinSolverRates.size()) m_vLinSolverRates.resize(loopCnt+1, 0);
255  m_vTotalLinSolverSteps[loopCnt] += numSteps;
256  m_vLinSolverCalls[loopCnt] += 1;
257  m_vLinSolverRates[loopCnt] += m_spLinearSolver->convergence_check()->avg_rate();
258 
259  // Line Search
260  try{
261  if(m_spLineSearch.valid())
262  {
263  m_spLineSearch->set_offset(" # ");
264  NEWTON_PROFILE_BEGIN(NewtonLineSearch);
265  if(!m_spLineSearch->search(m_N, u, *spC, *spD, m_spConvCheck->defect()))
266  {
267  UG_LOG("ERROR in 'NewtonSolver::apply': "
268  "Newton Solver did not converge.\n");
269  return false;
270  }
271  NEWTON_PROFILE_END();
272  }
273  // No line search: Compute new defect
274  else
275  {
276  // update solution
277  u -= *spC;
278 
279  // compute new Defect
280  NEWTON_PROFILE_BEGIN(NewtonComputeDefect);
281  m_N->prepare(u);
282  m_N->apply(*spD, u);
283  NEWTON_PROFILE_END();
284  }
285  }UG_CATCH_THROW("NewtonSolver::apply: Line Search update failed.");
286 
287 
288  // update counter
289  loopCnt++;
290  sprintf(ext, "_iter%03d", loopCnt);
291 
292  // check convergence
293  m_spConvCheck->update(*spD);
294  if(loopCnt-1 >= (int)m_vNonLinSolverRates.size()) m_vNonLinSolverRates.resize(loopCnt, 0);
295  m_vNonLinSolverRates[loopCnt-1] += m_spConvCheck->rate();
296 
297  // write defect for debug
298  std::string name("NEWTON_Defect"); name.append(ext);
299  write_debug(*spD, name.c_str());
300  std::string name2("NEWTON_Correction"); name2.append(ext);
301  write_debug(*spC, name2.c_str());
302  std::string name3("NEWTON_Solution"); name3.append(ext);
303  write_debug(u, name3.c_str());
304  }
305 
306  // reset offset of output for linear solver to previous value
307  m_spLinearSolver->convergence_check()->set_offset(stdLinOffset);
308 
309  return m_spConvCheck->post();
310 }

312 template <typename TAlgebra>
313 void NewtonSolver<TAlgebra>::print_average_convergence() const
314 {
315  UG_LOG("\nNewton solver convergence history:\n");
316  UG_LOG("Newton Step | Num Calls | Total Lin Iters | Avg Lin Iters | Avg Nonlin Rates | Avg Lin Rates \n");
317  int allCalls = 0, allLinSteps = 0;
318  number allLinRatesProduct = 1.0, allNonLinRatesProduct = 1.0;
319  for(int call = 0; call < (int)m_vLinSolverCalls.size(); ++call)
320  {
321  UG_LOG( " " << std::setw(10) << call+1 << " | ");
322  UG_LOG(std::setw(9) << m_vLinSolverCalls[call] << " | ");
323  allCalls += m_vLinSolverCalls[call];
324  UG_LOG(std::setw(15) << m_vTotalLinSolverSteps[call] << " | ");
325  allLinSteps += m_vTotalLinSolverSteps[call];
326  UG_LOG(std::setw(13) << std::setprecision(2) << std::fixed << m_vTotalLinSolverSteps[call] / (double)m_vLinSolverCalls[call] << " | ");
327  allNonLinRatesProduct *= pow((number)m_vNonLinSolverRates[call]/(double)m_vLinSolverCalls[call],(double)m_vLinSolverCalls[call]);
328  UG_LOG(std::setw(16) << std::setprecision(6) << std::scientific << m_vNonLinSolverRates[call] / (double)m_vLinSolverCalls[call] << " | ");
329  allLinRatesProduct *= (number)std::pow((number)m_vLinSolverRates[call]/(double)m_vLinSolverCalls[call],(number)m_vTotalLinSolverSteps[call]);
330  UG_LOG(std::setw(13) << std::setprecision(6) << std::scientific << m_vLinSolverRates[call] / (double)m_vLinSolverCalls[call]);
331  UG_LOG("\n");
332  }
333  UG_LOG( " all | ");
334  UG_LOG(std::setw(9) << allCalls << " | ");
335  UG_LOG(std::setw(15) << allLinSteps << " | ");
336  UG_LOG(std::setw(13) << std::setprecision(2) << std::fixed << allLinSteps / (number)allCalls << " | ");
337  UG_LOG(std::setw(16) << std::setprecision(6) << std::scientific << std::pow((number)allNonLinRatesProduct,(number)1.0/(number)allCalls) << " | ");
338  UG_LOG(std::setw(13) << std::setprecision(6) << std::scientific << std::pow((number)allLinRatesProduct,(number)1.0/(number)allLinSteps));
339  UG_LOG("\n");
340 }
