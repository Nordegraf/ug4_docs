331  {
332  // check that not already registered
333  for(size_t i = 0; i < m_vDomainElemDisc.size(); ++i)
334  if(m_vDomainElemDisc[i] == elem)
335  return;

351  void remove(SmartPtr<IElemDisc<TDomain> > elem)
352  {
353  // check that already registered
354  for (size_t i = 0; i < m_vDomainElemDisc.size(); i++)
355  {
356  if (m_vDomainElemDisc[i] == elem)
357  {
358  // remove constraint
