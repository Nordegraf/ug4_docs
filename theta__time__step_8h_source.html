169 class ThetaTimeStep
170  : public MultiStepTimeDiscretization<TAlgebra>
171 {
172  public:

197  : MultiStepTimeDiscretization<TAlgebra>(spDD),
198  m_stage(1), m_scheme("Theta")
199  {
200  set_theta(theta);
201  this->m_prevSteps = 1;
202  }
203 
205  ThetaTimeStep(SmartPtr<IDomainDiscretization<TAlgebra> > spDD, const char* scheme)
206  : MultiStepTimeDiscretization<TAlgebra>(spDD),
207  m_stage(1), m_scheme(scheme)
208  {
209  set_theta(1.0);
210  this->m_prevSteps = 1;
211  }
212 
213  virtual ~ThetaTimeStep() {};
214 
216  void set_scheme(const char* scheme) {m_scheme = scheme;}
217 
219  virtual size_t num_stages() const
220  {
221  if (m_scheme == "Theta") return 1;
222  else if (m_scheme == "Alexander") return 2;
223  else if (m_scheme == "FracStep") return 3;
224  else UG_THROW("Step Scheme not recognized.");
225  }
226 
228  virtual void set_stage(size_t stage) {m_stage = stage;}
229 
231  void set_theta(number theta) {m_theta = theta;}
232 
233  protected:
234  virtual number update_scaling(std::vector<number>& vSM,
235  std::vector<number>& vSA,
236  number dt, number currentTime,
237  ConstSmartPtr<VectorTimeSeries<vector_type> > prevSol)
238  {
239  // resize scaling factors
240  vSM.resize(2);
241  vSM[0] = 1.;
242  vSM[1] = -1.;
243 
244  if(m_scheme == "Theta")
245  {
246  vSA.resize(2);
247  vSA[0] = (m_theta) * dt;
248  vSA[1] = (1.- m_theta) * dt;
249  return currentTime + dt;
250  }
251  else if(m_scheme == "Alexander")
252  {
253  vSA.resize(2);
254  const number gamma = 1 - 1. / sqrt(2.);
255  switch(m_stage)
256  {
257  case 1:
258  vSA[0] = gamma * dt;
259  vSA[1] = 0;
260  return currentTime + gamma * dt;
261  case 2:
262  vSA[0] = gamma * dt;
263  vSA[1] = (1.- 2*gamma) * dt;
264  return currentTime + (1 - gamma) * dt;
265  default:
266  UG_THROW("Alexander scheme has only 2 stages")
267  }
268  }
269  else if(m_scheme == "FracStep")
270  {
271  vSA.resize(2);
272  switch(m_stage)
273  {
274  case 1:
275  vSA[0] = (2.-sqrt(2.)) * (1-1./sqrt(2.)) * dt;
276  vSA[1] = (1.- (2.-sqrt(2.)))* (1-1./sqrt(2.)) * dt;
277  return currentTime + (1-1./sqrt(2.)) * dt;
278  case 2:
279  vSA[0] = (sqrt(2.)-1) * (sqrt(2.)-1) * dt;
280  vSA[1] = (1.- (sqrt(2.)-1)) * (sqrt(2.)-1) * dt;
281  return currentTime + (sqrt(2.)-1) * dt;
282  case 3:
283  vSA[0] = (2.-sqrt(2.)) * (1-1./sqrt(2.)) * dt;
284  vSA[1] = (1.- (2.-sqrt(2.)))* (1-1./sqrt(2.)) * dt;
285  return currentTime + (1-1./sqrt(2.)) * dt;
286  default:
287  UG_THROW("FracStep scheme has only 3 stages")
288  }
289  }
290  else
291  UG_THROW("Unknown Multi-Stage Theta Scheme: "<< m_scheme<<".");
292 
293  }
294 
295  number m_theta;
296 
297  size_t m_stage;
298 
299  std::string m_scheme;
300 };

339 
341  virtual size_t num_stages() const {return 1;}
342

345  {
346  if(stage!=1) UG_THROW("BDF has only one stage.");
347  }
348 
349  protected:
350  virtual number update_scaling(std::vector<number>& vSM,
351  std::vector<number>& vSA,
352  number dt, number currentTime,
353  ConstSmartPtr<VectorTimeSeries<vector_type> > prevSol)
354  {
355  // resize scaling factors
356  vSM.resize(m_order+1);
357  vSA.resize(m_order+1);
358 
359  // future time
360  const number futureTime = currentTime + dt;
361 
362  // get time points
363  if(prevSol->size() < m_order)
364  UG_THROW("BDF("<<m_order<<") needs at least "<< m_order <<
365  " previous solutions, but only "<<prevSol->size()<<"passed.");
366 
367  std::vector<number> vTimePoint(m_order+1);
368  vTimePoint[0] = futureTime;
369  for(size_t i = 1; i <= m_order; ++i)
370  vTimePoint[i] = prevSol->time(i-1);
371 
372  // evaluate derivative of Lagrange Polynoms at future time
373  for(size_t i = 0; i <= m_order; ++i)
374  {
375  vSM[i] = 0;
376 
377  for(size_t j = 0; j < vTimePoint.size(); ++j)
378  {
379  if(j == i) continue;
380 
381  number prod = 1;
382  for(size_t k = 0; k < vTimePoint.size(); ++k)
383  {
384  if(k == i) continue;
385  if(k == j) continue;
386 
387  prod *= (vTimePoint[0]-vTimePoint[k])/
388  (vTimePoint[i]-vTimePoint[k]);
389  }
390  prod *= 1./(vTimePoint[i]-vTimePoint[j]);
391 
392  vSM[i] += prod;
393  }
394  }
395 
396  // only first value of vSA != 0
397  const number scale = 1.0 / vSM[0];
398  vSA[0] = scale;
399  for(size_t i = 1; i <= m_order; ++i) vSA[i] = 0;
400 
401  // scale first s_m to 1.0
402  for(int i = m_order; i >= 0; --i) vSM[i] *= scale;
403 
404  return futureTime;
405  }
406 
407  size_t m_order;
408 };
