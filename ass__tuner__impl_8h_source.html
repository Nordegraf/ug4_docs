94 template <typename TAlgebra>
95 void AssemblingTuner<TAlgebra>::set_dirichlet_row(matrix_type& mat, const DoFIndex& ind) const
96 {
97  // check if assembling has been carried out with respect to one index only.
98  // For that case assembling-matrices have been resized to a block-matrix at one DoF only.
99  if(single_index_assembling_enabled())
100  {
101  if (mat.num_rows() != 1 || mat.num_cols() != 1)
102  UG_THROW("#rows and #cols need to be 1 for setting dirichlet rows"
103  " in an index-wise manner.")
104 
105  const size_t index = ind[0];
106  if (index == m_SingleAssIndex)
107  SetDirichletRow(mat, 0, ind[1]);
108  }
109  else{
110  SetDirichletRow(mat, ind);
111  }
112 }

116 {
117  // check if assembling has been carried out with respect to one index only.
