68 // Reference-Object IDs
70 enum ReferenceObjectID
71 {
72  ROID_UNKNOWN = -1,
73  ROID_VERTEX,
74  ROID_EDGE,
75  ROID_TRIANGLE,
76  ROID_QUADRILATERAL,
77  ROID_TETRAHEDRON,
78  ROID_HEXAHEDRON,
79  ROID_PRISM,
80  ROID_PYRAMID,
81  ROID_OCTAHEDRON,
82  NUM_REFERENCE_OBJECTS
83 };
84 
85 inline
86 ReferenceObjectID operator++(ReferenceObjectID& roid, int)
87 {
88  int tmp = roid;
89  return roid = static_cast<ReferenceObjectID>(++tmp);
90 };
91 
93 inline
94 std::ostream& operator<< (std::ostream& outStream, ReferenceObjectID type)
95 {
96  switch(type)
97  {
98  case ROID_UNKNOWN: outStream << "(invalid)"; break;
99  case ROID_VERTEX: outStream << "Vertex"; break;
100  case ROID_EDGE: outStream << "Edge"; break;
101  case ROID_TRIANGLE: outStream << "Triangle"; break;
102  case ROID_QUADRILATERAL: outStream << "Quadrilateral"; break;
103  case ROID_TETRAHEDRON: outStream << "Tetrahedron"; break;
104  case ROID_OCTAHEDRON: outStream << "Octahedron"; break;
105  case ROID_HEXAHEDRON: outStream << "Hexahedron"; break;
106  case ROID_PRISM: outStream << "Prism"; break;
107  case ROID_PYRAMID: outStream << "Pyramid"; break;
108  default: throw(UGError("Unknown ReferenceObjectID in operator<<"));
109  }
110  return outStream;
111 };
