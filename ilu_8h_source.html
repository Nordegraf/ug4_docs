50 // ILU(0) solver, i.e. static pattern ILU w/ P=P(A)
51 // (cf. Y Saad, Iterative methods for Sparse Linear Systems, p. 270)
52 template<typename Matrix_type>
53 bool FactorizeILU(Matrix_type &A)
54 {
55  PROFILE_FUNC_GROUP("algebra ILU");
56  typedef typename Matrix_type::row_iterator row_iterator;
57  typedef typename Matrix_type::value_type block_type;
58 
59  // for all rows
60  for(size_t i=1; i < A.num_rows(); i++)
61  {
62  // eliminate all entries A(i, k) with k<i with rows A(k, .) and k<i
63  const row_iterator rowEnd = A.end_row(i);
64  for(row_iterator it_k = A.begin_row(i);
65  it_k != rowEnd && (it_k.index() < i); ++it_k)
66  {
67  const size_t k = it_k.index();
68  block_type &a_ik = it_k.value();
69 
70  // add row k to row i by A(i, .) -= A(k,.) A(i,k) / A(k,k)
71  // so that A(i,k) is zero.
72  // save A(i,k)/A(k,k) in A(i,k)
73  if(fabs(BlockNorm(A(k,k))) < 1e-15*BlockNorm(A(i,k)))
74  UG_THROW("Diag is Zero for k="<<k<<", cannot factorize ILU.");
75 
76  a_ik /= A(k,k);
77 
78  row_iterator it_j = it_k;
79  for(++it_j; it_j != rowEnd; ++it_j)
80  {
81  const size_t j = it_j.index();
82  block_type& a_ij = it_j.value();
83  bool bFound;
84  row_iterator p = A.get_connection(k,j, bFound);
85  if(bFound)
86  {
87  const block_type a_kj = p.value();
88  a_ij -= a_ik *a_kj;
89  }
90  }
91  }
92  }
93 
94  return true;
95 }

97 // ILU(0)-beta solver, i.e.
98 // -> static pattern ILU w/ P=P(A)
99 // -> Fill-in is computed and lumped onto the diagonal
100 template<typename Matrix_type>
101 bool FactorizeILUBeta(Matrix_type &A, number beta)
102 {
103  PROFILE_FUNC_GROUP("algebra ILUBeta");
104  typedef typename Matrix_type::row_iterator row_iterator;
105  typedef typename Matrix_type::value_type block_type;
106 
107  // for all rows i=1:n do
108  for(size_t i=1; i < A.num_rows(); i++)
109  {
110  block_type &Aii = A(i,i);
111  block_type Nii(Aii); Nii*=0.0;
112 
113  // for k=1:(i-1) do
114  // eliminate all entries A(i, k) with k<i with rows A(k, .) and k<i
115  const row_iterator it_iEnd = A.end_row(i);
116  for(row_iterator it_ik = A.begin_row(i);
117  it_ik != it_iEnd && (it_ik.index() < i); ++it_ik)
118  {
119 
120  // add row k to row i by A(i, .) -= [A(i,k) / A(k,k)] A(k,.)
121  // such that A(i,k) is zero.
122 
123  // 1) Contribution to L part:
124  // store A(i,k)/A(k,k) in A(i,k)
125  const size_t k = it_ik.index();
126  block_type &a_ik = it_ik.value();
127  a_ik /= A(k,k);
128 
129  // 2) Contribution to U part:
130  // compute contributions from row k for j=k:N
131  const row_iterator it_kEnd = A.end_row(k);
132  for (row_iterator it_kj=A.begin_row(k); it_kj != it_kEnd ;++it_kj)
133  {
134  const size_t j = it_kj.index();
135  if (j<i) continue; // index j belongs L part?
136 
137  // this index j belongs U part
138  const block_type& a_kj = it_kj.value();
139 
140  bool aijFound;
141  row_iterator pij = A.get_connection(i,j, aijFound);
142  if(aijFound) {
143  // entry belongs to pattern
144  // -> proceed with standard elimination
145  block_type a_ij = pij.value();
146  a_ij -= a_ik *a_kj ;
147 
148  } else {
149  // entry DOES NOT belong to pattern
150  // -> we lump it onto the diagonal
151  // TODO : non square matrices!!!
152  Nii -= a_ik * a_kj;
153  }
154 
155  }
156  }
157 
158  // add fill-in to diagonal
159  AddMult(Aii, beta, Nii);
160  }
161 
162  return true;
163 }

165 template<typename Matrix_type>
166 bool FactorizeILUSorted(Matrix_type &A, const number eps = 1e-50)
167 {
168  PROFILE_FUNC_GROUP("algebra ILU");
169  typedef typename Matrix_type::row_iterator row_iterator;
170  typedef typename Matrix_type::value_type block_type;
171 
172  // for all rows
173  for(size_t i=1; i < A.num_rows(); i++)
174  {
175 
176  // eliminate all entries A(i, k) with k<i with rows A(k, .) and k<i
177  for(row_iterator it_k = A.begin_row(i);
178  it_k != A.end_row(i) && (it_k.index() < i); ++it_k)
179  {
180  const size_t k = it_k.index();
181  block_type &a_ik = it_k.value();
182  block_type &a_kk = A(k,k);
183 
184  // add row k to row i by A(i, .) -= A(k,.) A(i,k) / A(k,k)
185  // so that A(i,k) is zero.
186  // safe A(i,k)/A(k,k) in A(i,k)
187  if(fabs(BlockNorm(A(k,k))) < eps * BlockNorm(A(i,k)))
188  UG_THROW("ILU: Blocknorm of diagonal is near-zero for k="<<k<<
189  " with eps: "<< eps <<", ||A_kk||="<<fabs(BlockNorm(A(k,k)))
190  <<", ||A_ik||="<<BlockNorm(A(i,k)));
191 
192  a_ik /= a_kk;
193 
194 
195  typename Matrix_type::row_iterator it_ij = it_k; // of row i
196  ++it_ij; // skip a_ik
197  typename Matrix_type::row_iterator it_kj = A.begin_row(k); // of row k
198 
199  while(it_ij != A.end_row(i) && it_kj != A.end_row(k))
200  {
201  if(it_ij.index() > it_kj.index())
202  ++it_kj;
203  else if(it_ij.index() < it_kj.index())
204  ++it_ij;
205  else
206  {
207  block_type &a_ij = it_ij.value();
208  const block_type &a_kj = it_kj.value();
209  a_ij -= a_ik * a_kj;
210  ++it_kj; ++it_ij;
211  }
212  }
213  }
214  }
215 
216  return true;
217 }

220 // solve x = L^-1 b
221 template<typename Matrix_type, typename Vector_type>
222 bool invert_L(const Matrix_type &A, Vector_type &x, const Vector_type &b)
223 {
224  PROFILE_FUNC_GROUP("algebra ILU");
225  typedef typename Matrix_type::const_row_iterator const_row_iterator;
226 
227  typename Vector_type::value_type s;
228  for(size_t i=0; i < x.size(); i++)
229  {
230  s = b[i];
231  for(const_row_iterator it = A.begin_row(i); it != A.end_row(i); ++it)
232  {
233  if(it.index() >= i) continue;
234  MatMultAdd(s, 1.0, s, -1.0, it.value(), x[it.index()]);
235  }
236  x[i] = s;
237  }
238 
239  return true;
240 }

242 // solve x = U^-1 * b
243 template<typename Matrix_type, typename Vector_type>
244 bool invert_U(const Matrix_type &A, Vector_type &x, const Vector_type &b,
245  const number eps = 1e-8)
246 {
247  PROFILE_FUNC_GROUP("algebra ILU");
248  typedef typename Matrix_type::const_row_iterator const_row_iterator;
249 
250  typename Vector_type::value_type s;
251 
252  size_t numNearZero=0;
253 
254  // last row diagonal U entry might be close to zero with corresponding close to zero rhs
255  // when solving Navier Stokes system, therefore handle separately
256  if(x.size() > 0)
257  {
258  size_t i=x.size()-1;
259  s = b[i];
260 
261  // check if diag part is significantly smaller than rhs
262  // This may happen when matrix is indefinite with one eigenvalue
263  // zero. In that case, the factorization on the last row is
264  // nearly zero due to round-off errors. In order to allow ill-
265  // scaled matrices (i.e. small matrix entries row-wise) this
266  // is compared to the rhs, that is small in this case as well.
267  if (BlockNorm(A(i,i)) <= eps * BlockNorm(s))
268  {
269  if(numNearZero++<5)
270  { UG_LOG("ILU Warning: Near-zero diagonal entry "
271  "with norm "<<BlockNorm(A(i,i))<<" in last row of U "
272  " with corresponding non-near-zero rhs with norm "
273  << BlockNorm(s) << ". Setting rhs to zero.\n");
274  UG_LOG("NOTE: Call this method with a smaller 'eps' parameter "
275  "to avoid this warning. (current eps: " << eps <<
276  "). If this method is called from the "
277  "ILU preconditioner class, you may want to call "
278  "ILU::set_inversion_eps(...) with a smaller threshold.\n")
279  }
280  // set correction to zero
281  x[i] = 0;
282  } else {
283  // c[i] = s/uii;
284  InverseMatMult(x[i], 1.0, A(i,i), s);
285  }
286  }
287  if(x.size() <= 1) return true;
288 
289  // handle all other rows
290  for(size_t i = x.size()-2; ; --i)
291  {
292  s = b[i];
293  for(const_row_iterator it = A.begin_row(i); it != A.end_row(i); ++it)
294  {
295  if(it.index() <= i) continue;
296  // s -= it.value() * x[it.index()];
297  MatMultAdd(s, 1.0, s, -1.0, it.value(), x[it.index()]);
298 
299  }
300  // x[i] = s/A(i,i);
301  InverseMatMult(x[i], 1.0, A(i,i), s);
302  if(i == 0) break;
303  }
304 
305  if(numNearZero>=5)
306  { UG_LOG("...\nILU Warning: " << numNearZero << " ( out of " << x.size() << ") near-zero diagonal entries in last row of U.\n"); }
307 
308  return true;
309 }
