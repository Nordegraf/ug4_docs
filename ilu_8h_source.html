53 // ILU(0) solver, i.e. static pattern ILU w/ P=P(A)
54 // (cf. Y Saad, Iterative methods for Sparse Linear Systems, p. 270)
55 template<typename Matrix_type>
56 bool FactorizeILU(Matrix_type &A)
57 {
58  PROFILE_FUNC_GROUP("algebra ILU");
59  typedef typename Matrix_type::row_iterator row_iterator;
60  typedef typename Matrix_type::value_type block_type;
61 
62  // for all rows
63  for(size_t i=1; i < A.num_rows(); i++)
64  {
65  // eliminate all entries A(i, k) with k<i with rows A(k, .) and k<i
66  const row_iterator rowEnd = A.end_row(i);
67  for(row_iterator it_k = A.begin_row(i);
68  it_k != rowEnd && (it_k.index() < i); ++it_k)
69  {
70  const size_t k = it_k.index();
71  block_type &a_ik = it_k.value();
72 
73  // add row k to row i by A(i, .) -= A(k,.) A(i,k) / A(k,k)
74  // so that A(i,k) is zero.
75  // save A(i,k)/A(k,k) in A(i,k)
76  if(fabs(BlockNorm(A(k,k))) < 1e-15*BlockNorm(A(i,k)))
77  UG_THROW("Diag is Zero for k="<<k<<", cannot factorize ILU.");
78 
79  a_ik /= A(k,k);
80 
81  row_iterator it_j = it_k;
82  for(++it_j; it_j != rowEnd; ++it_j)
83  {
84  const size_t j = it_j.index();
85  block_type& a_ij = it_j.value();
86  bool bFound;
87  row_iterator p = A.get_connection(k,j, bFound);
88  if(bFound)
89  {
90  const block_type a_kj = p.value();
91  a_ij -= a_ik *a_kj;
92  }
93  }
94  }
95  }
96 
97  return true;
98 }

100 // ILU(0)-beta solver, i.e.
101 // -> static pattern ILU w/ P=P(A)
102 // -> Fill-in is computed and lumped onto the diagonal
103 template<typename Matrix_type>
104 bool FactorizeILUBeta(Matrix_type &A, number beta)
105 {
106  PROFILE_FUNC_GROUP("algebra ILUBeta");
107  typedef typename Matrix_type::row_iterator row_iterator;
108  typedef typename Matrix_type::value_type block_type;
109 
110  // for all rows i=1:n do
111  for(size_t i=1; i < A.num_rows(); i++)
112  {
113  block_type &Aii = A(i,i);
114  block_type Nii(Aii); Nii*=0.0;
115 
116  // for k=1:(i-1) do
117  // eliminate all entries A(i, k) with k<i with rows A(k, .) and k<i
118  const row_iterator it_iEnd = A.end_row(i);
119  for(row_iterator it_ik = A.begin_row(i);
120  it_ik != it_iEnd && (it_ik.index() < i); ++it_ik)
121  {
122 
123  // add row k to row i by A(i, .) -= [A(i,k) / A(k,k)] A(k,.)
124  // such that A(i,k) is zero.
125 
126  // 1) Contribution to L part:
127  // store A(i,k)/A(k,k) in A(i,k)
128  const size_t k = it_ik.index();
129  block_type &a_ik = it_ik.value();
130  a_ik /= A(k,k);
131 
132  // 2) Contribution to U part:
133  // compute contributions from row k for j=k:N
134  const row_iterator it_kEnd = A.end_row(k);
135  for (row_iterator it_kj=A.begin_row(k); it_kj != it_kEnd ;++it_kj)
136  {
137  const size_t j = it_kj.index();
138  if (j<=k) continue; // index j belongs L part?
139 
140  // this index j belongs U part
141  const block_type& a_kj = it_kj.value();
142 
143  bool aijFound;
144  row_iterator pij = A.get_connection(i,j, aijFound);
145  if(aijFound) {
146  // entry belongs to pattern
147  // -> proceed with standard elimination
148  block_type& a_ij = pij.value();
149  a_ij -= a_ik *a_kj ;
150 
151  } else {
152  // entry DOES NOT belong to pattern
153  // -> we lump it onto the diagonal
154  // TODO : non square matrices!!!
155  Nii -= a_ik * a_kj;
156  }
157 
158  }
159  }
160 
161  // add fill-in to diagonal
162  AddMult(Aii, beta, Nii);
163  }
164 
165  return true;
166 }

168 template<typename Matrix_type>
169 bool FactorizeILUSorted(Matrix_type &A, const number eps = 1e-50)
170 {
171  PROFILE_FUNC_GROUP("algebra ILU");
172  typedef typename Matrix_type::row_iterator row_iterator;
173  typedef typename Matrix_type::value_type block_type;
174 
175  // for all rows
176  for(size_t i=1; i < A.num_rows(); i++)
177  {
178 
179  // eliminate all entries A(i, k) with k<i with rows A(k, .) and k<i
180  for(row_iterator it_k = A.begin_row(i);
181  it_k != A.end_row(i) && (it_k.index() < i); ++it_k)
182  {
183  const size_t k = it_k.index();
184  block_type &a_ik = it_k.value();
185  block_type &a_kk = A(k,k);
186 
187  // add row k to row i by A(i, .) -= A(k,.) A(i,k) / A(k,k)
188  // so that A(i,k) is zero.
189  // safe A(i,k)/A(k,k) in A(i,k)
190  if(fabs(BlockNorm(A(k,k))) < eps * BlockNorm(A(i,k)))
191  UG_THROW("ILU: Blocknorm of diagonal is near-zero for k="<<k<<
192  " with eps: "<< eps <<", ||A_kk||="<<fabs(BlockNorm(A(k,k)))
193  <<", ||A_ik||="<<BlockNorm(A(i,k)));
194 
195  try {a_ik /= a_kk;}
196  UG_CATCH_THROW("Failed to calculate A_ik /= A_kk "
197  "with i = " << i << " and k = " << k << ".");
198 
199 
200  typename Matrix_type::row_iterator it_ij = it_k; // of row i
201  ++it_ij; // skip a_ik
202  typename Matrix_type::row_iterator it_kj = A.begin_row(k); // of row k
203 
204  while(it_ij != A.end_row(i) && it_kj != A.end_row(k))
205  {
206  if(it_ij.index() > it_kj.index())
207  ++it_kj;
208  else if(it_ij.index() < it_kj.index())
209  ++it_ij;
210  else
211  {
212  block_type &a_ij = it_ij.value();
213  const block_type &a_kj = it_kj.value();
214  a_ij -= a_ik * a_kj;
215  ++it_kj; ++it_ij;
216  }
217  }
218  }
219  }
220 
221  return true;
222 }

225 // solve x = L^-1 b
226 template<typename Matrix_type, typename Vector_type>
227 bool invert_L(const Matrix_type &A, Vector_type &x, const Vector_type &b)
228 {
229  PROFILE_FUNC_GROUP("algebra ILU");
230  typedef typename Matrix_type::const_row_iterator const_row_iterator;
231 
232  typename Vector_type::value_type s;
233  for(size_t i=0; i < x.size(); i++)
234  {
235  s = b[i];
236  for(const_row_iterator it = A.begin_row(i); it != A.end_row(i); ++it)
237  {
238  if(it.index() >= i) continue;
239  MatMultAdd(s, 1.0, s, -1.0, it.value(), x[it.index()]);
240  }
241  x[i] = s;
242  }
243 
244  return true;
245 }

247 // solve x = U^-1 * b
248 template<typename Matrix_type, typename Vector_type>
249 bool invert_U(const Matrix_type &A, Vector_type &x, const Vector_type &b,
250  const number eps = 1e-8)
251 {
252  PROFILE_FUNC_GROUP("algebra ILU");
253  typedef typename Matrix_type::const_row_iterator const_row_iterator;
254 
255  typename Vector_type::value_type s;
256 
257  size_t numNearZero=0;
258 
259  // last row diagonal U entry might be close to zero with corresponding close to zero rhs
260  // when solving Navier Stokes system, therefore handle separately
261  if(x.size() > 0)
262  {
263  size_t i=x.size()-1;
264  s = b[i];
265 
266  // check if diag part is significantly smaller than rhs
267  // This may happen when matrix is indefinite with one eigenvalue
268  // zero. In that case, the factorization on the last row is
269  // nearly zero due to round-off errors. In order to allow ill-
270  // scaled matrices (i.e. small matrix entries row-wise) this
271  // is compared to the rhs, that is small in this case as well.
272  if (BlockNorm(A(i,i)) <= eps * BlockNorm(s))
273  {
274  if(numNearZero++<5)
275  { UG_LOG("ILU Warning: Near-zero diagonal entry "
276  "with norm "<<BlockNorm(A(i,i))<<" in last row of U "
277  " with corresponding non-near-zero rhs with norm "
278  << BlockNorm(s) << ". Setting rhs to zero.\n");
279  UG_LOG("NOTE: Call this method with a smaller 'eps' parameter "
280  "to avoid this warning. (current eps: " << eps <<
281  "). If this method is called from the "
282  "ILU preconditioner class, you may want to call "
283  "ILU::set_inversion_eps(...) with a smaller threshold.\n")
284  }
285  // set correction to zero
286  x[i] = 0;
287  } else {
288  // c[i] = s/uii;
289  InverseMatMult(x[i], 1.0, A(i,i), s);
290  }
291  }
292  if(x.size() <= 1) return true;
293 
294  // handle all other rows
295  for(size_t i = x.size()-2; ; --i)
296  {
297  s = b[i];
298  for(const_row_iterator it = A.begin_row(i); it != A.end_row(i); ++it)
299  {
300  if(it.index() <= i) continue;
301  // s -= it.value() * x[it.index()];
302  MatMultAdd(s, 1.0, s, -1.0, it.value(), x[it.index()]);
303 
304  }
305  // x[i] = s/A(i,i);
306  InverseMatMult(x[i], 1.0, A(i,i), s);
307  if(i == 0) break;
308  }
309 
310  if(numNearZero>=5)
311  { UG_LOG("...\nILU Warning: " << numNearZero << " ( out of " << x.size() << ") near-zero diagonal entries in last row of U.\n"); }
312 
313  return true;
314 }

317 #ifdef UG_PARALLEL
318 inline void
319 LayoutEntriesToEndPermutation(std::vector<size_t>& newIndexOut,
320  const IndexLayout& layout,
321  const size_t numRows)
322 {
323  using namespace std;
324  vector<size_t> inds;
325  CollectElements(inds, layout);
326 
327  newIndexOut.clear();
328  newIndexOut.resize(numRows, 0);
329 
330  size_t numUniqueInds = 0;
331  for(size_t i = 0; i < inds.size(); ++i){
332  if(newIndexOut[inds[i]] != 1){
333  newIndexOut[inds[i]] = 1;
334  ++numUniqueInds;
335  }
336  }
337 
338  size_t curNormalInd = 0;
339  size_t curSlaveInd = numRows - numUniqueInds;
340 
341 // 0 marks normal index, 1 marks layout index
342  for(size_t irow = 0; irow < numRows; ++irow){
343  if(newIndexOut[irow] == 0)
344  newIndexOut[irow] = curNormalInd++;
345  else
346  newIndexOut[irow] = curSlaveInd++;
347  }
348 }

468  write_debug(mat, "ILU_prep_01_A_BeforeMakeUnique");
469 
470  m_ILU = mat;
471 
472  // this is an experimental trigger. Best to leave it false for now.
473  const bool sortSlaveToEnd = false;
474 
475  #ifdef UG_PARALLEL
476  if(m_useOverlap){
477  CreateOverlap(m_ILU);
478  m_oD.set_layouts(m_ILU.layouts());

490  // slave-overlap rows shall be at the end of the matrix
491  if(sortSlaveToEnd){
492  m_bSort = true;
493  LayoutEntriesToEndPermutation( m_newIndex,
494  m_ILU.layouts()->slave(),
495  m_ILU.num_rows());
496  m_bSortIsIdentity = GetInversePermutation(m_newIndex, m_oldIndex);
497 
498  if(!m_bSortIsIdentity)
499  {
500  matrix_type mat;
501  mat = m_ILU;
502  SetMatrixAsPermutation(m_ILU, mat, m_newIndex);
503  }
504  }
505  }
506  else if(m_useConsistentInterfaces){
507  MatMakeConsistentOverlap0(m_ILU);
508  }
509  else {
510  MatAddSlaveRowsToMasterRowOverlap0(m_ILU);

516  #endif
517 
518  m_h.resize(m_ILU.num_cols());
519 
520  write_debug(m_ILU, "ILU_prep_02_A_AfterMakeUnique");
521 
522  // if using overlap we already sort in a different way
523  if(m_bSort && !(m_useOverlap && sortSlaveToEnd))
524  calc_cuthill_mckee();

526  // Debug output of matrices
527  write_debug(m_ILU, "ILU_prep_03_A_BeforeFactorize");
528 
529  // Compute ILU Factorization
530  if (m_beta!=0.0) FactorizeILUBeta(m_ILU, m_beta);
531  else if(matrix_type::rows_sorted) FactorizeILUSorted(m_ILU, m_sortEps);

533  m_ILU.defragment();
534 
535  // Debug output of matrices
536  write_debug(m_ILU, "ILU_prep_04_A_AfterFactorize");
537 
538  // we're done
539  return true;
540  }

543  void applyLU(vector_type &c, const vector_type &d, vector_type &tmp)
544  {
545  if(!m_bSort || m_bSortIsIdentity)
546  {
547  // apply iterator: c = LU^{-1}*d
548  invert_L(m_ILU, tmp, d); // h := L^-1 d
549  invert_U(m_ILU, c, tmp, m_invEps); // c := U^-1 h = (LU)^-1 d
550  }
551  else
552  {
553  // we save one vector here by renaming
554  SetVectorAsPermutation(tmp, d, m_newIndex);
555  invert_L(m_ILU, c, tmp); // c = L^{-1} d
556  invert_U(m_ILU, tmp, c, m_invEps); // tmp = (LU)^{-1} d
557  SetVectorAsPermutation(c, tmp, m_oldIndex);
558  }
559  }
