85 template <typename TGrid, typename TSubsetHandler>
86 void IDomain<TGrid,TSubsetHandler>::
87 update_subset_infos(int rootProc)
88 {
89  PROFILE_FUNC();
90 
91  TSubsetHandler& sh = *m_spSH;
92  for(int i = 0; i < sh.num_subsets(); ++i){
93  int dim = -1;
94  if(sh.contains_volumes(i))
95  dim = 3;
96  else if(sh.contains_faces(i))
97  dim = 2;
98  else if(sh.contains_edges(i))
99  dim = 1;
100  else if(sh.contains_vertices(i))
101  dim = 0;
102 
103  sh.subset_info(i).set_property("dim", dim);
104  }
105 
106 #ifdef UG_PARALLEL
107  pcl::ProcessCommunicator procCom;
108 
109 // prepare the subset-info package, send it to all procs and extract the info again.
110  BinaryBuffer buf;
111  if(pcl::ProcRank() == rootProc){
112  Serialize(buf, sh.num_subsets());
113  for(int i = 0; i < sh.num_subsets(); ++i){
114  Serialize(buf, sh.subset_info(i).name);
115  Serialize(buf, sh.subset_info(i).get_property("dim").to_int());
116  }
117  }
118 
119  procCom.broadcast(buf, rootProc);
120 
121  if(pcl::ProcRank() != rootProc){
122  int numSubsets;
123  Deserialize(buf, numSubsets);
124  if(numSubsets > 0)
125  sh.subset_required(numSubsets - 1);
126 
127  for(int i = 0; i < numSubsets; ++i){
128  Deserialize(buf, sh.subset_info(i).name);
129  int dim;
130  Deserialize(buf, dim);
131  sh.subset_info(i).set_property("dim", dim);
132  }
133  }
134 #endif
135 
136 }

189 template <typename TGrid, typename TSubsetHandler>
190 void IDomain<TGrid,TSubsetHandler>::
191 update_domain_info()
192 {
193  PROFILE_FUNC();
194 
195  TGrid& mg = *m_spGrid;
196  TSubsetHandler& sh = *m_spSH;
197 
198  GridBaseObjectId locElemType;
199  if(mg.template num<Volume>() > 0)
200  locElemType = VOLUME;
201  else if(mg.template num<Face>() > 0)
202  locElemType = FACE;
203  else if(mg.template num<Edge>() > 0)
204  locElemType = EDGE;
205  else
206  locElemType = VERTEX;
207 
208  GridBaseObjectId elemType;
209  #ifdef UG_PARALLEL
210  pcl::ProcessCommunicator commWorld;
211 
212  // all processes should have the same number of grid levels
213  int numLevLocal = m_spGrid->num_levels();
214  int numLevGlobal;
215  commWorld.allreduce(&numLevLocal, &numLevGlobal, 1, PCL_DT_INT, PCL_RO_MAX);
216 
217  if(numLevGlobal > 0){
218  m_spGrid->level_required(numLevGlobal-1);
219  m_spSH->level_required(numLevGlobal-1);
220  }
221 
222  // communicate the element type of highest dimension present in the global grid.
223  elemType = (GridBaseObjectId) commWorld.allreduce((int)locElemType, PCL_RO_MAX);
224  #else
225  elemType = locElemType;
226  #endif
227 
228 // the number of levels of the multi-grid is now equal on all processes
229 
230  std::vector<int> subsetDims;
231  subsetDims.reserve(sh.num_subsets());
232  for(int i = 0; i < sh.num_subsets(); ++i)
233  subsetDims.push_back(sh.subset_info(i).get_property("dim").to_int());
234 
235  std::vector<int> numLocalGhosts;
236  #ifdef UG_PARALLEL
237  switch(elemType){
238  case VOLUME:
239  count_ghosts<Volume>(numLocalGhosts);
240  break;
241  case FACE:
242  count_ghosts<Face>(numLocalGhosts);
243  break;
244  case EDGE:
245  count_ghosts<Edge>(numLocalGhosts);
246  break;
247  case VERTEX:
248  count_ghosts<Vertex>(numLocalGhosts);
249  break;
250  default:
251  UG_THROW("Unknown base object type");
252  break;
253  }
254  #else
255  numLocalGhosts.resize(mg.num_levels(), 0);
256  #endif
257 
258  std::vector<int> numLocalElems;
259  numLocalElems.reserve(mg.num_levels());
260 
261  switch(elemType){
262  case VOLUME:
263  for(size_t i = 0; i < mg.num_levels(); ++i)
264  numLocalElems.push_back(mg.template num<Volume>(i) - numLocalGhosts[i]);
265  break;
266  case FACE:
267  for(size_t i = 0; i < mg.num_levels(); ++i)
268  numLocalElems.push_back(mg.template num<Face>(i) - numLocalGhosts[i]);
269  break;
270  case EDGE:
271  for(size_t i = 0; i < mg.num_levels(); ++i)
272  numLocalElems.push_back(mg.template num<Edge>(i) - numLocalGhosts[i]);
273  break;
274  case VERTEX:
275  for(size_t i = 0; i < mg.num_levels(); ++i)
276  numLocalElems.push_back(mg.template num<Vertex>(i) - numLocalGhosts[i]);
277  break;
278  default:
279  UG_THROW("Unknown base object type");
280  break;
281  }
282 
283  std::vector<int> numGlobalElems, minNumLocalElems, maxNumLocalElems;
284  #ifdef UG_PARALLEL
285  // we have to sum local element counts excluding ghosts.
286  numGlobalElems.resize(numLocalElems.size());
287  minNumLocalElems.resize(numLocalElems.size());
288  maxNumLocalElems.resize(numLocalElems.size());
289  commWorld.allreduce(numLocalElems, numGlobalElems, PCL_RO_SUM);
290  commWorld.allreduce(numLocalElems, minNumLocalElems, PCL_RO_MIN);
291  commWorld.allreduce(numLocalElems, maxNumLocalElems, PCL_RO_MAX);
292  #else
293  numGlobalElems = numLocalElems;
294  minNumLocalElems = numLocalElems;
295  maxNumLocalElems = numLocalElems;
296  #endif
297 
298  m_domainInfo.set_info(elemType, numGlobalElems, numLocalElems, minNumLocalElems,
299  maxNumLocalElems, numLocalGhosts, subsetDims);
300 }

352 template <typename TGrid, typename TSubsetHandler>
353 template <class TElem>
354 void IDomain<TGrid,TSubsetHandler>::
355 count_ghosts(std::vector<int>& numGhostsOnLvlOut)
356 {
357  TGrid& mg = *m_spGrid;
358  DistributedGridManager& dgm = *mg.distributed_grid_manager();
359  GridLayoutMap& glm = dgm.grid_layout_map();
360 
361  numGhostsOnLvlOut.clear();
362  numGhostsOnLvlOut.resize(mg.num_levels(), 0);
363 
364  if(glm.has_layout<TElem>(INT_V_MASTER)){
365  typedef typename GridLayoutMap::Types<TElem>::Layout Layout;
366  typedef typename GridLayoutMap::Types<TElem>::Interface Interface;
367  Layout& layout = glm.get_layout<TElem>(INT_V_MASTER);
368  for(size_t lvl = 0; lvl < layout.num_levels(); ++lvl){
369  if(lvl >= mg.num_levels())
370  break;
371 
372  typename Layout::LevelLayout& lvlLayout = layout.layout_on_level(lvl);
373  for(typename Layout::LevelLayout::iterator iiter = lvlLayout.begin();
374  iiter != lvlLayout.end(); ++iiter)
375  {
376  Interface& intfc = lvlLayout.interface(iiter);
377  for(typename Interface::iterator eiter = intfc.begin();
378  eiter != intfc.end(); ++eiter)
379  {
380  if(dgm.is_ghost(intfc.get_element(eiter)))
381  ++numGhostsOnLvlOut[lvl];
382  }
383  }
384  }
385  }
386 }
387 #endif

393 template <int d, typename TGrid, typename TSubsetHandler>
394 Domain<d,TGrid,TSubsetHandler>::
395 Domain(bool isAdaptive) : IDomain<TGrid, TSubsetHandler>(isAdaptive)
396 {
397 // Depending on the dimesion, we'll activeate different options.
398 // Otherwise we probably would waste memory...
399 // In any case, sides of elements should always be present
400  uint gridOpts = GRIDOPT_AUTOGENERATE_SIDES;
401 
402 // Furthermore vertices should store associated elements.
403 // This option depends on the dimension of the domain
404  if(dim > 0)
405  gridOpts |= VRTOPT_STORE_ASSOCIATED_EDGES;
406  if(dim > 1)
407  gridOpts |= VRTOPT_STORE_ASSOCIATED_FACES;
408  if(dim > 2)
409  gridOpts |= VRTOPT_STORE_ASSOCIATED_VOLUMES;
410 
411 // thats it for now. One could think about enabling
412 // FACEOPT_STORE_ASSOCIATED_EDGES, VOLOPT_STORE_ASSOCIATED_EDGES
413 // and VOLOPT_STORE_ASSOCIATED_FACES. However this costs considerably
414 // more memory compared to the performance benefits.
415 // Now set the options
416  this->grid()->set_options(gridOpts);
417 
418 // get position attachment
419  m_aPos = GetDefaultPositionAttachment<position_attachment_type>();
420 
421 // let position accessor access Vertex Coordinates
422  if(!this->grid()->template has_attachment<Vertex>(m_aPos))
423  this->grid()->template attach_to<Vertex>(m_aPos);
424  m_aaPos.access(*(this->grid()), m_aPos);
425 }
