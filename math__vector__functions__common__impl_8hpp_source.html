372 template <typename vector_t>
373 inline
374 typename vector_t::value_type
375 VecAngle(const vector_t& v1, const vector_t& v2)
376 {
377  typedef typename vector_t::value_type value_t;
378 
379  value_t l = sqrt(VecLengthSq(v1) * VecLengthSq(v2));
380  if(l > 0){
381  number a = VecDot(v1, v2) / l;
382  if(a >= 1)
383  return 0;
384  else if(a <= -1)
385  return PI;
386  return acos(a);
387  }
388 
389  return 0;
390 }

408 template <typename vector_t>
409 inline
410 void
411 VecCross(vector_t& vOut, const vector_t& v1, const vector_t& v2)
412 {
413  if(&vOut != &v1 && &vOut != &v2)
414  {
415  vOut[0] = v1[1] * v2[2] - v2[1] * v1[2];
416  vOut[1] = v1[2] * v2[0] - v2[2] * v1[0];
417  vOut[2] = v1[0] * v2[1] - v2[0] * v1[1];
418  }
419  else
420  {
421  vector_t _temp;
422  _temp[0] = v1[1] * v2[2] - v2[1] * v1[2];
423  _temp[1] = v1[2] * v2[0] - v2[2] * v1[0];
424  _temp[2] = v1[0] * v2[1] - v2[0] * v1[1];
425  vOut = _temp;
426  }
427 }
