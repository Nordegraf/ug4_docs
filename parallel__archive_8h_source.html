73 template<typename TBuffer>
74 void WriteParallelArchive(pcl::ProcessCommunicator &pc, std::string strFilename, const std::map<std::string, TBuffer> &files)
75 {
76  std::vector<FileBufferDescriptor> fdesc;
77  for(typename std::map<std::string, TBuffer>::const_iterator it = files.begin(); it != files.end(); ++it)
78  files.push_back(FileBufferDescriptor(ug::FilenameWithoutPath(it->first), it->second));
79  WriteParallelArchive(pc, strFilename, fdesc);
80 }

219  void add_raw(FileBufferDescriptor f)
220  {
221  files[f.name] = make_sp(new ConstCharBuffer(f.buf, f.size));
222  m_bUnsafe = true;
223  }
224 
229  void write()
230  {
231  std::vector<FileBufferDescriptor> fdesc;
232  for(map_iterator it = files.begin(); it != files.end(); ++it)
233  {
234  (it->second)->update();
235  fdesc.push_back(
236  FileBufferDescriptor(ug::FilenameWithoutPath(it->first),
237  (it->second)->buffer() , (it->second)->size() ) );
238  }
239  WriteParallelArchive(m_pc, m_filename, fdesc);
240  files.clear();
241  m_bWritten = true;
242  }
243 
244  void create_new_archive(std::string filename)
245  {
